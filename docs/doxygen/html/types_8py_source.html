164# ==================== NAT 类型 ====================
165
166class NATType(Enum):
167 """
168 NAT类型枚举 (RFC 3489)
169
170 根据RFC 3489标准定义的NAT类型，用于判断NAT穿透的可行性。
171
172 属性:
173 FULL_CONE: 完全圆锥型NAT
174 - 所有来自同一内部IP和端口的请求都映射到同一外部IP和端口
175 - 任何外部主机都可以通过该外部地址向内部主机发送数据包
176 - 穿透难度：最低
177
178 RESTRICTED_CONE: 受限圆锥型NAT
179 - 所有来自同一内部IP和端口的请求都映射到同一外部IP和端口
180 - 外部主机只有在内部主机先向其发送过数据包后，才能向内部主机发送数据包
181 - 穿透难度：中等
182
183 PORT_RESTRICTED: 端口受限圆锥型NAT
184 - 所有来自同一内部IP和端口的请求都映射到同一外部IP和端口
185 - 外部主机只有在内部主机先向其特定端口发送过数据包后，才能向内部主机发送数据包
186 - 穿透难度：中等偏高
187
188 SYMMETRIC: 对称型NAT
189 - 每个新的会话都会创建一个新的外部IP和端口映射
190 - 穿透难度：最高，通常需要中继服务器
191
192 UNKNOWN: 未知NAT类型
193 """
194 FULL_CONE = "full_cone"
195 RESTRICTED_CONE = "restricted_cone"
196 PORT_RESTRICTED = "port_restricted"
197 SYMMETRIC = "symmetric"
198 UNKNOWN = "unknown"

203@dataclass
204class NATInfo:
205 """
206 NAT信息数据类
207
208 存储NAT检测结果和相关的网络信息，用于连接策略决策。
209
210 属性:
211 type (NATType): NAT类型
212 public_ip (str): 公网IP地址
213 public_port (int): 公网端口
214 local_ip (str): 本地IP地址
215 local_port (int): 本地端口
216 is_cgnat (bool): 是否为CGNAT（运营商级NAT）
217 cgnat_level (int): CGNAT层级（0表示非CGNAT）
218 hairpin_supported (bool): 是否支持发夹效应（Hairpin）
219 port_delta (int): 端口增量（用于对称NAT端口预测）
220 mapping_timeout_sec (int): 映射超时时间（秒）
221 device_vendor (DeviceVendor): NAT设备厂商
222
223 方法:
224 is_symmetric(): 判断是否为对称型NAT
225 can_predict_port(): 判断是否可以预测端口
226 """
227 type: NATType = NATType.UNKNOWN
228
229 # 地址信息
230 public_ip: str = ""
231 public_port: int = 0
232 local_ip: str = ""
233 local_port: int = 0
234
235 # 增强信息
236 is_cgnat: bool = False
237 cgnat_level: int = 0
238 hairpin_supported: bool = False
239 port_delta: int = 1
240 mapping_timeout_sec: int = 30
241
242 # 新增：设备厂商
243 device_vendor: DeviceVendor = DeviceVendor.UNKNOWN
244
245 def is_symmetric(self) -> bool:
246 return self.type == NATType.SYMMETRIC
247
248 def can_predict_port(self) -> bool:
249 return self.is_symmetric() and self.port_delta > 0

255class NetworkEnvironment:
256 """
257 网络环境信息数据类

261 属性:

263 is_behind_vpn (bool): 是否在VPN后
264 is_behind_cdn (bool): 是否在CDN后
265 is_enterprise_network (bool): 是否为企业网络
266 is_mobile_network (bool): 是否为移动网络（4G/5G）
267 firewall_type (str): 防火墙类型
268 - "home": 家用防火墙
269 - "enterprise": 企业防火墙
270 - "isp": ISP级防火墙
271 - "cross_border": 跨境防火墙
272 - "unknown": 未知
273 ipv6_available (bool): IPv6是否可用
274 ipv6_preferred (bool): 是否优先使用IPv6
275 packet_loss_rate (float): 丢包率（0.0-1.0）
276 avg_latency_ms (float): 平均延迟（毫秒）
277
278 方法:
279 is_complex_environment(): 判断是否为复杂网络环境
280 """
281 # NAT 层级
282 nat_level: int = 1 # NAT 层数
283
284 # 特殊环境
285 is_behind_vpn: bool = False # 是否在 VPN 后
286 is_behind_cdn: bool = False # 是否在 CDN 后
287 is_enterprise_network: bool = False # 是否企业网络
288 is_mobile_network: bool = False # 是否移动网络(4G/5G)
289
290 # 防火墙类型
291 firewall_type: str = "unknown" # home | enterprise | isp | cross_border
292
293 # IPv6
294 ipv6_available: bool = False
295 ipv6_preferred: bool = False
296
297 # 链路质量
298 packet_loss_rate: float = 0.0 # 丢包率
299 avg_latency_ms: float = 0.0 # 平均延迟
300
301 def is_complex_environment(self) -> bool:
302 """是否为复杂网络环境"""
303 return (
304 self.nat_level >= 3 or
305 self.is_behind_vpn or
306 self.is_enterprise_network or
307 self.packet_loss_rate > 0.1
308 )

328 DISCONNECTED: 已断开，连接已正常断开
329 FAILED: 失败，连接建立失败

335 CONNECTING = "connecting"
336 CONNECTED = "connected"
