80class P2PEngine:
81 """
82 P2P引擎主类
83
84 核心P2P连接引擎，整合所有子模块实现完整的P2P连接流程。
85 采用状态机驱动的架构，支持NAT穿透、中继降级等功能。
86
87 主要功能:
88 - 网络检测：ISP识别、NAT类型检测
89 - NAT穿透：UDP打孔、TCP连接
90 - 信令交换：与对端交换连接信息
91 - 连接管理：建立、维护、断开连接
92 - 心跳保活：保持连接活跃
93 - 降级策略：连接失败时自动降级到中继
94
95 使用示例:
96 ```python
97 engine = P2PEngine()
98 await engine.initialize()
99 result = await engine.connect_to_peer(peer_id, peer_info)
100 if result.success:
101 print(f"连接成功: {result.connection_type}")
102 ```
103
104 属性:
105 _config (P2PConfig): 引擎配置
106 _state (P2PState): 当前引擎状态
107 _event_queue (asyncio.Queue): 事件队列
108 _on_state_change (StateCallback): 状态变化回调
109 _on_error (ErrorCallback): 错误回调
110 _config_loader (ConfigLoader): 配置加载器
111 _isp_detector (ISPDetector): ISP检测器
112 _nat_detector (NATDetector): NAT检测器
113 _puncher (UDPPuncher): UDP打孔器
114 _fallback_decider (FallbackDecider): 降级决策器
115 _heartbeat (HeartbeatKeeper): 心跳保活器
116 _main_task (asyncio.Task): 主循环任务
117 _running (bool): 引擎是否运行中

180 async def connect(self, peer_info: PeerInfo) -> ConnectionResult:
181 """发起连接"""
182 if self._state.state != ConnectionState.IDLE:
183 raise RuntimeError(f"引擎状态错误: {self._state.state}")
184
185 self._state.peer_info = peer_info
186 await self._event_queue.put(Event(EventType.CONNECT, {"peer_info": peer_info}))
187
188 # 启动主循环
189 if not self._main_task:
190 self._main_task = asyncio.create_task(self._main_loop())
191
192 # 等待连接结果
193 while self._state.state not in (ConnectionState.CONNECTED, ConnectionState.FAILED):
194 await asyncio.sleep(0.1)
195
196 return self._state.connection_result or ConnectionResult(
197 success=False,
198 connection_type=ConnectionType.FAILED,
199 error="连接超时",
200 )
