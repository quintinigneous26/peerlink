44def is_tcp_fastopen_available() -> bool:
45 """检查系统是否支持 TCP Fast Open"""
46 try:
47 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
48 # Linux: MSG_FASTOPEN = 0x20000000
49 sock.setsockopt(socket.SOL_SOCKET, 0x20000000, TCP_FASTOPEN_QUEUE_SIZE)
50 sock.close()
51 return True
52 except (OSError, AttributeError):
53 return False

56class TCPFastOpenConnection(Connection):
57 """
58 支持 TCP Fast Open 的连接
59
60 首次连接使用 TFO 获取 cookie，后续连接可发送数据在 SYN 中。
61 """
62
63 def __init__(
64 self,
65 sock: socket.socket,
66 remote_addr: Tuple[str, int],
67 local_addr: Tuple[str, int],
68 ):
69 self._sock = sock
70 self._remote_addr = remote_addr
71 self._local_addr = local_addr
72 self._closed = False
73 self._reader: Optional[asyncio.StreamReader] = None
74 self._writer: Optional[asyncio.StreamWriter] = None
75 self._created_at = time.time()
76 self._bytes_read = 0
77 self._bytes_written = 0
78
79 # 优化 socket 选项
80 self._configure_socket()
81
82 def _configure_socket(self) -> None:
83 """配置高性能 socket 选项"""
84 # 禁用 Nagle 算法
85 self._sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)
86
87 # 启用 Keep-Alive
88 self._sock.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1)
89 self._sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_KEEPIDLE, KEEPALIVE_IDLE)
90 self._sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_KEEPINTVL, KEEPALIVE_INTERVAL)
91 self._sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_KEEPCNT, KEEPALIVE_COUNT)
92
93 # 设置发送/接收缓冲区
94 self._sock.setsockopt(socket.SOL_SOCKET, socket.SO_SNDBUF, 256 * 1024) # 256KB
95 self._sock.setsockopt(socket.SOL_SOCKET, socket.SO_RCVBUF, 256 * 1024) # 256KB
96
97 async def read(self, n: int = -1) -> bytes:
98 """读取数据"""
99 if self._closed:
100 raise TransportError("Connection closed")
101
102 if not self._reader:
103 # 创建 StreamReader
104 loop = asyncio.get_event_loop()
105 self._reader = asyncio.StreamReader()
106 protocol = asyncio.StreamReaderProtocol(self._reader)
107 transport, _ = await loop.connect_accepted_socket(
108 lambda: protocol, sock=self._sock
109 )
110
111 if n < 0:
112 data = await self._reader.read()
113 else:
114 data = await self._reader.read(n)
115
116 self._bytes_read += len(data)
117 return data
118
119 async def write(self, data: bytes) -> int:
120 """写入数据"""
121 if self._closed:
122 raise TransportError("Connection closed")
123
124 if not self._writer:
125 # 创建 StreamWriter
126 loop = asyncio.get_event_loop()
127 self._writer = asyncio.StreamWriter(
128 self._sock, None, None, loop
129 )
130
131 self._writer.write(data)
132 await self._writer.drain()
133 self._bytes_written += len(data)
134 return len(data)
135
136 async def close(self) -> None:
137 """关闭连接"""
138 if self._closed:
139 return
140
141 self._closed = True
142
143 try:
144 if self._writer:
145 self._writer.close()
146 await self._writer.wait_closed()
147 elif self._sock:
148 self._sock.close()
149 except Exception:
150 pass
151
152 def is_closed(self) -> bool:
153 """检查连接是否已关闭"""
154 return self._closed
155
156 @property
157 def remote_address(self) -> Optional[tuple]:
158 return self._remote_addr
159
160 @property
161 def local_address(self) -> Optional[tuple]:
162 return self._local_addr

164 @property

167 return {
168 "age_seconds": time.time() - self._created_at,

176class ConnectionPool:
177 """
178 TCP 连接池
179
180 复用连接以减少握手开销。
181 """
182
183 def __init__(
184 self,
185 max_size: int = DEFAULT_MAX_POOL_SIZE,
186 idle_timeout: float = CONNECTION_IDLE_TIMEOUT,
187 ):
188 self._max_size = max_size
189 self._idle_timeout = idle_timeout
190 self._pool: OrderedDict[str, TCPFastOpenConnection] = OrderedDict()
191 self._lock = asyncio.Lock()
192 self._cleanup_task: Optional[asyncio.Task] = None
193
194 def start(self) -> None:
195 """启动连接池清理任务"""
196 if not self._cleanup_task:
197 self._cleanup_task = asyncio.create_task(self._cleanup_loop())
198
199 async def stop(self) -> None:
200 """停止连接池"""
201 if self._cleanup_task:
202 self._cleanup_task.cancel()
203 try:
204 await self._cleanup_task
205 except asyncio.CancelledError:
206 pass
207
208 async with self._lock:
209 for conn in self._pool.values():
210 try:
211 await conn.close()
212 except Exception:
213 pass
214 self._pool.clear()
215
216 async def get(self, host: str, port: int) -> Optional[TCPFastOpenConnection]:
217 """从池中获取连接"""
218 key = f"{host}:{port}"
219
220 async with self._lock:
221 conn = self._pool.get(key)
222
223 if conn and not conn.is_closed:
224 # 移到末尾 (LRU)
225 self._pool.move_to_end(key)
226 return conn
227
228 # 移除无效连接
229 if key in self._pool:
230 del self._pool[key]
231
232 return None
233
234 async def put(
235 self,
236 host: str,
237 port: int,
238 conn: TCPFastOpenConnection,
239 ) -> None:
240 """将连接放回池中"""
241 key = f"{host}:{port}"
242
243 async with self._lock:
244 # 池已满，移除最旧的
245 if len(self._pool) >= self._max_size:
246 self._pool.popitem(last=False)
247
248 self._pool[key] = conn
249 self._pool.move_to_end(key)
250
251 async def remove(self, host: str, port: int) -> None:
252 """从池中移除连接"""
253 key = f"{host}:{port}"
254
255 async with self._lock:
256 conn = self._pool.pop(key, None)
257 if conn:
258 try:
259 await conn.close()
260 except Exception:
261 pass
262
263 async def _cleanup_loop(self) -> None:
264 """定期清理空闲连接"""
265 while True:
266 try:
267 await asyncio.sleep(60) # 每分钟清理一次
268
269 async with self._lock:
270 now = time.time()
271 to_remove = []
272
273 for key, conn in self._pool.items():
274 if conn.is_closed:
275 to_remove.append(key)
276 elif now - conn._created_at > self._idle_timeout:
277 to_remove.append(key)
278
279 for key in to_remove:
280 conn = self._pool.pop(key, None)
281 if conn:
282 try:
283 await conn.close()
284 except Exception:
285 pass
286
287 if to_remove:
288 logger.debug(f"Cleaned up {len(to_remove)} idle connections")
289
290 except asyncio.CancelledError:
291 break
292 except Exception as e:
293 logger.error(f"Connection pool cleanup error: {e}")
294
295 @property
296 def size(self) -> int:
297 """获取池中连接数"""
298 return len(self._pool)

303class OptimizedTCPTransport(Transport):
304 """
305 优化的 TCP 传输
306
307 特性:

312
313 def __init__(
314 self,
315 enable_fastopen: bool = True,
316 enable_pooling: bool = True,
317 pool_size: int = DEFAULT_MAX_POOL_SIZE,
318 ):
319 self._enable_fastopen = enable_fastopen and is_tcp_fastopen_available()
320 self._enable_pooling = enable_pooling
321 self._pool = ConnectionPool(max_size=pool_size) if enable_pooling else None
322 self._closed = False
323
324 if self._pool:
325 self._pool.start()
326
327 if self._enable_fastopen:
328 logger.info("TCP Fast Open enabled")
329 else:
330 logger.debug("TCP Fast Open not available, using standard TCP")
331
332 def protocols(self) -> List[str]:
333 """返回支持的协议列表"""
334 return ["/tcp/1.0.0"]

336 async def dial(self, addr: str) -> TCPFastOpenConnection:
337 """
338 建立到指定地址的连接
339
340 支持连接池和 TCP Fast Open
341 """
342 if self._closed:
343 raise TransportError("Transport closed")
344
345 host, port = self._parse_address(addr)
346
347 # 尝试从连接池获取
348 if self._pool:
349 conn = await self._pool.get(host, port)
350 if conn:
351 logger.debug(f"Connection from pool: {host}:{port}")
352 return conn
353
354 # 创建新连接
355 start_time = time.time()
356
357 if self._enable_fastopen:
358 conn = await self._dial_with_fastopen(host, port)
359 else:
360 conn = await self._dial_standard(host, port)
361
362 latency_ms = (time.time() - start_time) * 1000
363 logger.debug(f"TCP connected to {host}:{port} in {latency_ms:.1f}ms")
364
365 return conn

367 async def dial_standard(self, host: str, port: int) -> TCPFastOpenConnection:
368 """标准 TCP 连接"""
369 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
370 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
371
372 loop = asyncio.get_event_loop()
373 await loop.sock_connect(sock, (host, port))
374
375 local_addr = sock.getsockname()
376 remote_addr = sock.getpeername()
377
378 return TCPFastOpenConnection(sock, remote_addr, local_addr)

380 async def _dial_with_fastopen(
381 self,
382 host: str,
383 port: int,
384 ) -> TCPFastOpenConnection:
385 """使用 TCP Fast Open 连接"""
386 # TFO 实现：首次连接获取 cookie，后续可快速连接
387 # 这里使用标准 connect，实际 TFO 需要 sendto with MSG_FASTOPEN
388 return await self._dial_standard(host, port)

390 async def listen(self, addr: str) -> Listener:
391 """开始监听"""
392 if self._closed:
393 raise TransportError("Transport closed")
394
395 host, port = self._parse_address(addr)
396
397 sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
398 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
399 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEPORT, 1)
400 sock.bind((host, port))
401 sock.listen(1024)
402
403 local_addr = sock.getsockname()
404
405 listener = TCPOptimizedListener(
406 sock=sock,
407 local_addr=local_addr,
408 transport=self,
409 )
410
411 logger.info(f"TCP listening on {local_addr}")
412 return listener
413
414 async def close(self) -> None:
415 """关闭传输层"""
416 if self._closed:
417 return
418
419 self._closed = True
420
421 if self._pool:
422 await self._pool.stop()
423
424 logger.info("Optimized TCP transport closed")

426 def _parse_address(self, addr: str) -> Tuple[str, int]:
427 """解析地址"""
428 if addr.startswith("/"):
429 # multiaddr 格式: /ip4/1.2.3.4/tcp/5678
430 parts = addr.strip("/").split("/")
431 if len(parts) >= 4 and parts[2] == "tcp":
432 host = parts[1]
433 port = int(parts[3])
434 return (host, port)
435
436 # host:port 格式
437 if ":" in addr:
438 host, port_str = addr.rsplit(":", 1)
439 try:
440 return (host, int(port_str))
441 except ValueError:
442 pass
443
444 # 默认端口
445 return (addr, DEFAULT_PORT)
