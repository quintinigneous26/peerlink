84class CertificateManager:
85 """
86 Manages self-signed certificates for WebTransport
87
88 Per libp2p spec:
89 - Certificates are self-signed
90 - Validity period ≤ 14 days
91 - Cannot use RSA keys
92 - Certificate hashes are advertised in multiaddr
93 """
94
95 def __init__(self):
96 self._certificates: Dict[str, CertificateInfo] = {}
97 self._primary_cert_hash: Optional[str] = None
98
99 def generate_certificate(self, validity_days: int = MAX_CERT_VALIDITY_DAYS) -> CertificateInfo:
100 """
101 Generate a new self-signed certificate
102
103 Args:
104 validity_days: Validity period in days (max 14)
105
106 Returns:
107 CertificateInfo with certificate details
108 """
109 if validity_days > MAX_CERT_VALIDITY_DAYS:
110 validity_days = MAX_CERT_VALIDITY_DAYS
111
112 try:
113 from cryptography import x509
114 from cryptography.x509.oid import NameOID
115 from cryptography.hazmat.primitives import hashes, serialization
116 from cryptography.hazmat.primitives.asymmetric import ec
117 import cryptography.hazmat.backends as backends
118
119 # Generate EC private key (not RSA as per spec)
120 private_key = ec.generate_private_key(ec.SECP256R1(), backends.default_backend())
121
122 # Build certificate subject
123 subject = issuer = x509.Name([
124 x509.NameAttribute(NameOID.COUNTRY_NAME, u"US"),
125 x509.NameAttribute(NameOID.STATE_OR_PROVINCE_NAME, u"CA"),
126 x509.NameAttribute(NameOID.LOCALITY_NAME, u"San Francisco"),
127 x509.NameAttribute(NameOID.ORGANIZATION_NAME, u"libp2p"),
128 x509.NameAttribute(NameOID.COMMON_NAME, u"libp2p-node"),
129 ])
130
131 # Calculate validity dates
132 now = datetime.utcnow()
133 not_valid_before = now
134 not_valid_after = now + timedelta(days=validity_days)
135
136 # Build certificate
137 cert = x509.CertificateBuilder().subject_name(
138 subject
139 ).issuer_name(
140 issuer
141 ).public_key(
142 private_key.public_key()
143 ).serial_number(
144 x509.random_serial_number()
145 ).not_valid_before(
146 not_valid_before
147 ).not_valid_after(
148 not_valid_after
149 ).sign(private_key, hashes.SHA256(), backends.default_backend())
150
151 # Serialize for use with SSL
152 cert_pem = cert.public_bytes(serialization.Encoding.PEM)
153 key_pem = private_key.private_bytes(
154 encoding=serialization.Encoding.PEM,
155 format=serialization.PrivateFormat.TraditionalOpenSSL,
156 encryption_algorithm=serialization.NoEncryption()
157 )
158
159 # Calculate certificate hash (SHA-256)
160 cert_der = cert.public_bytes(serialization.Encoding.DER)
161 cert_hash = hashlib.sha256(cert_der).digest()
162 cert_hash_b64 = cert_hash.hex()
163
164 # Create SSL context
165 context = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
166 context.load_cert_chain(certfile=cert_pem, keyfile=key_pem)
167
168 cert_info = CertificateInfo(
169 certificate=cert_pem,
170 certificate_hash=cert_hash_b64,
171 not_valid_after=not_valid_after,
172 not_valid_before=not_valid_before,
173 )
174
175 # Store certificate
176 self._certificates[cert_hash_b64] = cert_info
177
178 # Set as primary if first cert
179 if self._primary_cert_hash is None:
180 self._primary_cert_hash = cert_hash_b64
181
182 logger.info(f"Generated new certificate, hash: {cert_hash_b64[:16]}...")
183 return cert_info
184
185 except ImportError:
186 logger.error("cryptography library not available, cannot generate certificates")
187 raise TransportError("Certificate generation requires 'cryptography' library")
188
189 def get_primary_certificate_hash(self) -> Optional[str]:
190 """Get the primary certificate hash"""
191 return self._primary_cert_hash
192
193 def get_all_certificate_hashes(self) -> List[str]:
194 """Get all certificate hashes for advertisement"""
195 return list(self._certificates.keys())
196
197 def get_certificate(self, cert_hash: str) -> Optional[CertificateInfo]:
198 """Get certificate by hash"""
199 return self._certificates.get(cert_hash)
200
201 def validate_certificate_hashes(self, expected_hashes: List[str]) -> bool:
202 """
203 Validate that primary certificate hash is in expected list
204
205 Args:
206 expected_hashes: List of expected certificate hashes
207
208 Returns:
209 True if primary hash is in expected list
210 """
211 if self._primary_cert_hash is None:
212 return False
213
214 return self._primary_cert_hash in expected_hashes

429 async def dial(self, addr: str) -> Connection:
430 """
431 Establish WebTransport connection to remote peer
432
433 Args:
434 addr: Multiaddr (e.g., /ip4/192.0.2.0/udp/443/quic/webtransport/...)
435
436 Returns:
437 Established connection
438
439 Raises:
440 ConnectionError: If connection fails
441 """
442 if self._closed:
443 raise TransportError("Transport is closed")
444
445 # Parse multiaddr
446 parsed = self._parse_multiaddr(addr)
447 if parsed is None:
448 raise ConnectionError(f"Invalid multiaddr: {addr}")
449
450 host, port, cert_hashes = parsed
451
452 # Create client configuration
453 config = QuicConfiguration(
454 alpn_protocols=ALPN_PROTOCOLS,
455 max_data=DEFAULT_MAX_DATA,
456 max_stream_data=DEFAULT_MAX_STREAM_DATA,

458 verify_mode=ssl.CERT_NONE, # Will verify via cert hash

481 )

544 @staticmethod
545 def _parse_multiaddr(addr: str) -> Optional[tuple]:
546 """
547 Parse WebTransport multiaddr
548
549 Supports formats:
550 - /ip4/<ip>/udp/<port>/quic/webtransport
551 - /ip4/<ip>/udp/<port>/quic/webtransport/certhash/<hash1>/certhash/<hash2>
552
553 Returns:
554 Tuple of (host, port, cert_hashes) or None
555 """
556 try:
557 if not addr.startswith('/'):
558 return None
559
560 parts = addr[1:].split('/')
561 if len(parts) < 6:
562 return None
563
564 proto = parts[0] # ip4 or ip6
565 host = parts[1]
566 transport = parts[2] # udp
567 port = int(parts[3]) if len(parts) > 3 else 443
568
569 if parts[4] != "quic":
570 return None
571
572 if len(parts) > 5 and parts[5] == "webtransport":
573 # Extract certificate hashes
574 cert_hashes = []
575 i = 6
576 while i + 1 < len(parts):
577 if parts[i] == "certhash":
578 cert_hashes.append(parts[i + 1])
579 i += 2
580 else:
581 break
582
583 return (host, port, cert_hashes)
584
585 return (host, port, [])
586
587 except (ValueError, IndexError):
588 return None
