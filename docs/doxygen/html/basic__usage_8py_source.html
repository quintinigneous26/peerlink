1"""
2Basic usage example for P2P Client SDK.

4This example demonstrates how to use the P2P SDK to:

12import asyncio

14
15from p2p_sdk import P2PClient, P2PConfig, NATType, ChannelType

17
18# Configure logging
19logging.basicConfig(level=logging.INFO)
20logger = logging.getLogger(__name__)
21
22
23async def main():
24 """Main example function."""
25
26 # Configure the client
27 config = P2PConfig(
28 signaling_server="localhost",
29 signaling_port=8443,
30 stun_server="stun.l.google.com",
31 stun_port=19302,
32 relay_server="localhost",
33 relay_port=5000,
34 connection_timeout=30.0,
35 auto_relay=True,
36 )
37
38 # Create client with device ID
39 client = P2PClient(did="device-001", config=config)
40
41 # Setup event handlers
42 @client.on_connected
43 async def on_connected():
44 logger.info("Connection established!")
45
46 @client.on_disconnected
47 async def on_disconnected():
48 logger.info("Connection closed")
49
50 @client.on_data
51 async def on_data(channel_id: int, data: bytes):
52 logger.info(f"Received {len(data)} bytes on channel {channel_id}")
53
54 @client.on_error
55 async def on_error(error: Exception):
56 logger.error(f"Error: {error}")
57
58 try:
59 # Initialize the client
60 await client.initialize()
61
62 # Detect NAT type
63 nat_type = await client.detect_nat()
64 logger.info(f"NAT Type: {nat_type.value}")
65
66 # Create a data channel
67 data_channel = client.create_channel(
68 channel_type=ChannelType.DATA,
69 reliable=True,
70 priority=1,
71 )
72 logger.info(f"Created data channel: {data_channel}")
73
74 # Connect to peer
75 peer_did = "device-002"

77
78 if success:
79 logger.info(f"Connected to {peer_did}")
80 logger.info(f"Connection type: {'P2P' if client.is_p2p else 'Relay'}")
81
82 # Send data
83 await client.send_data(data_channel, b"Hello, peer!")
84 logger.info("Sent data")
85
86 # Receive data
87 try:
88 data = await asyncio.wait_for(
89 client.recv_data(data_channel),
90 timeout=5.0,
91 )
92 logger.info(f"Received: {data.decode()}")
93 except asyncio.TimeoutError:
94 logger.info("No data received within timeout")
95
96 # Keep connection alive for a bit
97 await asyncio.sleep(10)
98
99 else:
100 logger.error(f"Failed to connect to {peer_did}")
101
102 except Exception as e:
103 logger.error(f"Error: {e}")
104
105 finally:
106 # Cleanup
107 await client.close()
108
109
110if __name__ == "__main__":
111 asyncio.run(main())
