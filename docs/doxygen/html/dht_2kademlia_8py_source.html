56@dataclass
57class KademliaMessage:
58 """
59 Kademlia DHT message.
60
61 Attributes:
62 message_type: Type of message
63 key: Target key (peer ID, content key, or CID)
64 value: Value (for PUT_VALUE responses)
65 providers: List of providers (for GET_PROVIDERS)
66 peers: List of closer peers (for all responses)
67 peer_id: Sender's peer ID
68 record: Record-level data (optional)
69 """
70 message_type: KademliaMessageType
71 key: Optional[bytes] = None
72 value: Optional[bytes] = None
73 providers: List[Dict[str, Any]] = field(default_factory=list)
74 peers: List[Dict[str, bytes]] = field(default_factory=list)
75 peer_id: Optional[bytes] = None
76 record: Optional[Dict[str, Any]] = None
77 cluster_level: int = 0 # For DHT crawling
78
79 def to_dict(self) -> dict:
80 """Convert message to dictionary for JSON encoding."""
81 result = {
82 "type": self.message_type.value,
83 "clusterLevel": self.cluster_level,
84 }
85
86 if self.key is not None:
87 result["key"] = self.key.hex()
88
89 if self.value is not None:
90 result["value"] = self.value.hex()
91
92 if self.providers:
93 # Convert peer IDs to hex encoding for JSON serialization
94 result["providers"] = [
95 {"id": p.get("id", b"").hex() if isinstance(p.get("id"), bytes) else p.get("id"),
96 "addrs": p.get("addrs", [])}
97 for p in self.providers
98 ]
99
100 if self.peers:
101 # Convert peer IDs to hex encoding for JSON serialization
102 result["closerPeers"] = [
103 {"id": p.get("id", b"").hex() if isinstance(p.get("id"), bytes) else p.get("id"),
104 "addrs": p.get("addrs", [])}
105 for p in self.peers
106 ]
107
108 if self.peer_id is not None:
109 result["peer_id"] = self.peer_id.hex()
110
111 if self.record is not None:
112 result["record"] = self.record
113
114 return result

116 def to_json(self) -> bytes:
117 """Convert message to JSON bytes."""
118 return json.dumps(self.to_dict()).encode("utf-8")

220class KademliaDHT(DHT):
221 """
222 Kademlia DHT implementation.
223
224 Implements the full Kademlia DHT protocol including:
225 - Node discovery (find_peer)
226 - Content routing (provide, find_providers)
227 - Key-value storage (put_value, get_value)
228 """

230 def __init__(
231 self,
232 local_peer_id: bytes,
233 network_send_func: Optional[Callable[[bytes, bytes, str], Awaitable[bytes]]] = None,
234 k_size: int = K,
235 ):
236 """
237 Initialize Kademlia DHT.
238
239 Args:
240 local_peer_id: Local peer's ID (32 bytes)
241 network_send_func: Async function to send DHT messages
242 (target_peer_id, message_bytes, protocol_id) -> response
243 k_size: Bucket size parameter (default 20)
244 """
245 if len(local_peer_id) != BYTE_COUNT:
246 raise ValueError(f"Peer ID must be {BYTE_COUNT} bytes")
247
248 self.local_peer_id = local_peer_id
249 self.protocol_id = PROTOCOL_ID
250 self._send_func = network_send_func
251
252 # Core components
253 self.routing_table = RoutingTable(local_peer_id, k_size)
254 self.provider_manager = ProviderManager(max_providers=k_size)
255 self.query_manager = QueryManager(alpha=3, timeout_ms=60000)
256
257 # Local key-value store
258 self._local_store: Dict[bytes, tuple[bytes, float]] = {}
259
260 # Event handlers
261 self._on_peer_found: List[Callable[[PeerEntry], Awaitable[None]]] = []
262 self._on_provider_found: List[Callable[[bytes, PeerEntry], Awaitable[None]]] = []
263
264 # Background tasks
265 self._running = False
266 self._refresh_task: Optional[asyncio.Task] = None
267
268 async def start(self) -> None:
269 """Start DHT background tasks."""
270 if self._running:
271 return
272
273 self._running = True
274
275 # Start periodic bucket refresh
276 self._refresh_task = asyncio.create_task(self._refresh_loop())
277
278 logger.info(f"Kademlia DHT started: peer_id={self.local_peer_id.hex()[:8]}...")
279
280 async def stop(self) -> None:
281 """Stop DHT background tasks."""
282 if not self._running:
283 return
284
285 self._running = False
286
287 if self._refresh_task:
288 self._refresh_task.cancel()
289 try:
290 await self._refresh_task
291 except asyncio.CancelledError:
292 pass
293
294 logger.info("Kademlia DHT stopped")

296 async def find_peer(self, peer_id: bytes) -> Optional[PeerEntry]:
297 """
298 Find a peer by ID.
299
300 First checks local routing table, then performs iterative lookup.
301
302 Args:
303 peer_id: Peer ID to find (32 bytes)
304
305 Returns:
306 Peer info if found, None otherwise
307 """
308 # Check local routing table first
309 local_peer = self.routing_table.find_peer(peer_id)
310 if local_peer:
311 return local_peer
312
313 # Perform iterative lookup
314 result = await self.query_manager.find_peer(
315 peer_id,
316 self.routing_table,
317 self._query_find_node
318 )
319
320 if result.success:
321 # Add found peers to routing table
322 for peer in result.peers:
323 await self.routing_table.add_peer(peer)
324
325 # Check if we found the target
326 if peer.peer_id == peer_id:
327 return peer
328
329 return None

331 async def provide(self, cid: bytes, announce: bool = True) -> bool:
332 """
333 Announce that we provide a content.
334
335 Args:
336 cid: Content ID (CID)
337 announce: Whether to announce to network
338
339 Returns:
340 True if successful
341 """
342 # Add to local providers
343 await self.provider_manager.add_local_provider(cid)
344
345 if not announce or not self._send_func:
346 return True
347
348 # Announce to closest peers
349 result = await self.query_manager.provide(
350 cid,
351 self.local_peer_id,
352 self.routing_table,
353 self._announce_provider
354 )
355
356 return result.success

358 async def find_providers(self, cid: bytes, count: int = 20) -> List[PeerEntry]:
359 """
360 Find providers for a content.
361
362 Args:
363 cid: Content ID (CID)
364 count: Maximum number of providers to return
365
366 Returns:
367 List of provider peers
368 """
369 # Check local providers first
370 local_providers = await self.provider_manager.get_provider_peer_ids(cid, count)
371 if local_providers:
372 # Convert to PeerEntry list
373 providers = []
374 for peer_id in local_providers:
375 peer = self.routing_table.find_peer(peer_id)
376 if peer:
377 providers.append(peer)
378 if providers:
379 return providers[:count]
380
381 # Perform network lookup
382 result = await self.query_manager.find_providers(
383 cid,
384 self.routing_table,
385 self._query_get_providers
386 )
387
388 if result.success:
389 # Add providers to routing table and local cache
390 for peer in result.providers:
391 await self.routing_table.add_peer(peer)
392
393 return result.providers[:count]
394
395 return []

397 async def put_value(self, key: bytes, value: bytes) -> bool:
398 """
399 Store a key-value pair in DHT.
400
401 Args:
402 key: Key to store under
403 value: Value to store
404
405 Returns:
406 True if successful
407 """
408 dht_key = compute_key(key)
409
410 # Store locally
411 self._local_store[dht_key] = (value, time.time() + 3600)
412
413 if not self._send_func:
414 return True
415
416 # Announce to closest peers
417 closest_peers = self.routing_table.find_closest_peers(dht_key, K)
418
419 success = True
420 for peer in closest_peers:
421 try:
422 message = KademliaMessage(
423 message_type=KademliaMessageType.PUT_VALUE,
424 key=dht_key,
425 value=value,
426 peer_id=self.local_peer_id,
427 )
428
429 response_data = await self._send_func(
430 peer.peer_id,
431 message.to_json(),
432 self.protocol_id
433 )
434
435 if response_data:
436 response = KademliaMessage.from_json(response_data)
437 if response.message_type == KademliaMessageType.PUT_VALUE:
438 continue
439
440 except Exception as e:
441 logger.warning(f"Failed to put value to peer: {e}")
442 success = False
443
444 return success

446 async def get_value(self, key: bytes) -> Optional[bytes]:
447 """
448 Retrieve a value from DHT.
449
450 Args:
451 key: Key to retrieve
452
453 Returns:
454 Value if found, None otherwise
455 """
456 dht_key = compute_key(key)
457
458 # Check local store first
459 if dht_key in self._local_store:
460 value, expires = self._local_store[dht_key]
461 if time.time() < expires:
462 return value
463 else:
464 del self._local_store[dht_key]
465
466 if not self._send_func:
467 return None
468
469 # Query network
470 result = await self.query_manager.find_value(
471 dht_key,
472 self.routing_table,
473 self._query_get_value
474 )
475
476 if result.success and result.value:
477 return result.value
478
479 return None

481 async def handle_message(self, peer_id: bytes, message: bytes) -> bytes:
482 """
483 Handle incoming DHT message.
484
485 Args:
486 peer_id: Sender's peer ID
487 message: Raw message bytes
488
489 Returns:
490 Response message bytes
491 """
492 try:
493 msg = KademliaMessage.from_json(message)
494 except Exception as e:
495 logger.error(f"Failed to parse DHT message: {e}")
496 return self._create_error_response("Invalid message").to_json()
497
498 # Add sender to routing table
499 await self.routing_table.add_peer(PeerEntry(peer_id=peer_id))
500
501 # Handle based on message type
502 if msg.message_type == KademliaMessageType.FIND_NODE:
503 return await self._handle_find_node(msg)
504 elif msg.message_type == KademliaMessageType.FIND_VALUE:
505 return await self._handle_find_value(msg)
506 elif msg.message_type == KademliaMessageType.PUT_VALUE:
507 return await self._handle_put_value(msg)
508 elif msg.message_type == KademliaMessageType.ADD_PROVIDER:
509 return await self._handle_add_provider(msg, peer_id)
510 elif msg.message_type == KademliaMessageType.GET_PROVIDERS:
511 return await self._handle_get_providers(msg)
512 elif msg.message_type == KademliaMessageType.PING:
513 return self._create_pong_response().to_json()
514 else:
515 return self._create_error_response("Unknown message type").to_json()

547 async def _handle_find_value(self, msg: KademliaMessage) -> bytes:
548 """Handle FIND_VALUE (GET_VALUE) request."""
549 if msg.key is None:
550 return self._create_error_response("Missing key").to_json()
551
552 # Check local store
553 value = None
554 if msg.key in self._local_store:
555 stored_value, expires = self._local_store[msg.key]
556 if time.time() < expires:
557 value = stored_value
558
559 # Get closest peers
560 closest_peers = self.routing_table.find_closest_peers(msg.key, K)
561
562 response = KademliaMessage(
563 message_type=KademliaMessageType.FIND_VALUE,
564 key=msg.key,
565 value=value,
566 peers=[{
567 "id": p.peer_id,
568 "addrs": p.addresses,
569 } for p in closest_peers],
570 peer_id=self.local_peer_id,
571 )
572
573 return response.to_json()

607 async def _handle_get_providers(self, msg: KademliaMessage) -> bytes:
608 """Handle GET_PROVIDERS request."""
609 if msg.key is None:
610 return self._create_error_response("Missing key").to_json()
611
612 # Hash the key for DHT operations
613 import hashlib
614 dht_key = hashlib.sha256(msg.key).digest()
615
616 # Get providers
617 providers = self.provider_manager.get_providers(msg.key)
618
619 # Get closest peers using the hashed key
620 closest_peers = self.routing_table.find_closest_peers(dht_key, K)
621
622 response = KademliaMessage(
623 message_type=KademliaMessageType.GET_PROVIDERS,
624 key=msg.key,
625 providers=[{
626 "id": p.peer_id,
627 "addrs": [], # Would get from routing table
628 } for p in providers],
629 peers=[{
630 "id": p.peer_id,
631 "addrs": p.addresses,
632 } for p in closest_peers],
633 peer_id=self.local_peer_id,
634 )
635
636 return response.to_json()

792 async def _refresh_loop(self) -> None:
793 """Periodically refresh routing table buckets."""
794 while self._running:
795 try:
796 await asyncio.sleep(300) # Refresh every 5 minutes
797
798 # Cleanup stale entries
799 removed = await self.routing_table.cleanup_stale()
800 if removed:
801 logger.debug(f"Cleaned up {removed} stale peers from routing table")
802
803 # Cleanup expired providers
804 expired = await self.provider_manager.cleanup_expired()
805 if expired:
806 logger.debug(f"Cleaned up {expired} expired provider records")
807
808 # Cleanup expired queries
809 expired_queries = await self.query_manager.cleanup_expired_queries()
810 if expired_queries:
811 logger.debug(f"Cleaned up {expired_queries} expired queries")
812
813 except asyncio.CancelledError:
814 break
815 except Exception as e:
816 logger.error(f"Error in refresh loop: {e}")
