49 async def punch(self) -> PunchResult:
50 """
51 执行打孔
52
53 决策树：
54 1. 双对称 NAT → 直接失败，走中继
55 2. 单对称 NAT → 端口预测 + 多端口并行
56 3. 非对称 NAT → 标准双向打孔
57 """
58 start_time = time.time()
59
60 # 获取运营商配置
61 local_profile = get_isp_profile(self.local_isp)
62
63 # 决策：双对称 NAT
64 if self.local_nat.is_symmetric() and self.peer_nat.is_symmetric():
65 logger.info("双对称 NAT，无法直接穿透")
66 return PunchResult(
67 success=False,
68 error="symmetric_symmetric",
69 )
70
71 # 创建 UDP socket
72 self._socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
73 self._socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
74
75 try:
76 # 绑定本地端口
77 self._socket.bind(("0.0.0.0", 0))
78 self._local_port = self._socket.getsockname()[1]
79 self._socket.setblocking(False)
80
81 logger.info(f"开始打孔，本地端口: {self._local_port}")
82
83 # 根据情况选择策略
84 if self.local_nat.is_symmetric() or self.peer_nat.is_symmetric():
85 result = await self._punch_symmetric(local_profile)
86 else:
87 result = await self._punch_standard(local_profile)
88
89 if result.success:
90 result.latency_ms = (time.time() - start_time) * 1000
91
92 return result
93
94 finally:
95 if self._socket and not result.success:
96 self._socket.close()

98 async def _punch_standard(self, profile: ISPProfile) -> PunchResult:
99 """
100 标准双向打孔（非对称 NAT）
101
102 双方同时向对方发送数据包
103 """
104 peer_addr = (self.peer_nat.public_ip, self.peer_nat.public_port)
105 timeout = profile.punch_timeout_ms / 1000
106
107 loop = asyncio.get_event_loop()
108
109 try:
110 # 发送打洞包
111 punch_packet = b"PUNCH_HELLO"
112 await loop.sock_sendto(self._socket, punch_packet, peer_addr)
113 logger.debug(f"发送打洞包到 {peer_addr}")
114
115 # 等待响应
116 async def receive():
117 while True:
118 data, addr = await loop.sock_recvfrom(self._socket, 1024)
119 if data.startswith(b"PUNCH"):
120 logger.debug(f"收到打洞响应: {addr}")
121 return addr
122
123 addr = await asyncio.wait_for(receive(), timeout=timeout)
124
125 return PunchResult(
126 success=True,
127 connection_type=ConnectionType.P2P_UDP,
128 local_addr=("0.0.0.0", self._local_port),
129 peer_addr=addr,
130 )
131
132 except asyncio.TimeoutError:
133 logger.warning("打孔超时")
134 return PunchResult(success=False, error="timeout")
135 except Exception as e:
136 logger.error(f"打孔失败: {e}")
137 return PunchResult(success=False, error=str(e))

139 async def _punch_symmetric(self, profile: ISPProfile) -> PunchResult:
140 """
141 对称 NAT 打孔（端口预测 + 多端口并行）
142 """
143 # 端口预测
144 predictor = PortPredictor(
145 base_port=self.peer_nat.public_port,
146 strategy=profile.punch_port_strategy,
147 predict_count=profile.punch_parallel_ports,
148 )
149 prediction = predictor.predict()
150
151 logger.debug(f"端口预测: {prediction.ports[:4]}... (策略: {prediction.strategy})")
152
153 # 并行向多个预测端口发送
154 peer_ip = self.peer_nat.public_ip
155 timeout = profile.punch_timeout_ms / 1000
156
157 loop = asyncio.get_event_loop()
158
159 try:
160 # 并行发送到多个端口
161 punch_packet = b"PUNCH_HELLO"
162 tasks = []
163 for port in prediction.ports:
164 task = loop.sock_sendto(self._socket, punch_packet, (peer_ip, port))
165 tasks.append(task)
166
167 await asyncio.gather(*tasks, return_exceptions=True)
168 logger.debug(f"已向 {len(prediction.ports)} 个预测端口发送打洞包")
169
170 # 等待响应
171 async def receive():
172 while True:
173 data, addr = await loop.sock_recvfrom(self._socket, 1024)
174 if data.startswith(b"PUNCH"):
175 return addr
176
177 addr = await asyncio.wait_for(receive(), timeout=timeout)
178
179 return PunchResult(
180 success=True,
181 connection_type=ConnectionType.P2P_UDP,
182 local_addr=("0.0.0.0", self._local_port),
183 peer_addr=addr,
184 )
185
186 except asyncio.TimeoutError:
187 return PunchResult(success=False, error="timeout")
188 except Exception as e:
189 return PunchResult(success=False, error=str(e))
