107 def to_protobuf_dict(self) -> dict:
108 """
109 Convert to protobuf-compatible dictionary.

111 Returns a dict with optional fields set only if they have values,
112 following protobuf semantics.
113 """
114 result = {}
115
116 if self.protocol_version:
117 result["protocolVersion"] = self.protocol_version
118 if self.agent_version:
119 result["agentVersion"] = self.agent_version
120 if self.public_key:
121 result["publicKey"] = self.public_key
122 if self.listen_addrs:
123 result["listenAddrs"] = self.listen_addrs
124 if self.observed_addr:
125 result["observedAddr"] = self.observed_addr
126 if self.protocols:
127 result["protocols"] = self.protocols
128
129 # Always include extension data (even if empty) for compatibility
130 ext_bytes = self.ext.to_bytes()
131 if ext_bytes:
132 result["ext"] = ext_bytes
133
134 return result

136 def to_json_dict(self) -> dict:
137 """
138 Convert to JSON-serializable dictionary.
139
140 Similar to to_protobuf_dict but encodes bytes fields as base64 strings
141 for JSON serialization.
142 """
143 result = {}
144
145 if self.protocol_version:
146 result["protocolVersion"] = self.protocol_version
147 if self.agent_version:
148 result["agentVersion"] = self.agent_version
149 if self.public_key:
150 result["publicKey"] = base64.b64encode(self.public_key).decode("ascii")
151 if self.listen_addrs:
152 result["listenAddrs"] = [
153 base64.b64encode(addr).decode("ascii") for addr in self.listen_addrs
154 ]
155 if self.observed_addr:
156 result["observedAddr"] = base64.b64encode(self.observed_addr).decode("ascii")
157 if self.protocols:
158 result["protocols"] = self.protocols
159
160 # Extension data
161 ext_bytes = self.ext.to_bytes()
162 if ext_bytes:
163 result["ext"] = base64.b64encode(ext_bytes).decode("ascii")
164
165 return result

167 @classmethod
168 def from_protobuf_dict(cls, data: dict) -> "IdentifyMessage":
169 """Create IdentifyMessage from protobuf dictionary."""
170 ext_data = data.get("ext", b"")
171 # Handle both bytes and base64-encoded strings
172 if isinstance(ext_data, str):
173 ext_data = base64.b64decode(ext_data)
174 extension = IdentifyExtension.from_bytes(ext_data)

176 # Handle bytes fields that might be base64-encoded

204class ConnectionWrapper:
205 """
206 Wrapper for network connections.
207
208 Abstracts the underlying transport (TCP, WebSocket, etc.)
209 to provide a consistent interface for protocol operations.
210 """
211
212 def __init__(self, reader: asyncio.StreamReader, writer: asyncio.StreamWriter):
213 self.reader = reader
214 self.writer = writer
215 self._closed = False
216
217 async def send(self, data: bytes) -> None:
218 """Send data with length prefix."""
219 if self._closed:
220 raise RuntimeError("Connection is closed")
221
222 # Use variable-length length prefix (varint)
223 length = len(data)
224 length_bytes = bytearray()
225 while length > 0x7F:
226 length_bytes.append((length & 0x7F) | 0x80)
227 length >>= 7
228 length_bytes.append(length)
229
230 self.writer.write(bytes(length_bytes))
231 self.writer.write(data)
232 await self.writer.drain()
233
234 async def recv(self) -> bytes:
235 """Receive data with length prefix."""
236 if self._closed:
237 raise RuntimeError("Connection is closed")
238
239 # Read varint length
240 length = 0
241 shift = 0
242 while True:
243 byte = await self.reader.readexactly(1)
244 b = ord(byte)
245 length |= (b & 0x7F) << shift
246 if not (b & 0x80):
247 break
248 shift += 7
249
250 # Read data
251 if length > 0:
252 return await self.reader.readexactly(length)
253 return b""
254
255 async def close(self) -> None:
256 """Close the connection."""
257 if self._closed:
258 return
259
260 self._closed = True
261 try:
262 self.writer.close()
263 await self.writer.wait_closed()
264 except Exception as e:
265 logger.debug(f"Error closing connection: {e}")
266
267 @property
268 def closed(self) -> bool:
269 """Check if connection is closed."""
270 return self._closed
271
272 def get_remote_address(self) -> tuple[str, int]:
273 """Get remote address."""
274 return self.writer.get_extra_info("peername") or ("", 0)

298 async def exchange(
299 self,
300 conn: ConnectionWrapper,
301 local_info: Optional[IdentifyMessage] = None,
302 ) -> IdentifyMessage:
303 """
304 Perform Identify exchange (query mode).
305
306 Opens a stream to the remote peer and sends an empty message
307 to request their identity information.
308
309 Args:
310 conn: Connection wrapper for the network stream
311 local_info: Optional local identity info to send
312
313 Returns:
314 IdentifyMessage from the remote peer
315
316 Raises:
317 RuntimeError: If connection fails or times out
318 """
319 logger.debug("Starting Identify exchange")
320
321 try:
322 # Send local info if provided (for future bidirectional exchange)
323 if local_info:
324 await self._send_message(conn, local_info)
325
326 # Receive remote identity
327 remote_info = await self._recv_message(conn)
328
329 logger.debug(
330 f"Identify exchange completed: "
331 f"agent={remote_info.agent_version}, "
332 f"protocols={len(remote_info.protocols)}"
333 )
334
335 return remote_info
336
337 except asyncio.TimeoutError:
338 logger.error("Identify exchange timed out")
339 raise RuntimeError("Identify exchange timeout")
340 except Exception as e:
341 logger.error(f"Identify exchange failed: {e}")
342 raise RuntimeError(f"Identify exchange failed: {e}")
