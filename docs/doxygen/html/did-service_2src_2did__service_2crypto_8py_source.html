99 # Format: PREFIX-XXXXXX-YYYYY
100 # PREFIX: Device type prefix (e.g., ios, android, web, desktop)
101 # XXXXXX: 6-character random hex (3 bytes)
102 # YYYYY: 5-character random alphanumeric
103
104 PREFIX_LENGTH = 6
105 SUFFIX_LENGTH = 5
106
107 @staticmethod
108 def generate(device_type: str) -> str:
109 """
110 Generate a DID for a device.
111
112 Args:
113 device_type: Type of device (ios, android, web, desktop)
114
115 Returns:
116 DID string in format: PREFIX-XXXXXX-YYYYY
117 """
118 # Generate middle section (6 hex chars = 3 bytes)
119 middle = secrets.token_hex(3).upper()
120
121 # Generate suffix section (5 alphanumeric chars)
122 alphabet = "0123456789ABCDEFGHJKLMNPQRSTUVWXYZ" # No ambiguous chars
123 suffix = "".join(secrets.choice(alphabet) for _ in range(DIDGenerator.SUFFIX_LENGTH))
124
125 return f"{device_type}-{middle}-{suffix}"
126
127 @staticmethod
128 def validate(device_id: str) -> bool:
129 """
130 Validate a DID format.
131
132 Args:
133 device_id: DID string to validate
134
135 Returns:
136 True if format is valid
137 """
138 parts = device_id.split("-")
139 if len(parts) != 3:
140 return False
141
142 prefix, middle, suffix = parts
143
144 # Validate prefix
145 valid_prefixes = {"ios", "android", "web", "desktop"}
146 if prefix.lower() not in valid_prefixes:
147 return False
148
149 # Validate middle (6 hex chars)
150 if len(middle) != 6 or not all(c in "0123456789ABCDEFabcdef" for c in middle):
151 return False
152
153 # Validate suffix (5 alphanumeric chars, no ambiguous chars)
154 alphabet = "0123456789ABCDEFGHJKLMNPQRSTUVWXYZ"
155 if len(suffix) != 5 or not all(c in alphabet for c in suffix):
156 return False
157
158 return True
