52 def _predict_sequential(self) -> PredictionResult:
53 """递增预测"""
54 ports = [self.base_port + i for i in range(1, self.predict_count + 1)]
55 return PredictionResult(ports=ports, strategy="sequential")

57 def _predict_random(self) -> PredictionResult:
58 """随机预测"""
59 import random
60 # 在 base_port 附近随机选择
61 ports = []
62 for _ in range(self.predict_count):
63 offset = random.randint(1, 100)
64 port = self.base_port + offset
65 if 1024 <= port <= 65535:
66 ports.append(port)
67 return PredictionResult(ports=ports, strategy="random")

69 def _predict_hybrid(self) -> PredictionResult:
70 """混合预测（推荐）"""
71 import random
72
73 ports = []
74
75 # 60% 递增
76 seq_count = int(self.predict_count * 0.6)
77 for i in range(1, seq_count + 1):
78 ports.append(self.base_port + i)
79
80 # 40% 随机
81 rand_count = self.predict_count - seq_count
82 for _ in range(rand_count):
83 offset = random.randint(1, 50)
84 port = self.base_port + offset
85 if port not in ports and 1024 <= port <= 65535:
86 ports.append(port)
87
88 return PredictionResult(ports=ports, strategy="hybrid")

116 logger.debug(f"学习到端口规律: 递增步长 {deltas[0]}")
117 return deltas[0]
