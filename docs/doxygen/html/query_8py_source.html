264 async def provide(
265 self,
266 key: bytes,
267 local_peer_id: bytes,
268 routing_table,
269 query_func: Callable[[bytes, bytes, bytes], Awaitable[bool]]
270 ) -> QueryResult:
271 """
272 Announce that we provide a content key.
273
274 Args:
275 key: Content key (CID)
276 local_peer_id: Our peer ID
277 routing_table: Routing table to use
278 query_func: Async function to announce to peer
279
280 Returns:
281 Query result
282 """
283 import hashlib
284 target_id = hashlib.sha256(key).digest()
285
286 # Find closest peers to announce to
287 closest_peers = routing_table.find_closest_peers(target_id, K)
288
289 if not closest_peers:
290 return QueryResult(
291 success=False,
292 error="No peers in routing table",
293 duration_ms=0.0
294 )
295
296 # Announce to all closest peers
297 announced = 0
298 start_time = time.time()
299
300 tasks = [
301 query_func(peer.peer_id, key, local_peer_id)
302 for peer in closest_peers
303 ]
304
305 results = await asyncio.gather(*tasks, return_exceptions=True)
306
307 for i, result in enumerate(results):
308 if isinstance(result, Exception):
309 logger.warning(f"Failed to announce to peer: {result}")
310 elif result:
311 announced += 1
312
313 return QueryResult(
314 success=announced > 0,
315 peers=closest_peers,
316 duration_ms=(time.time() - start_time) * 1000
317 )

319 async def _execute_iterative_query(
320 self,
321 context: QueryContext,
322 initial_peers: List[PeerEntry],
323 query_func: Callable,
324 find_value: bool = False,
325 find_providers: bool = False
326 ) -> QueryResult:
327 """
328 Execute an iterative Kademlia query.
329
330 Args:
331 context: Query context
332 initial_peers: Initial peers to query
333 query_func: Function to query peers
334 find_value: Whether this is a value lookup
335 find_providers: Whether this is a provider lookup
336
337 Returns:
338 Query result
339 """
340 context.state = QueryState.RUNNING
341
342 # Initialize query queue with initial peers
343 for peer in initial_peers:
344 distance = calculate_distance(peer.peer_id, context.target_id)
345 heapq.heappush(context.pending, QueryPeer(distance, peer))
346
347 # Track closest peer distance
348 closest_distance = min((p.distance for p in context.pending), default=None)
349
350 while context.pending and not context.is_expired(self.timeout_ms):
351 # Get ALPHA closest unqueried peers
352 to_query: List[QueryPeer] = []
353
354 while len(to_query) < self.alpha and context.pending:
355 peer_info = heapq.heappop(context.pending)
356
357 if peer_info.peer.peer_id not in context.contacted:
358 to_query.append(peer_info)
359
360 if not to_query:
361 break
362
363 # Query peers in parallel
364 tasks = [
365 self._query_peer(peer, context, query_func, find_value, find_providers)
366 for peer in to_query
367 ]
368
369 await asyncio.gather(*tasks, return_exceptions=True)
370
371 # Check if we found what we're looking for
372 if find_value and context.value:
373 context.state = QueryState.COMPLETED
374 break
375
376 # Check if we have enough providers
377 if find_providers and len(context.providers) >= K:
378 context.state = QueryState.COMPLETED
379 break
380
381 # Check if we've found the target peer
382 if context.query_type == QueryType.FIND_PEER:
383 for peer in context.results:
384 if peer.peer_id == context.target_id:
385 context.state = QueryState.COMPLETED
386 break
387
388 # Stop if we've exhausted closer peers
389 if context.pending:
390 current_closest = context.pending[0].distance
391 if current_closest >= closest_distance:
392 # No closer peers found
393 break
394 closest_distance = current_closest
395
396 # Finalize result
397 duration = context.elapsed_ms
398
399 if context.state == QueryState.RUNNING:
400 context.state = QueryState.COMPLETED if context.results else QueryState.TIMEOUT
401
402 return QueryResult(
403 success=context.state == QueryState.COMPLETED,
404 peers=context.results[:K],
405 value=context.value,
406 providers=context.providers[:K],
407 error=context.error,
408 duration_ms=duration
409 )

411 async def _query_peer(
412 self,
413 peer_info: QueryPeer,
414 context: QueryContext,
415 query_func: Callable,
416 find_value: bool,
417 find_providers: bool
418 ) -> None:
419 """
420 Query a single peer and update context.
421
422 Args:
423 peer_info: Peer to query with distance
424 context: Query context to update
425 query_func: Query function
426 find_value: Whether looking for value
427 find_providers: Whether looking for providers
428 """
429 peer_id = peer_info.peer.peer_id
430 context.contacted.add(peer_id)
431 peer_info.queried = True
432
433 try:
434 if find_value:
435 # Query for value
436 result = await query_func(peer_id, context.target_key)
437 if isinstance(result, tuple):
438 value, closer_peers = result
439 else:
440 value, closer_peers = result, []
441
442 if value:
443 context.value = value
444 context.results.append(peer_info.peer)
445
446 elif find_providers:
447 # Query for providers
448 result = await query_func(peer_id, context.target_key)
449 if isinstance(result, tuple):
450 providers, closer_peers = result
451 else:
452 providers, closer_peers = [], result
453
454 context.providers.extend(providers)
455
456 # Add closer peers
457 for peer in closer_peers:
458 distance = calculate_distance(peer.peer_id, context.target_id)
459 heapq.heappush(context.pending, QueryPeer(distance, peer))
460
461 else:
462 # Regular peer lookup
463 closer_peers = await query_func(peer_id, context.target_id)
464
465 # Add closer peers to results and pending queue
466 for peer in closer_peers:
467 context.results.append(peer)
468 distance = calculate_distance(peer.peer_id, context.target_id)
469 heapq.heappush(context.pending, QueryPeer(distance, peer))
470
471 except Exception as e:
472 logger.debug(f"Query to peer {peer_id.hex()[:8]} failed: {e}")
473 context.retries += 1
