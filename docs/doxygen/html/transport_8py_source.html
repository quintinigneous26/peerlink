35class UDPTransport(TransportBase):
36 """
37 UDP transport for direct P2P communication.
38
39 Handles raw UDP socket operations for hole punching
40 and direct peer-to-peer data transfer.
41 """
42
43 def __init__(self, local_ip: str = "0.0.0.0", local_port: int = 0):
44 """
45 Initialize UDP transport.
46
47 Args:
48 local_ip: Local IP to bind to
49 local_port: Local port (0 for auto-assign)
50 """
51 self.local_ip = local_ip
52 self.local_port = local_port
53 self._socket: Optional[socket.socket] = None
54 self._running = False
55 self._peer_addr: Optional[Tuple[str, int]] = None
56 self._recv_queue: asyncio.Queue[bytes] = asyncio.Queue()
57 self._recv_task: Optional[asyncio.Task] = None
58
59 async def start(self) -> None:
60 """Start the UDP transport."""
61 if self._running:
62 return
63
64 self._socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
65 self._socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
66 self._socket.bind((self.local_ip, self.local_port))
67
68 # Get assigned port
69 self.local_port = self._socket.getsockname()[1]
70
71 # Set non-blocking
72 self._socket.setblocking(False)
73
74 self._running = True
75 self._recv_task = asyncio.create_task(self._recv_loop())
76
77 logger.info(f"UDP transport started on {self.local_ip}:{self.local_port}")
78
79 async def stop(self) -> None:
80 """Stop the UDP transport."""
81 self._running = False
82
83 if self._recv_task:
84 self._recv_task.cancel()
85 try:
86 await self._recv_task
87 except asyncio.CancelledError:
88 pass
89
90 if self._socket:
91 self._socket.close()
92 self._socket = None
93
94 logger.info("UDP transport stopped")
95
96 def set_peer(self, ip: str, port: int) -> None:
97 """Set the peer address for send operations."""
98 self._peer_addr = (ip, port)
99 logger.debug(f"Peer address set to {ip}:{port}")
100
101 async def sendto(self, data: bytes, ip: str, port: int) -> None:
102 """
103 Send data to specific address.
104
105 Args:
106 data: Data to send
107 ip: Target IP address
108 port: Target port
109 """
110 if not self._socket:
111 raise RuntimeError("UDP transport not started")
112
113 loop = asyncio.get_event_loop()
114 await loop.sock_sendto(self._socket, data, (ip, port))
115
116 async def send(self, data: bytes) -> None:
117 """
118 Send data to configured peer.
119
120 Args:
121 data: Data to send
122
123 Raises:
124 RuntimeError: If peer address not set
125 """
126 if not self._peer_addr:
127 raise RuntimeError("Peer address not set. Use sendto() instead.")
128
129 await self.sendto(data, self._peer_addr[0], self._peer_addr[1])
130
131 async def recvfrom(self) -> Tuple[bytes, Tuple[str, int]]:
132 """
133 Receive data from any source.
134
135 Returns:
136 Tuple of (data, (ip, port))
137 """
138 if not self._socket:
139 raise RuntimeError("UDP transport not started")
140
141 loop = asyncio.get_event_loop()
142 data, addr = await loop.sock_recvfrom(self._socket, 65535)
143 return data, addr
144
145 async def recv(self) -> bytes:
146 """
147 Receive data from peer.
148
149 Returns:
150 Received data
151 """
152 return await self._recv_queue.get()
153
154 async def _recv_loop(self) -> None:
155 """Background receive loop."""
156 while self._running:
157 try:
158 data, addr = await self.recvfrom()
159
160 # Store peer address if not set
161 if self._peer_addr is None:
162 self._peer_addr = addr
163
164 # Put in queue if from peer
165 if self._peer_addr and addr == self._peer_addr:
166 await self._recv_queue.put(data)
167
168 except asyncio.CancelledError:
169 break
170 except Exception as e:
171 logger.error(f"Error in UDP recv loop: {e}")

174class RelayTransport(TransportBase):
175 """
176 Relay transport for fallback communication.
177
178 Uses a relay server when direct P2P connection is not possible.
179 """
180
181 def __init__(
182 self,
183 server: str,
184 port: int,
185 did: str,
186 timeout: float = 30.0,
187 ):
188 """
189 Initialize relay transport.
190
191 Args:
192 server: Relay server address
193 port: Relay server port
194 did: Device ID
195 timeout: Connection timeout
196 """
197 self.server = server
198 self.port = port
199 self.did = did
200 self.timeout = timeout
201 self._reader: Optional[asyncio.StreamReader] = None
202 self._writer: Optional[asyncio.StreamWriter] = None
203 self._connected = False
204 self._peer_did: Optional[str] = None
205
206 async def connect(self, peer_did: str) -> None:
207 """
208 Connect to relay server for peer communication.
209
210 Args:
211 peer_did: Peer's device ID
212 """
213 self._peer_did = peer_did
214
215 try:
216 self._reader, self._writer = await asyncio.wait_for(
217 asyncio.open_connection(self.server, self.port),
218 timeout=self.timeout,
219 )
220
221 # Send registration message
222 registration = f"CONNECT {self.did} {peer_did}\n"
223 self._writer.write(registration.encode())
224 await self._writer.drain()
225
226 # Wait for acknowledgment
227 response = await asyncio.wait_for(
228 self._reader.readline(),
229 timeout=self.timeout,
230 )
231
232 if b"OK" not in response:
233 raise ConnectionError(f"Relay connection rejected: {response.decode()}")
234
235 self._connected = True
236 logger.info(f"Relay transport connected to {peer_did} via {self.server}:{self.port}")
237
238 except asyncio.TimeoutError:
239 raise TimeoutError(f"Relay connection timeout to {self.server}:{self.port}")
240
241 async def send(self, data: bytes) -> None:
242 """
243 Send data via relay.
244
245 Args:
246 data: Data to send
247 """
248 if not self._connected or not self._writer:
249 raise RuntimeError("Relay transport not connected")
250
251 # Prefix with length
252 length_prefix = len(data).to_bytes(4, "big")
253 self._writer.write(length_prefix + data)
254 await self._writer.drain()
255
256 async def recv(self) -> bytes:
257 """
258 Receive data via relay.
259
260 Returns:
261 Received data
262 """
263 if not self._connected or not self._reader:
264 raise RuntimeError("Relay transport not connected")
265
266 # Read length prefix
267 length_bytes = await self._reader.readexactly(4)
268 length = int.from_bytes(length_bytes, "big")
269
270 # Read data
271 data = await self._reader.readexactly(length)
272 return data
273
274 async def close(self) -> None:
275 """Close relay connection."""
276 self._connected = False
277
278 if self._writer:
279 try:
280 self._writer.write(b"DISCONNECT\n")
281 await self._writer.drain()
282 except (ConnectionError, OSError) as e:
283 logger.debug(f"Error sending disconnect message: {e}")
284 self._writer.close()
285 try:
286 await self._writer.wait_closed()
287 except (ConnectionError, OSError) as e:
288 logger.debug(f"Error waiting for writer close: {e}")
289
290 self._reader = None
291 self._writer = None
292
293 logger.info("Relay transport closed")

296class WebSocketTransport(TransportBase):
297 """
298 WebSocket transport for signaling and relay communication.
299
300 Uses WebSocket protocol for reliable messaging over HTTP.
301 """
302
303 def __init__(
304 self,
305 url: str,
306 did: str,
307 timeout: float = 30.0,
308 ):
309 """
310 Initialize WebSocket transport.
311
312 Args:
313 url: WebSocket server URL
314 did: Device ID
315 timeout: Connection timeout
316 """
317 self.url = url
318 self.did = did
319 self.timeout = timeout
320 self._ws: Optional[Any] = None
321 self._connected = False
322 self._send_queue: asyncio.Queue[bytes] = asyncio.Queue()
323 self._recv_queue: asyncio.Queue[bytes] = asyncio.Queue()
324 self._send_task: Optional[asyncio.Task] = None
325 self._recv_task: Optional[asyncio.Task] = None
326
327 async def connect(self) -> None:
328 """Connect to WebSocket server."""
329 try:
330 import websockets
331
332 self._ws = await asyncio.wait_for(
333 websockets.connect(self.url),
334 timeout=self.timeout,
335 )
336
337 self._connected = True
338
339 # Start background tasks
340 self._send_task = asyncio.create_task(self._send_loop())
341 self._recv_task = asyncio.create_task(self._recv_loop())
342
343 logger.info(f"WebSocket transport connected to {self.url}")
344
345 except ImportError:
346 raise RuntimeError("websockets package required for WebSocketTransport")
347
348 async def send(self, data: bytes) -> None:
349 """Queue data to send."""
350 await self._send_queue.put(data)
351
352 async def recv(self) -> bytes:
353 """Receive data."""
354 return await self._recv_queue.get()
355
356 async def _send_loop(self) -> None:
357 """Background send loop."""
358 while self._connected and self._ws:
359 try:
360 data = await self._send_queue.get()
361 await self._ws.send(data)
362 except Exception as e:
363 logger.error(f"Error in WebSocket send loop: {e}")
364 break
365
366 async def _recv_loop(self) -> None:
367 """Background receive loop."""
368 while self._connected and self._ws:
369 try:
370 data = await self._ws.recv()
371 if isinstance(data, str):
372 data = data.encode("utf-8")
373 await self._recv_queue.put(data)
374 except Exception as e:
375 logger.error(f"Error in WebSocket recv loop: {e}")
376 break
377
378 async def close(self) -> None:
379 """Close WebSocket connection."""
380 self._connected = False
381
382 if self._send_task:
383 self._send_task.cancel()
384 if self._recv_task:
385 self._recv_task.cancel()
386
387 if self._ws:
388 await self._ws.close()
389
390 logger.info("WebSocket transport closed")
