55 async def binding_request(
56 self,
57 server: str,
58 port: int = 3478,
59 ) -> STUNResponse:
60 """
61 发送 STUN Binding Request
62
63 Args:
64 server: STUN 服务器地址
65 port: STUN 服务器端口
66
67 Returns:
68 STUNResponse
69 """
70 # 解析服务器地址
71 try:
72 server_ip = await self._resolve_host(server)
73 except Exception as e:
74 return STUNResponse(success=False, error=f"DNS 解析失败: {e}")
75
76 # 创建 UDP socket
77 sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
78 sock.settimeout(self.timeout_ms / 1000)
79
80 try:
81 # 构建 Binding Request
82 transaction_id = os.urandom(12)
83 request = self._build_binding_request(transaction_id)
84
85 # 发送请求
86 sock.sendto(request, (server_ip, port))
87
88 # 接收响应
89 data, addr = sock.recvfrom(1024)
90
91 # 解析响应
92 return self._parse_response(data, transaction_id)
93
94 except socket.timeout:
95 return STUNResponse(success=False, error="超时")
96 except Exception as e:
97 return STUNResponse(success=False, error=str(e))
98 finally:
99 sock.close()

107 def _build_binding_request(self, transaction_id: bytes) -> bytes:
108 """构建 Binding Request"""
109 # Type(2) + Length(2) + Magic Cookie(4) + Transaction ID(12)
110 header = struct.pack(
111 ">HHI12s",
112 STUN_BINDING_REQUEST,
113 0, # 无属性
114 STUN_MAGIC_COOKIE,
115 transaction_id
116 )
117 return header

119 def _parse_response(self, data: bytes, transaction_id: bytes) -> STUNResponse:
120 """解析 STUN 响应"""
121 if len(data) < 20:
122 return STUNResponse(success=False, error="响应太短")
123
124 # 解析头部
125 msg_type = struct.unpack(">H", data[0:2])[0]
126 msg_length = struct.unpack(">H", data[2:4])[0]
127 magic_cookie = struct.unpack(">I", data[4:8])[0]
128 resp_transaction_id = data[8:20]
129
130 # 验证
131 if magic_cookie != STUN_MAGIC_COOKIE:
132 return STUNResponse(success=False, error="Magic Cookie 不匹配")
133
134 if resp_transaction_id != transaction_id:
135 return STUNResponse(success=False, error="Transaction ID 不匹配")
136
137 if msg_type != STUN_BINDING_RESPONSE:
138 return STUNResponse(success=False, error=f"非成功响应: 0x{msg_type:04x}")
139
140 # 解析属性
141 response = STUNResponse(success=True)
142 offset = 20
143
144 while offset + 4 <= len(data):
145 attr_type = struct.unpack(">H", data[offset:offset+2])[0]
146 attr_length = struct.unpack(">H", data[offset+2:offset+4])[0]
147
148 if offset + 4 + attr_length > len(data):
149 break
150
151 attr_data = data[offset+4:offset+4+attr_length]
152
153 if attr_type == STUN_ATTR_XOR_MAPPED_ADDRESS:
154 mapped = self._parse_xor_mapped_address(attr_data)
155 if mapped:
156 response.mapped_ip = mapped[0]
157 response.mapped_port = mapped[1]
158
159 elif attr_type == STUN_ATTR_MAPPED_ADDRESS:
160 mapped = self._parse_mapped_address(attr_data)
161 if mapped:
162 response.mapped_ip = mapped[0]
163 response.mapped_port = mapped[1]
164
165 # 移动到下一个属性（4字节对齐）
166 padded_length = (attr_length + 3) & ~3
167 offset += 4 + padded_length
168
169 if not response.mapped_ip:
170 response.success = False
171 response.error = "未找到映射地址"
172
173 return response

175 def _parse_xor_mapped_address(self, data: bytes) -> Optional[Tuple[str, int]]:
176 """解析 XOR-MAPPED-ADDRESS"""
177 if len(data) < 8:
178 return None
179
180 family = data[1]
181 if family != 1: # 只支持 IPv4
182 return None
183
184 xport = struct.unpack(">H", data[2:4])[0]
185 xip = data[4:8]
186
187 # XOR 解码
188 port = xport ^ ((STUN_MAGIC_COOKIE >> 16) & 0xFFFF)
189 ip_bytes = bytes(b ^ m for b, m in zip(
190 xip,
191 struct.pack(">I", STUN_MAGIC_COOKIE)
192 ))
193 ip = ".".join(str(b) for b in ip_bytes)
194
195 return ip, port

198 """解析 MAPPED-ADDRESS"""
199 if len(data) < 8:
200 return None
201

203 if family != 1:
204 return None
205
206 port = struct.unpack(">H", data[2:4])[0]
207 ip = ".".join(str(b) for b in data[4:8])
208
209 return ip, port
