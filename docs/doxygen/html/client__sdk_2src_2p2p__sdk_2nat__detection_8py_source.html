55class STUNClient:
56 """
57 STUN protocol client for NAT detection.
58
59 Implements a subset of RFC 5389 (STUN) for NAT type discovery.
60 """
61
62 # STUN message types
63 BINDING_REQUEST = 0x0001
64 BINDING_RESPONSE = 0x0101
65
66 # STUN attributes
67 ATTR_MAPPED_ADDRESS = 0x0001
68 ATTR_XOR_MAPPED_ADDRESS = 0x0020
69 ATTR_CHANGE_REQUEST = 0x0003
70 ATTR_CHANGED_ADDRESS = 0x0004

72 def __init__(self, stun_server: str, stun_port: int = 3478):
73 """
74 Initialize STUN client.
75
76 Args:
77 stun_server: STUN server hostname or IP
78 stun_port: STUN server port (default 3478)
79 """
80 self.stun_server = stun_server
81 self.stun_port = stun_port
82 self.timeout = 5.0

84 def _pack_stun_request(self, request_type: int = BINDING_REQUEST) -> bytes:
85 """Pack a STUN request message."""
86 # STUN magic cookie and header
87 magic_cookie = bytes.fromhex("2112A442")
88 transaction_id = b"0123456789ab" # 12 bytes transaction ID
89
90 # Message header: type (2) + length (2) + magic cookie (4) + transaction_id (12)
91 header = request_type.to_bytes(2, "big")
92 header += (0).to_bytes(2, "big") # Length = 0 (no attributes)
93 header += magic_cookie
94 header += transaction_id
95
96 return header
97
98 def _unpack_stun_response(self, data: bytes) -> Tuple[Optional[str], Optional[int]]:
99 """
100 Unpack STUN response and extract mapped address.
101
102 Returns:
103 Tuple of (ip, port) or (None, None) if parsing fails
104 """
105 if len(data) < 20:
106 return None, None
107
108 message_type = int.from_bytes(data[0:2], "big")
109 message_len = int.from_bytes(data[2:4], "big")
110 magic_cookie = data[4:8]
111
112 if magic_cookie != bytes.fromhex("2112A442"):
113 logger.warning("Invalid STUN magic cookie")
114 return None, None
115
116 # Parse attributes
117 idx = 20
118 while idx < len(data):
119 if idx + 4 > len(data):
120 break
121
122 attr_type = int.from_bytes(data[idx:idx + 2], "big")
123 attr_len = int.from_bytes(data[idx + 2:idx + 4], "big")
124 idx += 4
125
126 # Padding to 4-byte boundary
127 padding = (4 - (attr_len % 4)) % 4
128
129 if idx + attr_len > len(data):
130 break
131
132 attr_value = data[idx:idx + attr_len]
133 idx += attr_len + padding
134
135 # Parse XOR-MAPPED-ADDRESS (preferred)
136 if attr_type == self.ATTR_XOR_MAPPED_ADDRESS and attr_len >= 8:
137 family = attr_value[1]
138 if family == 0x01: # IPv4
139 xored_port = int.from_bytes(attr_value[2:4], "big")
140 xored_ip = int.from_bytes(attr_value[4:8], "big")
141 magic = int.from_bytes(magic_cookie, "big")
142
143 port = xored_port ^ (magic >> 16)
144 ip = xored_ip ^ magic
145 ip_str = socket.inet_ntoa(ip.to_bytes(4, "big"))
146 return ip_str, port
147
148 # Parse MAPPED-ADDRESS (fallback)
149 elif attr_type == self.ATTR_MAPPED_ADDRESS and attr_len >= 8:
150 family = attr_value[1]
151 if family == 0x01: # IPv4
152 port = int.from_bytes(attr_value[2:4], "big")
153 ip = int.from_bytes(attr_value[4:8], "big")
154 ip_str = socket.inet_ntoa(ip.to_bytes(4, "big"))
155 return ip_str, port
156
157 return None, None

159 async def send_request(self) -> Tuple[Optional[str], Optional[int]]:
160 """
161 Send STUN binding request and get mapped address.
162
163 Returns:
164 Tuple of (public_ip, public_port) or (None, None) on failure
165 """
166 sock = None
167 try:
168 # Create UDP socket
169 sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
170 sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
171 sock.setblocking(False)
172
173 # Get local address
174 sock.bind(("", 0))
175 local_port = sock.getsockname()[1]
176
177 # Send STUN request
178 request = self._pack_stun_request()
179
180 # Resolve server address
181 loop = asyncio.get_event_loop()
182 server_addr = await loop.getaddrinfo(
183 self.stun_server, self.stun_port, proto=socket.IPPROTO_UDP
184 )
185 target = server_addr[0][4]
186
187 sock.sendto(request, target)
188
189 # Wait for response with timeout
190 loop = asyncio.get_event_loop()
191 try:
192 data, _ = await asyncio.wait_for(
193 loop.sock_recvfrom(sock, 512), timeout=self.timeout
194 )
195 except asyncio.TimeoutError:
196 logger.warning(f"STUN request to {self.stun_server} timed out")
197 return None, None
198
199 # Parse response
200 public_ip, public_port = self._unpack_stun_response(data)
201
202 if public_ip and public_port:
203 logger.info(
204 f"STUN response: {public_ip}:{public_port} (local port: {local_port})"
205 )
206
207 return public_ip, public_port
208
209 except OSError as e:
210 logger.error(f"Socket error during STUN request: {e}")
211 return None, None
212 finally:
213 if sock:
214 sock.close()

217async def detect_nat_type(
218 stun_server: str = "stun.l.google.com", stun_port: int = 19302
219) -> NATDetectionResult:
220 """
221 Detect NAT type using STUN server.
222
223 This is a simplified detection that determines:
224 - Public IP (no NAT)
225 - Symmetric NAT (different port for each destination)
226 - Cone NAT (same port for all destinations)
227 - Blocked (UDP filtered)
228
229 Full RFC 3489 detection requires multiple STUN servers with
230 specific IP address ranges.
231
232 Args:
233 stun_server: STUN server hostname
234 stun_port: STUN server port
235
236 Returns:
237 NATDetectionResult with detected type and address info
238 """
239 client = STUNClient(stun_server, stun_port)
240
241 # Test 1: Basic binding request
242 public_ip, public_port = await client.send_request()
243
244 if public_ip is None:
245 # Could be blocked or network issue
246 return NATDetectionResult(nat_type=NATType.BLOCKED)
247
248 # Get local IP
249 try:
250 local_ip = socket.gethostbyname(socket.gethostname())
251 except OSError:
252 # Failed to resolve hostname, use loopback
253 local_ip = "127.0.0.1"
254
255 # Check if we have a public IP
256 # Simple heuristic: if public IP matches local IP, likely no NAT
257 # This is not perfect but works for basic detection
258 try:
259 # Try to determine if we're behind NAT by checking if public IP is private range
260 import ipaddress
261
262 pub_addr = ipaddress.ip_address(public_ip)
263 if not pub_addr.is_private:
264 # We got a public IP, but need to check if it's our actual IP
265 # For now, assume if we get a response, we can do P2P
266 return NATDetectionResult(
267 nat_type=NATType.PUBLIC_IP,
268 public_ip=public_ip,
269 public_port=public_port,
270 local_ip=local_ip,
271 )
272 except (ValueError, ipaddress.AddressValueError):
273 # Invalid IP address format, continue with default assumption
274 pass
275
276 # Test 2: Check for symmetric NAT by sending to another server
277 # (simplified - would need second STUN server for proper detection)
278
279 # Default assumption for single server detection
280 return NATDetectionResult(
281 nat_type=NATType.RESTRICTED_CONE,
282 public_ip=public_ip,
283 public_port=public_port,
284 local_ip=local_ip,
285 )
