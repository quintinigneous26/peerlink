16def encode_varint(value: int) -> bytes:
17 """
18 将无符号整数编码为 varint 格式
19
20 Args:
21 value: 要编码的无符号整数
22
23 Returns:
24 varint 编码的字节序列
25
26 Raises:
27 ValueError: 如果值为负数
28
29 Examples:
30 >>> encode_varint(0)
31 b'\\x00'
32 >>> encode_varint(3)
33 b'\\x03'
34 >>> encode_varint(300)
35 b'\\xac\\x02'
36 """
37 if value < 0:
38 raise ValueError("varint 只支持无符号整数")
39
40 if value == 0:
41 return b'\x00'
42
43 result = bytearray()
44 while value > 0:
45 # 取低 7 位
46 byte = value & 0x7f
47 value >>= 7
48 # 如果还有更多数据，设置最高位为 1
49 if value > 0:
50 byte |= 0x80
51 result.append(byte)
52
53 return bytes(result)

56def decode_varint(data: bytes, offset: int = 0) -> tuple[int, int]:
57 """
58 从 varint 格式解码无符号整数
59
60 Args:
61 data: 包含 varint 编码的字节序列
62 offset: 开始解码的位置偏移量
63
64 Returns:
65 (解码后的值, 消耗的字节数)
66
67 Raises:
68 ValueError: 如果 varint 格式无效或数据不足
69
70 Examples:
71 >>> decode_varint(b'\\x00')
72 (0, 1)
73 >>> decode_varint(b'\\xac\\x02')
74 (300, 2)
75 """
76 if offset >= len(data):
77 raise ValueError("数据不足，无法解码 varint")
78
79 result = 0
80 shift = 0
81 bytes_consumed = 0
82
83 for i in range(offset, len(data)):
84 byte = data[i]
85 bytes_consumed += 1
86
87 # 取低 7 位，累加到结果
88 result |= (byte & 0x7f) << shift
89
90 # 如果最高位为 0，表示这是最后一个字节
91 if (byte & 0x80) == 0:
92 return result, bytes_consumed
93
94 shift += 7
95
96 # 防止 varint 过长导致的问题 (最多 9 字节对应 64 位)
97 if bytes_consumed > 9:
98 raise ValueError("varint 过长")
99
100 raise ValueError("varint 不完整")

111def encode_message(data: str) -> bytes:
112 """
113 编码 multistream-select 消息
114
115 格式: varint(长度) + UTF-8字符串 + \n
116
117 Args:
118 data: 要编码的字符串
119
120 Returns:
121 编码后的字节序列
122
123 Examples:
124 >>> encode_message("na")
125 b'\\x03na\\n'
126 >>> encode_message("/multistream/1.0.0")
127 b'\\x13/multistream/1.0.0\\n'
128 """
129 # 将字符串编码为 UTF-8，并添加换行符
130 message_bytes = (data + '\n').encode('utf-8')
131
132 # 添加 varint 长度前缀
133 length_prefix = encode_varint(len(message_bytes))
134
135 return length_prefix + message_bytes

138def decode_message(data: bytes) -> str:
139 """
140 解码 multistream-select 消息
141
142 Args:
143 data: 要解码的字节序列
144
145 Returns:
146 解码后的字符串 (不含换行符)
147
148 Raises:
149 ValueError: 如果消息格式无效
150
151 Examples:
152 >>> decode_message(b'\\x03na\\n')
153 'na'
154 >>> decode_message(b'\\x13/multistream/1.0.0\\n')
155 '/multistream/1.0.0'
156 """
157 # 解码 varint 长度前缀
158 length, varint_size = decode_varint(data)
159
160 # 验证数据长度
161 if len(data) < varint_size + length:
162 raise ValueError(f"数据长度不足: 期望 {varint_size + length}, 实际 {len(data)}")
163
164 # 提取消息内容
165 message_bytes = data[varint_size:varint_size + length]
166
167 # 解码为字符串并去除换行符
168 message = message_bytes.decode('utf-8').strip()
169
170 return message

205def is_valid_protocol_id(protocol_id: str) -> bool:
206 """
207 验证协议 ID 是否有效
208
209 有效的协议 ID 应该:
210 1. 以 '/' 开头
211 2. 不包含空格或控制字符
212 3. 不为空
213
214 Args:
215 protocol_id: 要验证的协议 ID
216
217 Returns:
218 是否有效
219 """
220 if not protocol_id:
221 return False
222
223 if not protocol_id.startswith('/'):
224 return False
225
226 # 检查是否包含无效字符
227 for char in protocol_id:
228 if ord(char) < 0x20: # 控制字符
229 return False
230 if char in (' ', '\t', '\n', '\r'):
231 return False
232
233 return True
