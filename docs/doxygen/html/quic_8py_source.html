147 async def read(self, n: int = -1) -> bytes:
148 """
149 读取数据
150

152 n: 读取字节数，-1 表示读取所有
153

155 读取的数据

184 raise ConnectionError("连接已关闭")

293 def _setup_protocol_callbacks(self) -> None:
294 """设置协议回调"""
295 if not self._protocol:
296 return

299 self._protocol.event_handler = self._handle_quic_event

301 def _handle_quic_event(self, event: 'QuicEvent') -> None:
302 """
303 处理 QUIC 事件
304
305 Args:
306 event: QUIC 事件
307 """
308 if isinstance(event, HandshakeCompleted):
309 # 握手完成
310 logger.debug("QUIC 握手完成")
311 self._handshake_complete.set()
312
313 # 检查是否支持 0-RTT
314 if event.session_ticket:
315 self._session_ticket = event.session_ticket
316 self._supports_0rtt = True
317
318 elif isinstance(event, StreamDataReceived):
319 # 流数据接收
320 if event.stream_id == self._stream_id:
321 # 主流数据
322 asyncio.create_task(self._read_queue.put(event.data))
323
324 # 处理多流数据
325 stream = self._active_streams.get(event.stream_id)
326 if stream:
327 stream._handle_data(event.data, event.end_stream)
328
329 async def _get_or_create_stream(self) -> int:
330 """获取或创建流"""
331 if self._stream_id:
332 return self._stream_id
333
334 # 创建新的双向流
335 stream = await self.open_stream(QuicStreamType.BIDI)
336 self._stream_id = stream.stream_id
337 return self._stream_id

656 async def dial(self, addr: str) -> QUICConnection:
657 """
658 建立到指定地址的连接
659
660 Args:
661 addr: 目标地址（格式: host:port 或 /ip4/ip/udp/port/quic-v1）
662
663 Returns:
664 QUIC 连接
665 """
666 if self._closed:
667 raise TransportError("传输已关闭")
668
669 # 解析地址
670 host, port = self._parse_address(addr)
671
672 # 创建配置
673 configuration = self._create_client_configuration(host)
674
675 # 建立 QUIC 连接
676 try:
677 logger.debug(f"QUIC 连接中: {host}:{port}")
678
679 # 使用 aioquic 连接
680 protocol = await connect(
681 host,
682 port,
683 configuration=configuration,
684 create_protocol=self._create_protocol,
685 )
686
687 # 创建连接对象
688 conn = QUICConnection(protocol, is_initiator=True)
689
690 # 等待握手完成
691 await conn.wait_handshake()
692
693 # 保存连接
694 self._connections.append(conn)
695
696 # 设置地址
697 conn._remote_address = (host, port)
698 conn._local_address = protocol._transport.get_extra_info('sockname')
699
700 logger.info(f"QUIC 连接成功: {host}:{port}")
701 return conn
702
703 except Exception as e:
704 raise TransportError(f"QUIC 连接失败: {e}") from e

844 def _parse_address(self, addr: str) -> Tuple[str, int]:
845 """
846 解析地址
847
848 支持格式:
849 - host:port
850 - /ip4/ip/udp/port/quic-v1
851 - /ip6/ip/udp/port/quic-v1
852
853 Returns:
854 (host, port) 元组
855 """
856 # multiaddr 格式
857 if addr.startswith('/'):
858 parts = addr.strip('/').split('/')
859 if len(parts) >= 4:
860 proto = parts[0] # ip4 or ip6
861 host = parts[1]
862 transport = parts[2] # udp
863 port = int(parts[3]) if len(parts) > 3 else 4242
864 return (host, port)
865
866 # host:port 格式
867 if ':' in addr:
868 host, port_str = addr.rsplit(':', 1)
869 try:
870 return (host, int(port_str))
871 except ValueError:
872 pass
873
874 # 默认端口
875 return (addr, 4242)

877 def _generate_selfsigned_cert(self) -> Tuple[Any, Any]:
878 """
879 生成自签名证书
880
881 Returns:
882 (certificate, private_key) 元组
883 """
884 from cryptography import x509
885 from cryptography.x509.oid import NameOID
886 from cryptography.hazmat.primitives import hashes
887 from cryptography.hazmat.backends import default_backend
888 from cryptography.hazmat.primitives.asymmetric import rsa
889 from cryptography.hazmat.primitives import serialization
890 import datetime
891
892 # 生成私钥
893 private_key = rsa.generate_private_key(
894 public_exponent=65537,
895 key_size=2048,
896 backend=default_backend()
897 )
898
899 # 创建证书
900 subject = issuer = x509.Name([
901 x509.NameAttribute(NameOID.COUNTRY_NAME, "CN"),
902 x509.NameAttribute(NameOID.STATE_OR_PROVINCE_NAME, "Beijing"),
903 x509.NameAttribute(NameOID.LOCALITY_NAME, "Beijing"),
904 x509.NameAttribute(NameOID.ORGANIZATION_NAME, "libp2p-python"),
905 x509.NameAttribute(NameOID.COMMON_NAME, "localhost"),
906 ])
907
908 cert = x509.CertificateBuilder().subject_name(
909 subject
910 ).issuer_name(
911 issuer
912 ).public_key(
913 private_key.public_key()
914 ).serial_number(
915 x509.random_serial_number()
916 ).not_valid_before(
917 datetime.datetime.utcnow()
918 ).not_valid_after(
919 datetime.datetime.utcnow() + datetime.timedelta(days=365)
920 ).add_extension(
921 x509.SubjectAlternativeName([
922 x509.DNSName("localhost"),
923 x509.IPAddress(ip_address("127.0.0.1")),
924 x509.IPAddress(ip_address("::1")),
925 ]),
926 critical=False,
927 ).sign(private_key, hashes.SHA256(), default_backend())
928
929 # 序列化证书和私钥
930 cert_pem = cert.public_bytes(serialization.Encoding.PEM)
931 key_pem = private_key.private_bytes(
932 encoding=serialization.Encoding.PEM,
933 format=serialization.PrivateFormat.TraditionalOpenSSL,
934 encryption_algorithm=serialization.NoEncryption()
935 )
936
937 return (cert_pem, key_pem)
