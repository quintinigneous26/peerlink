2P2P事件总线系统
3
4提供增强的事件驱动架构，支持基于主题的事件订阅和异步事件处理。

68 ERROR: 错误事件
69 - 各类错误信息

162class EventSubscriber:
163 """Event subscriber with filtering options"""
164
165 def __init__(
166 self,
167 handler: EventHandler,
168 filter_topics: Optional[Set[EventTopic]] = None,
169 filter_types: Optional[Set[P2PEventType]] = None,
170 filter_peer: Optional[str] = None,
171 ):
172 """
173 Initialize event subscriber.
174
175 Args:
176 handler: Async event handler function
177 filter_topics: Optional topic filter (only these topics)
178 filter_types: Optional event type filter
179 filter_peer: Optional peer ID filter (only events for this peer)
180 """
181 self.handler = handler
182 self.filter_topics = filter_topics
183 self.filter_types = filter_types
184 self.filter_peer = filter_peer
185 self._id = id(self)
186
187 def __hash__(self) -> int:
188 return self._id
189
190 def matches(self, event: P2PEvent) -> bool:
191 """Check if this subscriber should receive the event"""
192 if self.filter_topics and event.topic not in self.filter_topics:
193 return False
194
195 if self.filter_types and event.event_type not in self.filter_types:
196 return False
197
198 if self.filter_peer and event.peer_id != self.filter_peer:
199 return False
200
201 return True

206class EventBus:
207 """
208 Async event bus for P2P engine.
209
210 Provides topic-based event publishing and subscription with
211 async handling and filtering capabilities.
212 """
213
214 def __init__(
215 self,
216 max_queue_size: int = 1000,
217 worker_count: int = 2,
218 enable_metrics: bool = True,
219 ):
220 """
221 Initialize event bus.
222
223 Args:
224 max_queue_size: Maximum event queue size
225 worker_count: Number of event worker tasks
226 enable_metrics: Track event metrics
227 """
228 self._subscribers: Dict[EventTopic, List[EventSubscriber]] = defaultdict(list)
229 self._global_subscribers: List[EventSubscriber] = []
230
231 self._queue: Optional[asyncio.Queue] = None
232 self._max_queue_size = max_queue_size
233 self._worker_count = worker_count
234 self._enable_metrics = enable_metrics
235
236 self._running = False
237 self._workers: List[asyncio.Task] = []
238 self._lock = Lock()
239
240 # Metrics
241 self._events_published = 0
242 self._events_processed = 0
243 self._events_dropped = 0
244 self._handler_errors = 0
245 self._start_time: Optional[float] = None
246
247 # Thread-safe sync wrapper support
248 self._executor = Optional[ThreadPoolExecutor]
249
250 async def start(self) -> None:
251 """Start the event bus"""
252 if self._running:
253 return
254
255 self._running = True
256 self._queue = asyncio.Queue(maxsize=self._max_queue_size)
257 self._start_time = time.time()
258 self._executor = ThreadPoolExecutor(max_workers=2)
259
260 # Start worker tasks
261 for i in range(self._worker_count):
262 worker = asyncio.create_task(self._worker(f"worker-{i}"))
263 self._workers.append(worker)
264
265 logger.info(f"Event bus started with {self._worker_count} workers")
266
267 async def stop(self) -> None:
268 """Stop the event bus"""
269 if not self._running:
270 return
271
272 self._running = False
273
274 # Cancel workers
275 for worker in self._workers:
276 worker.cancel()
277
278 # Wait for workers to finish
279 await asyncio.gather(*self._workers, return_exceptions=True)
280 self._workers.clear()
281
282 # Shutdown executor
283 if self._executor:
284 self._executor.shutdown(wait=True)
285
286 logger.info("Event bus stopped")
287
288 def subscribe(
289 self,
290 topic: EventTopic,
291 handler: EventHandler,
292 filter_types: Optional[Set[P2PEventType]] = None,
293 filter_peer: Optional[str] = None,
294 ) -> EventSubscriber:
295 """
296 Subscribe to events on a topic.
297
298 Args:
299 topic: Event topic to subscribe to
300 handler: Async event handler function
301 filter_types: Optional event type filter
302 filter_peer: Optional peer ID filter
303
304 Returns:
305 EventSubscriber that can be used to unsubscribe
306 """
307 subscriber = EventSubscriber(
308 handler=handler,
309 filter_topics={topic},
310 filter_types=filter_types,
311 filter_peer=filter_peer,
312 )
313
314 with self._lock:
315 self._subscribers[topic].append(subscriber)
316
317 logger.debug(f"Subscribed to topic: {topic.value}")
318 return subscriber
319
320 def subscribe_all(
321 self,
322 handler: EventHandler,
323 filter_topics: Optional[Set[EventTopic]] = None,
324 filter_types: Optional[Set[P2PEventType]] = None,
325 filter_peer: Optional[str] = None,
326 ) -> EventSubscriber:
327 """
328 Subscribe to all events (or filtered topics).
329
330 Args:
331 handler: Async event handler function
332 filter_topics: Optional topic filter (default: all topics)
333 filter_types: Optional event type filter
334 filter_peer: Optional peer ID filter
335
336 Returns:
337 EventSubscriber that can be used to unsubscribe
338 """
339 subscriber = EventSubscriber(
340 handler=handler,
341 filter_topics=filter_topics,
342 filter_types=filter_types,
343 filter_peer=filter_peer,
344 )
345
346 with self._lock:
347 self._global_subscribers.append(subscriber)
348
349 logger.debug("Subscribed to all topics")
350 return subscriber
351
352 def unsubscribe(self, subscriber: EventSubscriber) -> bool:
353 """
354 Unsubscribe a subscriber.
355
356 Args:
357 subscriber: The subscriber to remove
358
359 Returns:
360 True if subscriber was found and removed
361 """
362 with self._lock:
363 # Remove from topic-specific subscribers
364 for topic, subs in self._subscribers.items():
365 if subscriber in subs:
366 subs.remove(subscriber)
367 logger.debug(f"Unsubscribed from topic: {topic.value}")
368 return True
369
370 # Remove from global subscribers
371 if subscriber in self._global_subscribers:
372 self._global_subscribers.remove(subscriber)
373 logger.debug("Unsubscribed from all topics")
374 return True
375
376 return False

378 async def publish(self, event: P2PEvent) -> bool:
379 """
380 Publish an event.
381
382 Args:
383 event: The event to publish
384
385 Returns:
386 True if event was queued, False if queue is full
387 """
388 if not self._running:
389 logger.warning("Event bus not running, event dropped")
390 return False
391
392 try:
393 self._queue.put_nowait(event)
394 self._events_published += 1
395 return True
396 except asyncio.QueueFull:
397 self._events_dropped += 1
398 logger.warning(f"Event queue full, dropping event: {event.event_type.value}")
399 return False
400
401 async def publish_sync(self, event: P2PEvent) -> None:
402 """
403 Publish an event and wait for it to be queued.
404
405 Args:
406 event: The event to publish
407 """
408 if not self._running:
409 logger.warning("Event bus not running, event dropped")
410 return
411
412 await self._queue.put(event)
413 self._events_published += 1
414
415 def publish_nowait(self, event: P2PEvent) -> bool:
416 """
417 Publish an event without waiting (non-async).
418
419 Useful for publishing from sync contexts.
420
421 Args:
422 event: The event to publish
423
424 Returns:
425 True if event was queued, False otherwise
426 """
427 if not self._running or not self._queue:
428 return False
429
430 try:
431 self._queue.put_nowait(event)
432 self._events_published += 1
433 return True
434 except asyncio.QueueFull:
435 self._events_dropped += 1
436 return False
437
438 async def _worker(self, name: str) -> None:
439 """Event worker task"""
440 logger.debug(f"Event worker {name} started")
441
442 while self._running:
443 try:
444 event = await asyncio.wait_for(
445 self._queue.get(),
446 timeout=1.0,
447 )
448
449 await self._process_event(event)
450 self._events_processed += 1
451
452 except asyncio.TimeoutError:
453 continue
454 except asyncio.CancelledError:
455 break
456 except Exception as e:
457 logger.error(f"Worker {name} error: {e}")
458 self._handler_errors += 1
459
460 logger.debug(f"Event worker {name} stopped")
461
462 async def _process_event(self, event: P2PEvent) -> None:
463 """Process a single event"""
464 # Get matching subscribers
465 matching_subscribers = []
466
467 with self._lock:
468 # Topic-specific subscribers
469 for subscriber in self._subscribers.get(event.topic, []):
470 if subscriber.matches(event):
471 matching_subscribers.append(subscriber)
472
473 # Global subscribers
474 for subscriber in self._global_subscribers:
475 if subscriber.matches(event):
476 matching_subscribers.append(subscriber)
477
478 # Notify all matching subscribers
479 if matching_subscribers:
480 tasks = []
481 for subscriber in matching_subscribers:
482 task = asyncio.create_task(
483 self._call_handler(subscriber.handler, event),
484 name=f"handler-{subscriber._id}",
485 )
486 tasks.append(task)
487
488 # Wait for all handlers with timeout
489 if tasks:
490 await asyncio.gather(*tasks, return_exceptions=True)
491
492 async def _call_handler(self, handler: EventHandler, event: P2PEvent) -> None:
493 """Call an event handler with error handling"""
494 try:
495 await handler(event)
496 except Exception as e:
497 logger.error(f"Event handler error for {event.event_type.value}: {e}")
498 self._handler_errors += 1
499
500 def get_metrics(self) -> Dict[str, Any]:
501 """Get event bus metrics"""
502 uptime = time.time() - self._start_time if self._start_time else 0
503
504 return {
505 "uptime_seconds": uptime,
506 "events_published": self._events_published,
507 "events_processed": self._events_processed,
508 "events_dropped": self._events_dropped,
509 "handler_errors": self._handler_errors,
510 "queue_size": self._queue.qsize() if self._queue else 0,
511 "max_queue_size": self._max_queue_size,
512 "subscribers": sum(len(subs) for subs in self._subscribers.values()) + len(self._global_subscribers),
513 "running": self._running,
514 }
515
516 def reset_metrics(self) -> None:
517 """Reset event metrics"""
518 with self._lock:
519 self._events_published = 0
520 self._events_processed = 0
521 self._events_dropped = 0
522 self._handler_errors = 0
