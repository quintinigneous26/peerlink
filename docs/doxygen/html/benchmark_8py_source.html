55 async def benchmark_stream_creation(
56 self,
57 session_factory: Callable,
58 target_latency_ms: float = 2.0,
59 iterations: int = 1000,
60 ) -> BenchmarkResult:
61 """
62 Benchmark stream creation latency.
63
64 Target: < 2ms per stream
65 """
66 latencies = []
67
68 for _ in range(iterations):
69 session = session_factory()
70
71 start = time.perf_counter()
72 stream = await session.open_stream()
73 end = time.perf_counter()
74
75 latencies.append((end - start) * 1000) # Convert to ms
76
77 await stream.close()
78 await session.close()
79
80 avg = statistics.mean(latencies)
81 minimum = min(latencies)
82 maximum = max(latencies)
83 p50 = statistics.median(latencies)
84 p95 = statistics.quantiles(latencies, n=20)[18] # 95th percentile
85 p99 = statistics.quantiles(latencies, n=100)[98] # 99th percentile
86
87 result = BenchmarkResult(
88 name="stream_creation",
89 iterations=iterations,
90 total_time=sum(latencies) / 1000,
91 avg_time=avg,
92 min_time=minimum,
93 max_time=maximum,
94 p50_time=p50,
95 p95_time=p95,
96 p99_time=p99,
97 throughput=1000 / avg, # streams per second
98 metadata={
99 "target_latency_ms": target_latency_ms,
100 "passed": avg < target_latency_ms,
101 },
102 )
103
104 self.results.append(result)
105 return result

107 async def benchmark_message_throughput(
108 self,
109 stream_factory: Callable,
110 message_size: int = 1024, # 1KB
111 target_mbps: float = 200.0,
112 duration_sec: float = 5.0,
113 ) -> BenchmarkResult:
114 """
115 Benchmark message throughput.
116
117 Target: > 200 Mbps
118 """
119 stream = await stream_factory()
120
121 total_bytes = 0
122 start_time = time.time()
123 times = []
124
125 while time.time() - start_time < duration_sec:
126 data = b"x" * message_size
127
128 write_start = time.perf_counter()
129 await stream.write(data)
130 write_end = time.perf_counter()
131
132 times.append((write_end - write_start) * 1000)
133 total_bytes += message_size
134
135 await stream.close()
136
137 total_time = time.time() - start_time
138 throughput_mbps = (total_bytes * 8) / (total_time * 1_000_000)
139
140 avg_time = statistics.mean(times)
141
142 result = BenchmarkResult(
143 name="message_throughput",
144 iterations=len(times),
145 total_time=total_time,
146 avg_time=avg_time,
147 min_time=min(times),
148 max_time=max(times),
149 p50_time=statistics.median(times),
150 p95_time=statistics.quantiles(times, n=20)[18],
151 p99_time=statistics.quantiles(times, n=100)[98],
152 throughput=throughput_mbps,
153 metadata={
154 "message_size": message_size,
155 "target_mbps": target_mbps,
156 "passed": throughput_mbps > target_mbps,
157 },
158 )
159
160 self.results.append(result)
161 return result

163 async def benchmark_concurrent_streams(
164 self,
165 session_factory: Callable,
166 target_streams: int = 1000,
167 messages_per_stream: int = 10,
168 ) -> BenchmarkResult:
169 """
170 Benchmark concurrent stream handling.
171
172 Target: > 1000 concurrent streams
173 """
174 session = session_factory()
175
176 async def use_stream(stream_id: int):
177 """Single stream workload."""
178 stream = await session.open_stream()
179
180 for _ in range(messages_per_stream):
181 data = b"test" * 256 # 1KB
182 await stream.write(data)
183 received = await stream.read(len(data))
184
185 await stream.close()
186 return stream_id
187
188 start_time = time.time()
189
190 # Create concurrent streams
191 tasks = [use_stream(i) for i in range(target_streams)]
192 await asyncio.gather(*tasks)
193
194 total_time = time.time() - start_time
195
196 result = BenchmarkResult(
197 name="concurrent_streams",
198 iterations=target_streams,
199 total_time=total_time,
200 avg_time=total_time / target_streams,
201 min_time=0, # Not applicable for concurrent test
202 max_time=0,
203 p50_time=0,
204 p95_time=0,
205 p99_time=0,
206 throughput=target_streams / total_time, # streams/sec
207 metadata={
208 "messages_per_stream": messages_per_stream,
209 "target_streams": target_streams,
210 "passed": True, # If we completed, we passed
211 },
212 )
213
214 self.results.append(result)
215 await session.close()
216 return result

218 async def benchmark_memory_usage(
219 self,
220 session_factory: Callable,
221 streams: int = 100,
222 ) -> BenchmarkResult:
223 """
224 Benchmark memory usage for multiple streams.
225 """
226 tracemalloc.start()
227
228 # Baseline memory
229 snapshot1 = tracemalloc.take_snapshot()
230
231 # Create streams
232 session = session_factory()
233 active_streams = []
234
235 for _ in range(streams):
236 stream = await session.open_stream()
237 active_streams.append(stream)
238
239 # Peak memory
240 snapshot2 = tracemalloc.take_snapshot()
241
242 # Cleanup
243 for stream in active_streams:
244 await stream.close()
245 await session.close()
246
247 # Final memory
248 snapshot3 = tracemalloc.take_snapshot()
249
250 tracemalloc.stop()
251
252 # Calculate memory usage
253 top_stats = snapshot2.compare_to(snapshot1, 'lineno')
254 total_kb = sum(stat.size_diff for stat in top_stats) / 1024
255 per_stream_kb = total_kb / streams
256
257 result = BenchmarkResult(
258 name="memory_usage",
259 iterations=streams,
260 total_time=0,
261 avg_time=per_stream_kb,
262 min_time=0,
263 max_time=0,
264 p50_time=0,
265 p95_time=0,
266 p99_time=0,
267 throughput=0,
268 metadata={
269 "total_memory_kb": total_kb,
270 "per_stream_kb": per_stream_kb,
271 "streams": streams,
272 },
273 )
274
275 self.results.append(result)
276 return result
