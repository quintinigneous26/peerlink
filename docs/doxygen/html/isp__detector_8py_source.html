84 async def detect(self) -> ISP:
85 """
86 检测当前运营商
87
88 步骤：
89 1. 通过 STUN 获取公网 IP
90 2. 根据 IP 段匹配运营商
91 """
92 logger.info("开始运营商检测...")
93
94 # 尝试多个 STUN 服务器
95 for server in self.stun_client.servers:
96 parts = server.split(":")
97 host = parts[0]
98 port = int(parts[1]) if len(parts) > 1 else 3478
99
100 response = await self.stun_client.binding_request(host, port)
101
102 if response.success and response.mapped_ip:
103 isp = self._identify_by_ip(response.mapped_ip)
104 logger.info(f"运营商检测完成: {isp.value}, 公网IP: {response.mapped_ip}")
105 return isp
106
107 logger.warning("运营商检测失败，使用默认值")
108 return ISP.UNKNOWN

110 def _identify_by_ip(self, ip: str) -> ISP:
111 """根据 IP 地址识别运营商"""
112
113 # 检查中国大陆运营商
114 for isp, patterns in ISP_IP_RANGES.items():
115 for pattern in patterns:
116 if re.match(pattern, ip):
117 return isp
118
119 # 检查香港运营商
120 for isp, patterns in HK_ISP_RANGES.items():
121 for pattern in patterns:
122 if re.match(pattern, ip):
123 return isp
124
125 # 检查新加坡运营商
126 for isp, patterns in SG_ISP_RANGES.items():
127 for pattern in patterns:
128 if re.match(pattern, ip):
129 return isp
130
131 # 根据私有 IP 判断
132 if ip.startswith("10.") or ip.startswith("192.168.") or ip.startswith("172."):
133 logger.debug("检测到私有 IP，可能在多层 NAT 后")
134 return ISP.UNKNOWN
135
136 return ISP.UNKNOWN
