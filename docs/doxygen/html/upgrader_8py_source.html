185class MuxedSession:
186 """
187 复用会话
188
189 管理多个复用流。
190 """
191
192 def __init__(
193 self,
194 conn: Connection,
195 protocol_id: str,
196 is_initiator: bool
197 ):
198 self._conn = conn
199 self._protocol_id = protocol_id
200 self._is_initiator = is_initiator
201 self._streams: dict[int, MuxedStream] = {}
202 self._next_stream_id = 1 if is_initiator else 2
203 self._closed = False
204 self._lock = asyncio.Lock()
205 self._incoming_queue: asyncio.Queue[MuxedStream] = asyncio.Queue()
206 self._accept_timeout = 30.0
207
208 @property
209 def protocol_id(self) -> str:
210 return self._protocol_id
211
212 @property
213 def is_initiator(self) -> bool:
214 return self._is_initiator

216 @property
217 def connection(self) -> Connection:
218 return self._conn

221 return self._closed

223 async def open_stream(self) -> MuxedStream:
224 """打开新的复用流"""
225 async with self._lock:
226 if self._closed:
227 raise TransportError("Session is closed")
228
229 stream_id = self._next_stream_id
230 self._next_stream_id += 2
231
232 # 创建流
233 reader = asyncio.StreamReader()
234 stream = MuxedStream(stream_id, reader, asyncio.StreamWriter())
235 self._streams[stream_id] = stream
236 return stream
237
238 async def accept_stream(self) -> MuxedStream:
239 """
240 接受传入的复用流
241
242 等待并返回下一个传入的流。此方法会阻塞直到有新流到达
243 或会话关闭。
244
245 Returns:
246 传入的复用流
247
248 Raises:
249 TransportError: 如果会话已关闭
250 """
251 while not self._closed:
252 try:
253 stream = await asyncio.wait_for(
254 self._incoming_queue.get(),
255 timeout=1.0
256 )
257 return stream
258 except asyncio.TimeoutError:
259 # 继续等待，除非会话已关闭
260 if self._closed:
261 raise TransportError("Session is closed")
262 continue
263 except asyncio.CancelledError:
264 # 任务被取消，优雅退出
265 raise TransportError("Accept stream cancelled")
266
267 raise TransportError("Session is closed")
268
269 async def _handle_incoming_stream(self, stream_id: int) -> MuxedStream:
270 """
271 处理传入流，由复用器协议调用
272
273 Args:
274 stream_id: 流ID
275
276 Returns:
277 创建的传入流
278 """
279 async with self._lock:
280 if self._closed:
281 raise TransportError("Session is closed")
282
283 # 创建传入流
284 reader = asyncio.StreamReader()
285 stream = MuxedStream(stream_id, reader, asyncio.StreamWriter())
286 self._streams[stream_id] = stream
287
288 # 通知 accept_stream 等待者
289 await self._incoming_queue.put(stream)
290
291 logger.debug(f"Accepted incoming stream: {stream_id}")
292 return stream
293
294 async def close(self) -> None:
295 """关闭会话"""
296 async with self._lock:
297 if self._closed:
298 return
299
300 self._closed = True
301
302 # 关闭所有流
303 for stream in list(self._streams.values()):
304 try:
305 await stream.close()
306 except Exception:
307 pass
308
309 self._streams.clear()
310
311 # 清理传入队列，唤醒等待的 accept_stream
312 try:
313 # 发送哨兵值唤醒等待者
314 while not self._incoming_queue.empty():
315 self._incoming_queue.get_nowait()
316 except Exception:
317 pass
318
319 # 关闭底层连接
320 await self._conn.close()

323class TransportUpgrader:
324 """
325 传输升级器
326
327 负责将原始传输连接升级为支持安全通道和流复用的完整连接。
328 """
329
330 def __init__(
331 self,
332 security_transports: Optional[List[SecurityTransport]] = None,
333 muxers: Optional[List[StreamMuxer]] = None
334 ):
335 """
336 初始化传输升级器
337
338 Args:
339 security_transports: 支持的安全传输列表（按优先级排序）
340 muxers: 支持的流复用器列表（按优先级排序）
341 """
342 self._security_transports = security_transports or []
343 self._muxers = muxers or []
344 self._upgrade_callbacks: List[Callable[[], Awaitable[None]]] = []
345
346 def add_security_transport(self, transport: SecurityTransport) -> None:
347 """添加安全传输"""
348 self._security_transports.append(transport)
349
350 def add_muxer(self, muxer: StreamMuxer) -> None:
351 """添加流复用器"""
352 self._muxers.append(muxer)
353
354 def on_upgrade(self, callback: Callable[[], Awaitable[None]]) -> None:
355 """注册升级完成回调"""
356 self._upgrade_callbacks.append(callback)
357
358 async def upgrade_inbound(
359 self,
360 conn: Connection,
361 peer_id: Optional[str] = None
362 ) -> "UpgradedConnection":
363 """
364 升级入站连接
365
366 Args:
367 conn: 原始连接
368 peer_id: 可选的对等节点ID
369
370 Returns:
371 升级后的连接
372 """
373 logger.debug("Upgrading inbound connection")
374
375 # 安全升级
376 secure_conn = await self._upgrade_security_inbound(conn, peer_id)
377
378 # 流复用升级
379 muxed_session = await self._upgrade_muxer(secure_conn, is_initiator=False)
380
381 # 触发回调
382 for callback in self._upgrade_callbacks:
383 try:
384 await callback()
385 except Exception as e:
386 logger.error(f"Upgrade callback error: {e}")
387
388 return UpgradedConnection(secure_conn, muxed_session)
389
390 async def upgrade_outbound(
391 self,
392 conn: Connection,
393 peer_id: str
394 ) -> "UpgradedConnection":
395 """
396 升级出站连接
397
398 Args:
399 conn: 原始连接
400 peer_id: 目标对等节点ID
401
402 Returns:
403 升级后的连接
404 """
405 logger.debug(f"Upgrading outbound connection to {peer_id}")
406
407 # 安全升级
408 secure_conn = await self._upgrade_security_outbound(conn, peer_id)
409
410 # 流复用升级
411 muxed_session = await self._upgrade_muxer(secure_conn, is_initiator=True)
412
413 # 触发回调
414 for callback in self._upgrade_callbacks:
415 try:
416 await callback()
417 except Exception as e:
418 logger.error(f"Upgrade callback error: {e}")
419
420 return UpgradedConnection(secure_conn, muxed_session)

422 async def _upgrade_security_inbound(
423 self,
424 conn: Connection,
425 peer_id: Optional[str]
426 ) -> SecureConnection:
427 """入站安全升级"""
428 if not self._security_transports:
429 # 无安全传输，返回包装的连接
430 return SecureConnection(
431 conn,
432 peer_id or "unknown",
433 "/insecure/1.0.0"
434 )
435
436 last_error = None
437 for security in self._security_transports:
438 try:
439 logger.debug(f"Attempting security upgrade: {security.get_protocol_id()}")
440 return await security.secure_inbound(conn, peer_id)
441 except Exception as e:
442 logger.debug(f"Security upgrade failed: {security.get_protocol_id()}: {e}")
443 last_error = e
444
445 raise TransportError(
446 f"All security transports failed. Last error: {last_error}"
447 )
448
449 async def _upgrade_security_outbound(
450 self,
451 conn: Connection,
452 peer_id: str
453 ) -> SecureConnection:
454 """出站安全升级"""
455 if not self._security_transports:
456 return SecureConnection(conn, peer_id, "/insecure/1.0.0")
457
458 last_error = None
459 for security in self._security_transports:
460 try:
461 logger.debug(f"Attempting security upgrade: {security.get_protocol_id()}")
462 return await security.secure_outbound(conn, peer_id)
463 except Exception as e:
464 logger.debug(f"Security upgrade failed: {security.get_protocol_id()}: {e}")
465 last_error = e
466
467 raise TransportError(
468 f"All security transports failed. Last error: {last_error}"
469 )
470
471 async def _upgrade_muxer(
472 self,
473 conn: SecureConnection,
474 is_initiator: bool
475 ) -> MuxedSession:
476 """流复用升级"""
477 if not self._muxers:
478 raise TransportError("No muxers available")
479
480 last_error = None
481 for muxer in self._muxers:
482 try:
483 logger.debug(f"Attempting muxer upgrade: {muxer.get_protocol_id()}")
484 return await muxer.create_session(conn, is_initiator)
485 except Exception as e:
486 logger.debug(f"Muxer upgrade failed: {muxer.get_protocol_id()}: {e}")
487 last_error = e
488
489 raise TransportError(
490 f"All muxers failed. Last error: {last_error}"
491 )
