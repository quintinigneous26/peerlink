212@app.middleware("http")
213async def rate_limit_middleware(request: Request, call_next):
214 """Apply rate limiting to all HTTP requests."""
215 limiter = get_rate_limiter()
216 allowed, info = limiter.check_rate_limit(request)
217
218 if not allowed:
219 raise HTTPException(
220 status_code=status.HTTP_429_TOO_MANY_REQUESTS,
221 detail={
222 "success": False,
223 "error": {
224 "code": "RATE_LIMIT_EXCEEDED",
225 "message": f"Rate limit exceeded. Retry after {info.get('retry_after', 60)} seconds",
226 "retry_after": info.get("retry_after", 60),
227 },
228 },
229 headers={"Retry-After": str(info.get("retry_after", 60))},
230 )
231
232 # Add rate limit headers to response
233 response = await call_next(request)
234 response.headers["X-RateLimit-Limit"] = str(limiter.config.requests_per_minute)
235 response.headers["X-RateLimit-Remaining"] = str(info.get("minute_remaining", limiter.config.requests_per_minute))
236 return response

251@app.post("/api/v1/did/generate", response_model=dict)
252async def generate_did(request: GenerateDIDRequest) -> dict:
253 """
254 Generate a new device identity.
255
256 Creates a new DID, generates a keypair, and registers the device.
257 The private key is only returned once.
258 """
259 # Generate DID
260 device_id = DIDGenerator.generate(request.device_type)
261
262 # Generate keypair
263 private_key, public_key = KeyGenerator.generate_keypair()
264 public_key_hex = KeyGenerator.serialize_public_key(public_key)
265 private_key_hex = KeyGenerator.serialize_private_key(private_key)
266
267 # Store device
268 storage = get_storage()
269 device_info = await storage.register_device(
270 device_id=device_id,
271 device_type=request.device_type,
272 public_key=public_key_hex,
273 capabilities=request.capabilities,
274 metadata=request.metadata,
275 )
276
277 response = APIResponse(
278 success=True,
279 data={
280 "device_id": device_info.device_id,
281 "device_type": device_info.device_type,
282 "public_key": public_key_hex,
283 "private_key": private_key_hex,
284 "capabilities": device_info.capabilities,
285 "created_at": device_info.created_at,
286 },
287 )
288
289 logger.info(f"Generated DID: {device_id}")
290 return response.to_dict()

293@app.post("/api/v1/did/verify", response_model=dict)
294async def verify_did(request: VerifyDIDRequest) -> dict:
295 """
296 Verify a device identity using signature challenge.
297
298 The device should sign the challenge with its private key.
299 """
300 storage = get_storage()
301 device = await storage.get_device(request.device_id)
302
303 if not device:
304 response = APIResponse(
305 success=False,
306 error={"code": "DEVICE_NOT_FOUND", "message": "Device not registered"},
307 )
308 return response.to_dict()
309
310 # Verify signature
311 try:
312 public_key = KeyGenerator.deserialize_public_key(device.public_key)
313 challenge_bytes = request.challenge.encode()
314 signature_bytes = bytes.fromhex(request.signature)
315
316 is_valid = KeyGenerator.verify(public_key, challenge_bytes, signature_bytes)
317 except Exception as e:
318 logger.warning(f"Signature verification error: {e}")
319 is_valid = False
320
321 response = APIResponse(
322 success=is_valid,
323 data={
324 "device_id": device.device_id,
325 "valid": is_valid,
326 "device_type": device.device_type,
327 },
328 error=None if is_valid else {"code": "INVALID_SIGNATURE", "message": "Signature verification failed"},
329 )
330
331 return response.to_dict()

469@app.get("/api/v1/devices", response_model=dict)
470async def list_devices(device_type: Optional[str] = None, online_only: bool = False) -> dict:
471 """
472 List devices.
473
474 Query parameters:
475 - device_type: Filter by device type
476 - online_only: Only return online devices
477 """
478 # Validate device_type if provided
479 if device_type:
480 device_type = device_type.lower().strip()
481 if device_type not in config.valid_device_types:
482 return APIResponse(
483 success=False,
484 error={"code": "INVALID_DEVICE_TYPE", "message": f"Must be one of: {config.valid_device_types}"},
485 ).to_dict()
486
487 storage = get_storage()
488
489 if device_type:
490 device_ids = await storage.get_devices_by_type(device_type)
491 elif online_only:
492 device_ids = list(await storage.get_online_devices())
493 else:
494 # Return all devices of all types
495 device_ids = []
496 for dt in config.valid_device_types:
497 device_ids.extend(await storage.get_devices_by_type(dt))
498
499 devices = []
500 for device_id in device_ids:
501 device = await storage.get_device(device_id)
502 if device:
503 is_online = await storage.is_online(device_id)
504 if not online_only or is_online:
505 devices.append({
506 "device_id": device.device_id,
507 "device_type": device.device_type,
508 "status": device.status.value,
509 "is_online": is_online,
510 "last_seen": device.last_seen,
511 })
512
513 response = APIResponse(
514 success=True,
515 data={
516 "devices": devices,
517 "count": len(devices),
518 },
519 )
520
521 return response.to_dict()
