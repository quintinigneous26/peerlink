72 def __init__(self, conn: 'p2p_engine.transport.Connection'):

76 async def read(self, n: int = -1) -> bytes:
77 """读取数据"""
78 if n == -1:
79 data = await self._queue.get()
80 else:
81 data = await self._queue.get()
82 return data[:n] if len(data) > n else data
83 return data
84
85 async def readexactly(self, n: int) -> bytes:
86 """读取精确字节数"""
87 data = bytearray()
88 while len(data) < n:
89 chunk = await self.read(n - len(data))
90 if not chunk:
91 raise asyncio.IncompleteReadError(data, n)
92 data.extend(chunk)
93 return bytes(data)

95 def feed(self, data: bytes) -> None:
96 """喂入数据（用于写入后读取）"""
