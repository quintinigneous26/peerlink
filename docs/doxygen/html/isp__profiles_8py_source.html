566def is_cross_border(local_isp: ISP, peer_isp: ISP) -> bool:
567 """判断是否跨境"""
568 local_profile = ISP_PROFILES.get(local_isp, ISP_PROFILES[ISP.UNKNOWN])
569 peer_profile = ISP_PROFILES.get(peer_isp, ISP_PROFILES[ISP.UNKNOWN])
570 return local_profile.region != peer_profile.region
