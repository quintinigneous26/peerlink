23def deep_merge(base: dict, override: dict) -> dict:
24 """
25 深度合并两个字典。
26
27 规则:
28 1. override 中的键值覆盖 base 中的同名键
29 2. 如果两个值都是字典，递归合并
30 3. 其他情况直接覆盖
31
32 Args:
33 base: 基础字典
34 override: 覆盖字典
35
36 Returns:
37 合并后的新字典（不修改原字典）
38 """
39 result = base.copy()
40
41 for key, value in override.items():
42 if key in result and isinstance(result[key], dict) and isinstance(value, dict):
43 result[key] = deep_merge(result[key], value)
44 else:
45 result[key] = value
46
47 return result
