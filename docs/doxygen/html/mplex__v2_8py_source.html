2Mplex Stream Multiplexer Implementation for libp2p
3
4Based on libp2p mplex specification:
5https://github.com/libp2p/specs/blob/master/mplex.md
6
7Protocol string: /mplex/6.7.0
8
9Features:
10- Stream multiplexing over a single connection
11- Simple framing with variable-length integer encoding
12- Backpressure via receive window

115# ==================== Variable-Length Integer ====================

117def write_uvarint(data: bytearray, value: int) -> None:
118 """
119 Write a variable-length integer to bytearray.
120
121 Uses unsigned LEB128 encoding.
122
123 Args:
124 data: Bytearray to write to
125 value: Integer value to encode
126 """
127 while value >= 0x80:
128 data.append((value & 0x7F) | 0x80)
129 value >>= 7
130 data.append(value)
131
132
133def read_uvarint(data: bytes, offset: int = 0) -> tuple[int, int]:
134 """
135 Read a variable-length integer from bytes.
136
137 Args:
138 data: Bytes to read from
139 offset: Starting offset
140
141 Returns:
142 Tuple of (value, bytes_consumed)
143
144 Raises:
145 MplexProtocolError: If encoding is invalid
146 """
147 value = 0
148 shift = 0
149 idx = offset
150
151 while idx < len(data):
152 byte = data[idx]
153 idx += 1
154 value |= (byte & 0x7F) << shift
155
156 if (byte & 0x80) == 0:
157 return value, idx - offset
158
159 shift += 7
160 if shift >= 64:
161 raise MplexProtocolError("Invalid uvarint encoding")
162
163 raise MplexProtocolError("Incomplete uvarint")

166# ==================== Mplex Frame (Optimized) ====================
167
168@dataclass
169class MplexFrame:
170 """Mplex frame structure."""
171 flag: MplexFlag
172 stream_id: int
173 data: bytes = b""
174
175 def pack(self) -> bytes:
176 """Pack frame to bytes."""
177 # Construct tag: [MSB][stream_id][flag]
178 # MSB is set for remote-initiated streams
179 msb = (self.stream_id & 1) == 0
180 tag = self.stream_id << 3 | self.flag
181
182 if msb:
183 tag |= 0x80
184
185 result = bytearray()
186 write_uvarint(result, tag)
187
188 # Only MESSAGE frames have data payload
189 if self.flag == MplexFlag.MESSAGE and self.data:
190 # Add length prefix for data
191 write_uvarint(result, len(self.data))
192 result.extend(self.data)
193
194 return bytes(result)
195
196 @classmethod
197 def unpack(cls, data: bytes) -> tuple['MplexFrame', int]:
198 """
199 Unpack frame from bytes.
200
201 Returns:
202 Tuple of (frame, bytes_consumed)
203 """
204 tag, offset = read_uvarint(data)
205
206 flag = MplexFlag(tag & 0x07)
207 msb = (tag & 0x80) != 0
208
209 # Stream ID: tag >> 3 (after removing the 3-bit flag)
210 stream_id = tag >> 3
211
212 frame_data = b""
213 data_offset = offset
214
215 # Only MESSAGE frames have data payload
216 if flag == MplexFlag.MESSAGE:
217 remaining = data[offset:]
218 if remaining:
219 try:
220 length, len_len = read_uvarint(remaining)
221 if len_len + length <= len(remaining):
222 frame_data = remaining[len_len:len_len + length]
223 data_offset += len_len + length
224 except (MplexProtocolError, ValueError):
225 # Incomplete data, return empty for now
226 pass
227
228 return cls(flag=flag, stream_id=stream_id, data=frame_data), data_offset
229
230 def __repr__(self) -> str:
231 return (f"MplexFrame(flag={self.flag.name}, "
232 f"stream_id={self.stream_id}, "
233 f"data_len={len(self.data)})")

333 async def read(self, n: int = -1) -> bytes:
334 """
335 Read data from stream.
336
337 Args:
338 n: Number of bytes to read, -1 for all available
339
340 Returns:
341 Read data
342
343 Raises:
344 MplexStreamClosed: Stream is closed
345 MplexStreamReset: Stream was reset
346 """
347 async with self._read_lock:
348 if self._reset:
349 raise MplexStreamReset("Stream was reset")
350
351 if self._remote_closed and not self._recv_buffer:
352 return b""
353
354 data = bytearray()
355
356 if n < 0:
357 # Read all available data
358 while self._recv_buffer:
359 chunk = self._recv_buffer.popleft()
360 data.extend(chunk)
361 self._recv_size -= len(chunk)
362
363 if not self._remote_closed:
364 await self._send_window_update()
365
366 if not data and not self._remote_closed:
367 await self._recv_event.wait()
368 self._recv_event.clear()
369 return await self.read(-1)
370 else:
371 # Read specific number of bytes
372 while len(data) < n:
373 if not self._recv_buffer:
374 if self._remote_closed:
375 break
376 await self._recv_event.wait()
377 self._recv_event.clear()
378 continue
379
380 chunk = self._recv_buffer[0]
381 needed = n - len(data)
382 if len(chunk) <= needed:
383 data.extend(self._recv_buffer.popleft())
384 self._recv_size -= len(chunk)
385 else:
386 data.extend(chunk[:needed])
387 self._recv_buffer[0] = chunk[needed:]
388 self._recv_size -= needed
389
390 if not self._remote_closed:
391 await self._send_window_update()
392
393 # Send window update if we consumed data
394 if data and not self._remote_closed:
395 await self._send_window_update()
396
397 return bytes(data)
398
399 async def write(self, data: bytes) -> int:
400 """
401 Write data to stream.
402
403 Args:
404 data: Data to write
405
406 Returns:
407 Number of bytes written
408
409 Raises:
410 MplexStreamClosed: Stream is closed
411 MplexStreamReset: Stream was reset
412 """
413 async with self._write_lock:
414 if self._reset:
415 raise MplexStreamReset("Stream was reset")
416
417 if self._local_closed:
418 raise MplexStreamClosed("Local side already closed")
419
420 if not data:
421 return 0
422
423 # Wait for stream to be established
424 if not self._established.is_set():
425 await self._established.wait()
426
427 # Split into chunks if needed
428 chunk_size = min(DEFAULT_BUFFER_SIZE, len(data))
429 written = 0
430
431 for i in range(0, len(data), chunk_size):
432 chunk = data[i:i + chunk_size]
433 frame = MplexFrame(
434 flag=MplexFlag.MESSAGE,
435 stream_id=self._id,
436 data=chunk
437 )
438 await self._session._send_frame(frame)
439 written += len(chunk)
440 self._bytes_sent += len(chunk)
441
442 return written
443
444 async def close(self) -> None:
445 """
446 Half-close stream (close write side).
447
448 Send CLOSE flag to indicate no more data will be sent.
449 """
450 async with self._write_lock:
451 if self._local_closed:
452 return
453
454 self._local_closed = True
455
456 frame = MplexFrame(
457 flag=MplexFlag.CLOSE,
458 stream_id=self._id,
459 )
460 await self._session._send_frame(frame)
461
462 if self._remote_closed:
463 self._state = StreamState.CLOSED
464 else:
465 self._state = StreamState.CLOSED_LOCAL
466
467 async def reset(self) -> None:
468 """
469 Immediately reset stream.
470
471 Send RESET flag to abort stream.
472 """
473 self._reset = True
474 self._state = StreamState.RESET
475
476 frame = MplexFrame(
477 flag=MplexFlag.RESET,
478 stream_id=self._id,
479 )
480 await self._session._send_frame(frame)
481
482 # Clear receive buffer
483 self._recv_buffer.clear()
484 self._recv_size = 0
485 self._recv_event.set()
486
487 async def _handle_data(self, data: bytes) -> None:
488 """Handle received data."""
489 if self._reset:
490 return
491
492 if self._recv_size + len(data) > MAX_BUFFER_SIZE:
493 # Buffer overflow, reset stream
494 await self.reset()
495 raise MplexWindowExceeded("Receive buffer exceeded")
496
497 self._recv_buffer.append(data)
498 self._recv_size += len(data)
499 self._bytes_received += len(data)
500 self._recv_event.set()
501
502 def _handle_close(self) -> None:
503 """Handle remote close."""
504 self._remote_closed = True
505 self._recv_event.set()
506
507 if self._local_closed:
508 self._state = StreamState.CLOSED
509 else:
510 self._state = StreamState.CLOSED_REMOTE
511
512 def _handle_reset(self) -> None:
513 """Handle remote reset."""
514 self._reset = True
515 self._state = StreamState.RESET
516 self._recv_event.set()
517
518 async def _send_window_update(self) -> None:
519 """Send window update (implicit in mplex via backpressure)."""
520 # Mplex doesn't have explicit window updates like yamux
521 # Backpressure is handled by not reading from underlying transport
522 pass
523

525 """Set stream state."""
526 self._state = state
527 if state == StreamState.OPEN:
528 self._established.set()

533class MplexSessionPool:
534 """
535 Pool of Mplex sessions for connection reuse.
536
537 Manages multiple sessions and provides connection pooling.
538 """
539
540 def __init__(
541 self,
542 max_sessions: int = DEFAULT_MAX_SESSIONS,
543 idle_timeout: float = SESSION_IDLE_TIMEOUT,
544 ):
545 self._max_sessions = max_sessions
546 self._idle_timeout = idle_timeout
547 self._sessions: Dict[str, MplexSession] = {}
548 self._session_last_used: Dict[str, float] = {}
549 self._lock = asyncio.Lock()
550 self._cleanup_task: Optional[asyncio.Task] = None
551
552 def start(self) -> None:
553 """Start the session pool cleanup task."""
554 if not self._cleanup_task:
555 self._cleanup_task = asyncio.create_task(self._cleanup_loop())
556
557 async def stop(self) -> None:
558 """Stop the session pool and close all sessions."""
559 if self._cleanup_task:
560 self._cleanup_task.cancel()
561 try:
562 await self._cleanup_task
563 except asyncio.CancelledError:
564 pass
565
566 async with self._lock:
567 for session in self._sessions.values():
568 try:
569 await session.close()
570 except Exception:
571 pass
572 self._sessions.clear()
573 self._session_last_used.clear()
574
575 async def get_session(
576 self,
577 host: str,
578 port: int,
579 ) -> 'MplexSession':
580 """
581 Get or create a session for the given address.
582
583 Args:
584 host: Target host
585 port: Target port
586
587 Returns:
588 MplexSession instance
589 """
590 key = f"{host}:{port}"
591
592 async with self._lock:
593 # Check for existing session
594 if key in self._sessions:
595 session = self._sessions[key]
596 if not session.is_closed:
597 self._session_last_used[key] = time.time()
598 return session
599 else:
600 # Remove closed session
601 del self._sessions[key]
602 del self._session_last_used[key]
603
604 # Create new session
605 session = await MplexSession.create_client(host, port)
606 self._sessions[key] = session
607 self._session_last_used[key] = time.time()
608
609 return session
610
611 async def return_session(self, session: 'MplexSession') -> None:
612 """Return a session to the pool."""
613 # Sessions stay in the pool for reuse
614 # Cleanup task will remove idle sessions
615 pass
616
617 async def _cleanup_loop(self) -> None:
618 """Periodically cleanup idle sessions."""
619 while True:
620 try:
621 await asyncio.sleep(60) # Check every minute
622
623 async with self._lock:
624 now = time.time()
625 to_remove = []
626
627 for key, last_used in self._session_last_used.items():
628 if now - last_used > self._idle_timeout:
629 to_remove.append(key)
630
631 for key in to_remove:
632 session = self._sessions.pop(key, None)
633 self._session_last_used.pop(key, None)
634 if session and not session.is_closed:
635 try:
636 await session.close()
637 except Exception:
638 pass
639
640 logger.debug(f"Cleaned up idle session: {key}")
641
642 except asyncio.CancelledError:
643 break
644 except Exception as e:
645 logger.error(f"Session pool cleanup error: {e}")

751 async def open_stream(
752 self,
753 priority: StreamPriority = StreamPriority.NORMAL,
754 ) -> MplexStream:
755 """
756 Open new stream.
757
758 Args:
759 priority: Stream priority level
760
761 Returns:
762 New MplexStream instance
763
764 Raises:
765 MplexClosedError: Session is closed
766 """
767 if self._closed:
768 raise MplexClosedError("Session is closed")
769
770 async with self._stream_id_lock:
771 stream_id = self._next_stream_id
772 self._next_stream_id += 2 # Odd/even based on client/server
773
774 if stream_id > MAX_STREAM_ID:
775 raise MplexProtocolError("Stream ID exhausted")
776
777 async with self._streams_lock:
778 if stream_id in self._streams:
779 raise MplexProtocolError(f"Stream ID conflict: {stream_id}")
780
781 stream = MplexStream(
782 stream_id=stream_id,
783 session=self,
784 is_initiator=True,
785 priority=priority,
786 )
787 self._streams[stream_id] = stream
788 self._write_queues[priority].append(stream_id)
789
790 # Send NEW_STREAM frame
791 frame = MplexFrame(
792 flag=MplexFlag.NEW_STREAM,
793 stream_id=stream_id,
794 )
795 await self._send_frame(frame)
796
797 stream._set_state(StreamState.OPEN)
798 self._streams_opened += 1
799
800 logger.debug(f"Opened new stream: {stream_id} (priority={priority.name})")
801
802 return stream

883 async def _read_loop(self) -> None:
884 """Read loop for incoming frames."""
885 buffer = bytearray()
886
887 while not self._closed:
888 try:
889 # Read at least one byte for uvarint
890 data = await self._reader.read(1)
891 if not data:
892 logger.debug("Connection closed by peer")
893 await self.close()
894 return
895
896 buffer.extend(data)
897
898 # Try to read tag
899 try:
900 tag, tag_len = read_uvarint(bytes(buffer))
901 flag = MplexFlag(tag & 0x07)
902 # Stream ID: tag >> 3 (after removing the 3-bit flag)
903 stream_id = tag >> 3
904
905 frame_data = b""
906 data_offset = tag_len
907
908 # Only MESSAGE frames have data payload
909 if flag == MplexFlag.MESSAGE:
910 remaining = buffer[tag_len:]
911 if remaining:
912 try:
913 length, len_len = read_uvarint(remaining)
914 total_len = tag_len + len_len + length
915
916 # Read more data if needed
917 while len(buffer) < total_len:
918 data = await self._reader.read(total_len - len(buffer))
919 if not data:
920 await self.close()
921 return
922 buffer.extend(data)
923 remaining = buffer[tag_len:]
924
925 frame_data = remaining[len_len:len_len + length]
926 data_offset += len_len + length
927 except (MplexProtocolError, ValueError):
928 # Not enough data yet for MESSAGE frame
929 continue
930
931 frame = MplexFrame(flag=flag, stream_id=stream_id, data=frame_data)
932 self._frames_received += 1
933
934 buffer.clear()
935
936 await self._handle_frame(frame)
937
938 except MplexProtocolError:
939 # Need more data
940 continue
941
942 except Exception as e:
943 if not self._closed:
944 logger.error(f"Read error: {e}")
945 await self.close()
946 return

948 async def _handle_frame(self, frame: MplexFrame) -> None:
949 """Handle incoming frame."""
950 logger.debug(f"Received frame: {frame}")
951
952 async with self._streams_lock:
953 stream = self._streams.get(frame.stream_id)
954
955 # New stream
956 if frame.flag == MplexFlag.NEW_STREAM:
957 if stream is None:
958 stream = MplexStream(
959 stream_id=frame.stream_id,
960 session=self,
961 is_initiator=False,
962 )
963 self._streams[frame.stream_id] = stream
964 stream._set_state(StreamState.OPEN)
965
966 try:
967 self._accept_queue.put_nowait(stream)
968 except asyncio.QueueFull:
969 # Reject stream
970 await stream.reset()
971 return
972
973 logger.debug(f"Accepted new stream: {frame.stream_id}")
974 return
975
976 if stream is None:
977 logger.warning(f"Unknown stream ID: {frame.stream_id}")
978 return
979
980 # Handle frame based on flag
981 if frame.flag == MplexFlag.MESSAGE:
982 if frame.data:
983 await stream._handle_data(frame.data)
984
985 elif frame.flag == MplexFlag.CLOSE:
986 stream._handle_close()
987
988 elif frame.flag == MplexFlag.RESET:
989 stream._handle_reset()
990
991 # Cleanup closed streams
992 async with self._streams_lock:
993 if stream.is_closed and stream._recv_size == 0:
994 self._streams.pop(frame.stream_id, None)
995 self._streams_closed += 1
