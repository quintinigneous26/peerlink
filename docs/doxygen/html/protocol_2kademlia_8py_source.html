87 async def handle_stream(
88 self,
89 reader: asyncio.StreamReader,
90 writer: asyncio.StreamWriter
91 ) -> None:
92 """
93 Handle incoming DHT stream.
94
95 Args:
96 reader: Stream reader
97 writer: Stream writer
98 """
99 peer_addr = writer.get_extra_info("peername")
100
101 try:
102 # Read message length (varint)
103 length = 0
104 shift = 0
105 while True:
106 byte_buf = await reader.readexactly(1)
107 byte = byte_buf[0]
108 length |= (byte & 0x7F) << shift
109 if not (byte & 0x80):
110 break
111 shift += 7
112
113 # Read message
114 if length > 10 * 1024 * 1024: # 10MB limit
115 logger.warning(f"Message too large: {length} bytes from {peer_addr}")
116 return
117
118 message_data = await reader.readexactly(length)
119
120 # Get peer ID from connection (would normally come from Identify)
121 # For now, use a placeholder
122 peer_id = b"\x00" * 32 # Should get from Identify protocol
123
124 # Handle message
125 response = await self.dht.handle_message(peer_id, message_data)
126
127 # Send response
128 if response:
129 # Encode response length
130 response_length = len(response)
131 length_bytes = bytearray()
132 while response_length > 0x7F:
133 length_bytes.append((response_length & 0x7F) | 0x80)
134 response_length >>= 7
135 length_bytes.append(response_length)
136
137 writer.write(bytes(length_bytes))
138 writer.write(response)
139 await writer.drain()
140
141 except asyncio.IncompleteReadError:
142 logger.debug(f"Incomplete read from {peer_addr}")
143 except Exception as e:
144 logger.error(f"Error handling DHT stream: {e}")
145 finally:
146 try:
147 writer.close()
148 await writer.wait_closed()
149 except Exception:
150 pass

294) -> Optional[PeerEntry]:
295 """
296 Convenience function to find a peer without persistent DHT instance.

298 Args:
299 local_peer_id: Local peer ID
300 target_peer_id: Target peer ID to find
301 bootstrap_peers: List of (peer_id, addresses) tuples
