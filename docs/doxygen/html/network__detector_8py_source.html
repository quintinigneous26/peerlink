86 async def _detect_nat_level(self) -> int:
87 """
88 检测 NAT 层级
89
90 方法：
91 1. 比较本地 IP 和 STUN 返回的公网 IP
92 2. 如果差异大（不同网段），说明有多层 NAT
93 3. CGNAT IP 段检测
94 """
95 observations = []
96
97 # 收集多次观测
98 for i in range(3):
99 response = await self.stun_client.binding_request(
100 self.stun_client.servers[0].split(":")[0],
101 3478
102 )
103
104 if response.success:
105 # 获取本地 IP
106 local_ip = self._get_local_ip()
107
108 observations.append(NetworkObservation(
109 timestamp=time.time(),
110 public_ip=response.mapped_ip,
111 public_port=response.mapped_port,
112 local_ip=local_ip,
113 local_port=0,
114 latency_ms=0,
115 ))
116
117 await asyncio.sleep(0.3)
118
119 if not observations:
120 return 1
121
122 first_obs = observations[0]
123
124 # 检查是否在 CGNAT 后（100.64.0.0/10）
125 if first_obs.public_ip.startswith("100."):
126 logger.debug("检测到 CGNAT")
127 return 2
128
129 # 比较本地 IP 和公网 IP 的网段
130 local_parts = first_obs.local_ip.split(".")
131 public_parts = first_obs.public_ip.split(".")
132
133 if len(local_parts) == 4 and len(public_parts) == 4:
134 # 检查前两个网段是否相同
135 if local_parts[0] == public_parts[0] and local_parts[1] == public_parts[1]:
136 # 同网段，可能是 1 层 NAT
137 return 1
138 else:
139 # 不同网段，可能有多层 NAT
140 return 2
141
142 # 默认返回 1 层
143 return 1

145 def _get_local_ip(self) -> str:
146 """获取本地 IP"""
147 try:
148 sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
149 sock.connect(("8.8.8.8", 80))
150 ip = sock.getsockname()[0]
151 sock.close()
152 return ip
153 except OSError:
154 return "127.0.0.1"

156 async def _detect_vpn(self) -> bool:
157 """
158 检测 VPN 环境
159
160 方法：
161 1. 检查常见 VPN 接口（tun0, ppp0 等）
162 2. 检查 MTU 大小（VPN 通常较小）
163 3. 检查公网 IP 是否为 VPN 出口
164 """
165 # 检查 VPN 接口
166 vpn_interfaces = ["tun", "ppp", "utun", "ipsec"]
167
168 try:
169 import netifaces
170 interfaces = netifaces.interfaces()
171 for iface in interfaces:
172 for vpn_prefix in vpn_interfaces:
173 if iface.startswith(vpn_prefix):
174 logger.debug(f"检测到 VPN 接口: {iface}")
175 return True
176 except ImportError:
177 pass
178
179 # 检查公网 IP 是否属于 VPN 服务商
180 # 常见 VPN IP 段
181 vpn_ip_ranges = [
182 "10.", # 私有网络 VPN
183 "172.16.", # 私有网络 VPN
184 ]
185
186 response = await self.stun_client.binding_request(
187 self.stun_client.servers[0].split(":")[0],
188 3478
189 )
190
191 if response.success:
192 for vpn_range in vpn_ip_ranges:
193 if response.mapped_ip.startswith(vpn_range):
194 # 可能是 VPN，但也可能是 CGNAT
195 # 需要进一步确认
196 pass
197
198 return False

226 async def _detect_enterprise_network(self) -> bool:
227 """
228 检测企业网络
229
230 方法：
231 1. 检查内网 IP 段（10.x, 172.16-31.x）
232 2. 检查是否有企业防火墙特征
233 3. 检查 DNS 是否为企业 DNS
234 """
235 local_ip = self._get_local_ip()
236
237 # 企业网络通常使用 10.x.x.x 或 172.16-31.x.x
238 if local_ip.startswith("10."):
239 # 10.x 可能是企业网，也可能是 CGNAT
240 # 检查是否有企业防火墙特征
241 pass
242
243 # 172.16-31.x.x 私有网段
244 if local_ip.startswith("172."):
245 parts = local_ip.split(".")
246 if len(parts) >= 2:
247 second_octet = int(parts[1])
248 if 16 <= second_octet <= 31:
249 # 企业私有网段
250 logger.debug("检测到企业私有网段")
251 return True
252
253 return False

255 async def _detect_mobile_network(self) -> bool:
256 """
257 检测移动网络（4G/5G）
258
259 方法：
260 1. 检查 MTU 大小（移动网络通常较小）
261 2. 检查网络延迟和抖动
262 3. 检查 IP 变化频率
263 """
264 # 测量延迟和抖动
265 latencies = []
266
267 for i in range(5):
268 start = time.time()
269 response = await self.stun_client.binding_request(
270 self.stun_client.servers[0].split(":")[0],
271 3478
272 )
273 latency = (time.time() - start) * 1000
274
275 if response.success:
276 latencies.append(latency)
277
278 await asyncio.sleep(0.2)
279
280 if len(latencies) >= 3:
281 # 计算抖动
282 avg = sum(latencies) / len(latencies)
283 jitter = sum(abs(latencies[i] - latencies[i-1])
284 for i in range(1, len(latencies))) / (len(latencies) - 1)
285
286 # 移动网络特征：高延迟 + 高抖动
287 if avg > 80 and jitter > 30:
288 logger.debug(f"检测到移动网络特征: 延迟={avg:.0f}ms, 抖动={jitter:.0f}ms")
289 return True
290
291 return False

320 async def _detect_link_quality(self) -> tuple:
321 """
322 检测链路质量
323
324 Returns:
325 (丢包率, 平均延迟)
326 """
327 success_count = 0
328 total_count = 10
329 latencies = []
330
331 for i in range(total_count):
332 start = time.time()
333 response = await self.stun_client.binding_request(
334 self.stun_client.servers[0].split(":")[0],
335 3478
336 )
337
338 if response.success:
339 success_count += 1
340 latencies.append((time.time() - start) * 1000)
341
342 await asyncio.sleep(0.2)
343
344 loss_rate = 1 - (success_count / total_count)
345 avg_latency = sum(latencies) / len(latencies) if latencies else 0
346
347 logger.debug(f"链路质量: 丢包率={loss_rate:.1%}, 延迟={avg_latency:.0f}ms")
348
349 return loss_rate, avg_latency
