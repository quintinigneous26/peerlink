51@asynccontextmanager
52async def lifespan(app: FastAPI):
53 """Manage application lifespan."""
54 # Startup
55 logger.info("Starting Relay Server...")
56
57 # Start components
58 relay_handler = get_relay_handler()
59 await relay_handler.start(config.host, (config.min_port, config.max_port))
60
61 # Start cleanup task
62 async def cleanup_task():
63 """Periodically cleanup expired sessions."""
64 while True:
65 await asyncio.sleep(60)
66 await relay_handler.cleanup_expired()
67
68 cleanup = asyncio.create_task(cleanup_task())
69
70 logger.info(f"Relay Server started on {config.host}:{config.port}")
71
72 yield
73
74 # Shutdown
75 logger.info("Shutting down Relay Server...")
76 cleanup.cancel()
77 await relay_handler.stop()
78 logger.info("Relay Server shut down")

111@app.post("/api/v1/relay/allocate")
112async def allocate_relay(request: AllocationRequest):
113 """
114 Allocate a relay session.
115
116 Creates a new relay allocation for the device.
117 """
118 handler = get_relay_handler()
119
120 # For this API, we use device_id to identify client
121 # In production, this would use the actual client IP:port from the connection
122 import uuid
123 client_addr = (request.device_id, 0) # Simplified
124
125 # Get available port
126 import socket
127 sock = socket.socket()
128 sock.bind((config.host, 0))
129 relay_port = sock.getsockname()[1]
130 sock.close()
131
132 relay_addr = (config.host, relay_port)
133
134 session = await handler.create_session(
135 client_addr=client_addr,
136 relay_addr=relay_addr,
137 lifetime=request.lifetime,
138 transport=TransportProtocol(request.transport),
139 )
140
141 if not session:
142 raise HTTPException(
143 status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
144 detail="Failed to allocate relay session",
145 )
146
147 response = APIResponse(
148 success=True,
149 data={
150 "allocation_id": session.session_id,
151 "relay_addr": f"{relay_addr[0]}:{relay_addr[1]}",
152 "relay_host": relay_addr[0],
153 "relay_port": relay_addr[1],
154 "lifetime": session.lifetime,
155 "transport": session.transport.value,
156 "expires_at": session.created_at + session.lifetime,
157 },
158 )
159
160 return response.to_dict()
