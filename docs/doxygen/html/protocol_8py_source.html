43@dataclass
44class P2PMessage:
45 """Base P2P message."""
46
47 msg_type: MessageTypes
48 sender_did: str
49 receiver_did: str
50 message_id: str = field(default_factory=lambda: str(uuid.uuid4()))
51 channel_id: Optional[int] = None

56 def encode(self) -> bytes:
57 """Encode message to bytes for transmission."""
58 # Create message dict
59 msg_dict = {
60 "msg_type": self.msg_type.value,
61 "sender_did": self.sender_did,
62 "receiver_did": self.receiver_did,
63 "message_id": self.message_id,
64 "channel_id": self.channel_id,
65 "timestamp": self.timestamp,
66 "metadata": self.metadata,
67 }
68
69 # Convert to JSON
70 json_bytes = json.dumps(msg_dict).encode("utf-8")
71
72 # Format: [length(4)][json_length(4)][json_bytes][payload]
73 header = struct.pack(">II", len(json_bytes) + len(self.payload), len(json_bytes))
74
75 return header + json_bytes + self.payload

77 @classmethod
78 def decode(cls, data: bytes) -> Optional["P2PMessage"]:
79 """Decode message from bytes."""
80 try:
81 if len(data) < 8:
82 return None
83
84 # Parse header
85 total_length, json_length = struct.unpack(">II", data[:8])
86
87 if len(data) < total_length + 8:
88 return None
89
90 # Parse JSON
91 json_bytes = data[8 : 8 + json_length]
92 msg_dict = json.loads(json_bytes.decode("utf-8"))
93
94 # Extract payload
95 payload = data[8 + json_length : 8 + total_length]
96
97 # Create message
98 return cls(
99 msg_type=MessageTypes(msg_dict["msg_type"]),
100 sender_did=msg_dict["sender_did"],
101 receiver_did=msg_dict["receiver_did"],
102 message_id=msg_dict.get("message_id", str(uuid.uuid4())),
103 channel_id=msg_dict.get("channel_id"),
104 payload=payload,
105 timestamp=msg_dict.get("timestamp", 0.0),
106 metadata=msg_dict.get("metadata", {}),
107 )
108
109 except Exception:
110 return None
