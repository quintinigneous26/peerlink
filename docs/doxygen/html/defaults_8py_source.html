100 def deep_merge(base: dict, update: dict) -> dict:
101 """递归合并字典"""
102 result = base.copy()
103 for key, value in update.items():
104 if key in result and isinstance(result[key], dict) and isinstance(value, dict):
105 result[key] = deep_merge(result[key], value)
106 else:
107 result[key] = value
108 return result
