111def write_uvarint(data: bytearray, value: int) -> None:
112 """
113 Write a variable-length integer to bytearray.
114
115 Uses unsigned LEB128 encoding.
116
117 Args:
118 data: Bytearray to write to
119 value: Integer value to encode
120 """
121 while value >= 0x80:
122 data.append((value & 0x7F) | 0x80)
123 value >>= 7
124 data.append(value)
125
126
127def read_uvarint(data: bytes, offset: int = 0) -> tuple[int, int]:
128 """
129 Read a variable-length integer from bytes.
130
131 Args:
132 data: Bytes to read from
133 offset: Starting offset
134
135 Returns:
136 Tuple of (value, bytes_consumed)
137
138 Raises:
139 MplexProtocolError: If encoding is invalid
140 """
141 value = 0
142 shift = 0
143 idx = offset
144
145 while idx < len(data):
146 byte = data[idx]
147 idx += 1
148 value |= (byte & 0x7F) << shift
149
150 if (byte & 0x80) == 0:
151 return value, idx - offset
152
153 shift += 7
154 if shift >= 64:
155 raise MplexProtocolError("Invalid uvarint encoding")
156
157 raise MplexProtocolError("Incomplete uvarint")

162@dataclass
163class MplexFrame:
164 """Mplex frame structure."""
165 flag: MplexFlag
166 stream_id: int
167 data: bytes = b""
168
169 def pack(self) -> bytes:
170 """Pack frame to bytes."""
171 # Construct tag: [MSB][stream_id][flag]
172 # MSB is set for remote-initiated streams
173 msb = (self.stream_id & 1) == 0
174 tag = self.stream_id << 3 | self.flag
175
176 if msb:
177 tag |= 0x80
178
179 result = bytearray()
180 write_uvarint(result, tag)
181
182 if self.data:
183 # Add length prefix for data
184 write_uvarint(result, len(self.data))
185 result.extend(self.data)
186
187 return bytes(result)
188
189 @classmethod
190 def unpack(cls, data: bytes) -> tuple['MplexFrame', int]:
191 """
192 Unpack frame from bytes.
193
194 Returns:
195 Tuple of (frame, bytes_consumed)
196 """
197 tag, offset = read_uvarint(data)
198
199 flag = MplexFlag(tag & 0x07)
200 msb = (tag & 0x80) != 0
201
202 # Stream ID is encoded in the bits between flag and MSB
203 # Format: [MSB][stream_id << 3][flag]
204 # According to mplex spec: stream_id = (tag >> 3)
205 stream_id = tag >> 3
206
207 frame_data = b""
208 remaining = data[offset:]
209
210 # Check if there's a data length
211 if remaining:
212 try:
213 length, data_offset = read_uvarint(remaining)
214 if data_offset + length <= len(remaining):
215 frame_data = remaining[data_offset:data_offset + length]
216 offset += data_offset + length
217 except (MplexProtocolError, ValueError):
218 pass
219
220 return cls(flag=flag, stream_id=stream_id, data=frame_data), offset
221
222 def __repr__(self) -> str:
223 return (f"MplexFrame(flag={self.flag.name}, "
224 f"stream_id={self.stream_id}, "
225 f"data_len={len(self.data)})")

290 async def read(self, n: int = -1) -> bytes:
291 """
292 Read data from stream.
293
294 Args:
295 n: Number of bytes to read, -1 for all available
296
297 Returns:
298 Read data
299
300 Raises:
301 MplexStreamClosed: Stream is closed
302 MplexStreamReset: Stream was reset
303 """
304 async with self._read_lock:
305 if self._reset:
306 raise MplexStreamReset("Stream was reset")
307
308 if self._remote_closed and not self._recv_buffer:
309 return b""
310
311 data = bytearray()
312
313 if n < 0:
314 # Read all available data
315 while self._recv_buffer:
316 chunk = self._recv_buffer.popleft()
317 data.extend(chunk)
318 self._recv_size -= len(chunk)
319
320 if not self._remote_closed:
321 self._send_window_update()
322
323 if not data and not self._remote_closed:
324 await self._recv_event.wait()
325 self._recv_event.clear()
326 return await self.read(-1)
327 else:
328 # Read specific number of bytes
329 while len(data) < n:
330 if not self._recv_buffer:
331 if self._remote_closed:
332 break
333 await self._recv_event.wait()
334 self._recv_event.clear()
335 continue
336
337 chunk = self._recv_buffer[0]
338 needed = n - len(data)
339 if len(chunk) <= needed:
340 data.extend(self._recv_buffer.popleft())
341 self._recv_size -= len(chunk)
342 else:
343 data.extend(chunk[:needed])
344 self._recv_buffer[0] = chunk[needed:]
345 self._recv_size -= needed
346
347 if not self._remote_closed:
348 self._send_window_update()
349
350 # Send window update if we consumed data
351 if data and not self._remote_closed:
352 await self._send_window_update()
353
354 return bytes(data)

356 async def write(self, data: bytes) -> int:
357 """
358 Write data to stream.
359
360 Args:
361 data: Data to write
362
363 Returns:
364 Number of bytes written
365
366 Raises:
367 MplexStreamClosed: Stream is closed
368 MplexStreamReset: Stream was reset
369 """
370 async with self._write_lock:
371 if self._reset:
372 raise MplexStreamReset("Stream was reset")
373
374 if self._local_closed:
375 raise MplexStreamClosed("Local side already closed")
376
377 if not data:
378 return 0
379
380 # Wait for stream to be established
381 if not self._established.is_set():
382 await self._established.wait()
383
384 # Split into chunks if needed
385 chunk_size = min(DEFAULT_BUFFER_SIZE, len(data))
386 written = 0
387
388 for i in range(0, len(data), chunk_size):
389 chunk = data[i:i + chunk_size]
390 frame = MplexFrame(
391 flag=MplexFlag.MESSAGE,
392 stream_id=self._id,
393 data=chunk
394 )
395 await self._session._send_frame(frame)
396 written += len(chunk)
397
398 return written

561 async def open_stream(self) -> MplexStream:
562 """
563 Open new stream.
564
565 Returns:
566 New MplexStream instance
567
568 Raises:
569 MplexClosedError: Session is closed
570 """
571 if self._closed:
572 raise MplexClosedError("Session is closed")
573
574 async with self._stream_id_lock:
575 stream_id = self._next_stream_id
576 self._next_stream_id += 2 # Odd/even based on client/server
577
578 if stream_id > MAX_STREAM_ID:
579 raise MplexProtocolError("Stream ID exhausted")
580
581 async with self._streams_lock:
582 if stream_id in self._streams:
583 raise MplexProtocolError(f"Stream ID conflict: {stream_id}")
584
585 stream = MplexStream(
586 stream_id=stream_id,
587 session=self,
588 is_initiator=True,
589 )
590 self._streams[stream_id] = stream
591
592 # Send NEW_STREAM frame
593 frame = MplexFrame(
594 flag=MplexFlag.NEW_STREAM,
595 stream_id=stream_id,
596 )
597 await self._send_frame(frame)
598
599 stream._set_state(StreamState.OPEN)

601 logger.debug(f"Opened new stream: {stream_id}")

683 async def _read_loop(self) -> None:
684 """Read loop for incoming frames."""
685 buffer = bytearray()
686
687 while not self._closed:
688 try:
689 # Read at least one byte for uvarint
690 data = await self._reader.read(1)
691 if not data:
692 logger.debug("Connection closed by peer")
693 await self.close()
694 return
695
696 buffer.extend(data)
697
698 # Try to read tag
699 try:
700 tag, tag_len = read_uvarint(bytes(buffer))
701 flag = MplexFlag(tag & 0x07)
702 # Stream ID: tag >> 3 (after removing the 3-bit flag)
703 stream_id = tag >> 3
704
705 # Check if data follows
706 remaining = buffer[tag_len:]
707 if remaining:
708 try:
709 length, len_len = read_uvarint(remaining)
710 total_len = tag_len + len_len + length
711
712 # Read more data if needed
713 while len(buffer) < total_len:
714 data = await self._reader.read(total_len - len(buffer))
715 if not data:
716 await self.close()
717 return
718 buffer.extend(data)
719
720 frame_data = remaining[len_len:len_len + length]
721 frame = MplexFrame(flag=flag, stream_id=stream_id, data=frame_data)
722 except (MplexProtocolError, ValueError):
723 # Not enough data yet
724 continue
725 else:
726 frame = MplexFrame(flag=flag, stream_id=stream_id)
727
728 buffer.clear()
729
730 await self._handle_frame(frame)
731
732 except MplexProtocolError:
733 # Need more data
734 continue
735
736 except Exception as e:
737 if not self._closed:
738 logger.error(f"Read error: {e}")
739 await self.close()
740 return

742 async def _handle_frame(self, frame: MplexFrame) -> None:
743 """Handle incoming frame."""
744 logger.debug(f"Received frame: {frame}")
745
746 async with self._streams_lock:
747 stream = self._streams.get(frame.stream_id)
748
749 # New stream
750 if frame.flag == MplexFlag.NEW_STREAM:
751 if stream is None:
752 stream = MplexStream(
753 stream_id=frame.stream_id,
754 session=self,
755 is_initiator=False,
756 )
757 self._streams[frame.stream_id] = stream
758 stream._set_state(StreamState.OPEN)
759
760 try:
761 self._accept_queue.put_nowait(stream)
762 except asyncio.QueueFull:
763 # Reject stream
764 await stream.reset()
765 return
766
767 logger.debug(f"Accepted new stream: {frame.stream_id}")
768 return
769
770 if stream is None:
771 logger.warning(f"Unknown stream ID: {frame.stream_id}")
772 return
773
774 # Handle frame based on flag
775 if frame.flag == MplexFlag.MESSAGE:
776 if frame.data:
777 await stream._handle_data(frame.data)
778
779 elif frame.flag == MplexFlag.CLOSE:
780 stream._handle_close()
781
782 elif frame.flag == MplexFlag.RESET:
783 stream._handle_reset()
784
785 # Cleanup closed streams
786 async with self._streams_lock:
787 if stream.is_closed and stream._recv_size == 0:
788 self._streams.pop(frame.stream_id, None)
