66class QueryCache:
67 """
68 LRU cache for DHT query results.
69
70 Reduces redundant network queries for frequently accessed data.
71 """
72
73 def __init__(self, max_size: int = DEFAULT_CACHE_SIZE):
74 self._max_size = max_size
75 self._cache: Dict[bytes, CachedResult] = {}
76 self._access_order: List[bytes] = []
77 self._lock = asyncio.Lock()
78 self._hits = 0
79 self._misses = 0
80
81 async def get(self, key: bytes) -> Optional[QueryResult]:
82 """
83 Get cached result.
84
85 Args:
86 key: Query key
87
88 Returns:
89 Cached result if available and not expired
90 """
91 async with self._lock:
92 cached = self._cache.get(key)
93
94 if cached is None:
95 self._misses += 1
96 return None
97
98 if cached.is_expired:
99 # Remove expired entry
100 del self._cache[key]
101 self._access_order.remove(key)
102 self._misses += 1
103 return None
104
105 # Update access order (move to end)
106 self._access_order.remove(key)
107 self._access_order.append(key)
108
109 cached.hit_count += 1
110 self._hits += 1
111
112 return cached.result
113
114 async def put(
115 self,
116 key: bytes,
117 result: QueryResult,
118 ) -> None:
119 """
120 Cache a result.
121
122 Args:
123 key: Query key
124 result: Query result to cache
125 """
126 async with self._lock:
127 # Update existing entry
128 if key in self._cache:
129 self._access_order.remove(key)
130 else:
131 # Evict oldest if at capacity
132 if len(self._cache) >= self._max_size:
133 oldest = self._access_order.pop(0)
134 del self._cache[oldest]
135
136 cached = CachedResult(
137 key=key,
138 result=result,
139 timestamp=time.time(),
140 )
141
142 self._cache[key] = cached
143 self._access_order.append(key)
144
145 async def invalidate(self, key: bytes) -> None:
146 """Invalidate a cache entry."""
147 async with self._lock:
148 if key in self._cache:
149 del self._cache[key]
150 self._access_order.remove(key)
151
152 async def clear(self) -> None:
153 """Clear all cache entries."""
154 async with self._lock:
155 self._cache.clear()
156 self._access_order.clear()
157
158 async def cleanup_expired(self) -> int:
159 """Remove expired entries."""
160 async with self._lock:
161 to_remove = [
162 key for key, cached in self._cache.items()
163 if cached.is_expired
164 ]
165
166 for key in to_remove:
167 del self._cache[key]
168 self._access_order.remove(key)
169
170 return len(to_remove)
171
172 @property
173 def hit_rate(self) -> float:
174 """Get cache hit rate."""
175 total = self._hits + self._misses
176 return self._hits / total if total > 0 else 0.0
177
178 @property
179 def size(self) -> int:
180 """Get cache size."""
181 return len(self._cache)

194class PeerSelector:
195 """
196 Selects optimal peers for DHT queries.
197
198 Uses various strategies to optimize query performance.
199 """

201 def __init__(
202 self,
203 strategy: PeerSelectionStrategy = PeerSelectionStrategy.ADAPTIVE,
204 ):
205 self._strategy = strategy
206 self._peer_performance: Dict[bytes, Dict[str, float]] = {}

208 async def select_peers(
209 self,
210 target_id: bytes,
211 candidates: List[PeerEntry],
212 count: int = ALPHA,
213 ) -> List[PeerEntry]:
214 """
215 Select peers for query.
216
217 Args:
218 target_id: Target ID for distance calculation
219 candidates: Available peers
220 count: Number of peers to select
221
222 Returns:
223 Selected peers
224 """
225 if not candidates:
226 return []
227
228 if self._strategy == PeerSelectionStrategy.CLOSEST:
229 return self._select_closest(target_id, candidates, count)
230 elif self._strategy == PeerSelectionStrategy.RANDOM:
231 return self._select_random(candidates, count)
232 elif self._strategy == PeerSelectionStrategy.LATENCY_AWARE:
233 return self._select_latency_aware(target_id, candidates, count)
234 else:
235 return self._select_adaptive(target_id, candidates, count)

237 def _select_closest(
238 self,
239 target_id: bytes,
240 candidates: List[PeerEntry],
241 count: int,
242 ) -> List[PeerEntry]:
243 """Select closest peers by XOR distance."""
244 peers_with_dist = [
245 (calculate_distance(peer.peer_id, target_id), peer)
246 for peer in candidates
247 ]
248 peers_with_dist.sort(key=lambda x: x[0])
249 return [peer for _, peer in peers_with_dist[:count]]
250
251 def _select_random(
252 self,
253 candidates: List[PeerEntry],
254 count: int,
255 ) -> List[PeerEntry]:
256 """Random peer selection."""
257 import random
258 return random.sample(candidates, min(count, len(candidates)))
259
260 def _select_latency_aware(
261 self,
262 target_id: bytes,
263 candidates: List[PeerEntry],
264 count: int,
265 ) -> List[PeerEntry]:
266 """Select peers considering latency."""
267 scored_peers = []
268
269 for peer in candidates:
270 distance = calculate_distance(peer.peer_id, target_id)
271
272 # Get peer latency info
273 perf = self._peer_performance.get(peer.peer_id, {})
274 latency = perf.get("avg_latency_ms", 1000.0)
275
276 # Score: balance distance and latency
277 score = distance / max(latency, 1.0)
278 scored_peers.append((score, peer))
279
280 scored_peers.sort(key=lambda x: x[0])
281 return [peer for _, peer in scored_peers[:count]]
282
283 def _select_adaptive(
284 self,
285 target_id: bytes,
286 candidates: List[PeerEntry],
287 count: int,
288 ) -> List[PeerEntry]:
289 """Adaptive peer selection based on historical performance."""
290 # Combine multiple factors
291 scored_peers = []
292
293 for peer in candidates:
294 distance = calculate_distance(peer.peer_id, target_id)
295 perf = self._peer_performance.get(peer.peer_id, {})
296
297 # Factors: distance, success rate, latency
298 success_rate = perf.get("success_rate", 0.5)
299 latency = perf.get("avg_latency_ms", 1000.0)
300
301 # Lower is better
302 score = (distance * 0.6) + (1000 / max(success_rate, 0.01) * 0.3) + (latency * 0.1)
303
304 scored_peers.append((score, peer))
305
306 scored_peers.sort(key=lambda x: x[0])
307 return [peer for _, peer in scored_peers[:count]]
308
309 def record_success(self, peer_id: bytes, latency_ms: float) -> None:
310 """Record successful query."""
311 if peer_id not in self._peer_performance:
312 self._peer_performance[peer_id] = {
313 "queries": 0,
314 "successes": 0,
315 "total_latency_ms": 0.0,
316 }
317
318 perf = self._peer_performance[peer_id]
319 perf["queries"] += 1
320 perf["successes"] += 1
321 perf["total_latency_ms"] += latency_ms
322 perf["avg_latency_ms"] = perf["total_latency_ms"] / perf["successes"]
323 perf["success_rate"] = perf["successes"] / perf["queries"]
324
325 def record_failure(self, peer_id: bytes) -> None:
326 """Record failed query."""
327 if peer_id not in self._peer_performance:
328 self._peer_performance[peer_id] = {
329 "queries": 0,
330 "successes": 0,
331 "total_latency_ms": 0.0,
332 }
333
334 perf = self._peer_performance[peer_id]
335 perf["queries"] += 1
336 perf["success_rate"] = perf["successes"] / perf["queries"]

342 """
343 Optimized DHT query manager.

345 Implements:

364 async def find_peer(
365 self,
366 target_id: bytes,
367 routing_table,
368 query_func: Callable[[bytes, bytes], Awaitable[List[PeerEntry]]],
369 use_cache: bool = True,
370 ) -> QueryResult:
371 """
372 Find peer with caching and optimizations.
373
374 Args:
375 target_id: Peer ID to find
376 routing_table: Routing table
377 query_func: Query function
378 use_cache: Whether to use cache
379
380 Returns:
381 Query result
382 """
383 # Check cache first
384 if use_cache:
385 cached = await self._cache.get(target_id)
386 if cached:
387 logger.debug(f"Cache hit for peer {target_id.hex()[:8]}")
388 return cached
389
390 # Get initial peers with smart selection
391 initial_peers = routing_table.find_closest_peers(
392 target_id,
393 self._current_alpha * 2 # Get more for selection
394 )
395
396 if not initial_peers:
397 return QueryResult(
398 success=False,
399 error="No peers in routing table",
400 duration_ms=0.0
401 )
402
403 # Select best peers
404 selected_peers = await self._peer_selector.select_peers(
405 target_id,
406 initial_peers,
407 self._current_alpha,
408 )
409
410 # Execute optimized query
411 result = await self._execute_adaptive_query(
412 target_id,
413 selected_peers,
414 query_func,
415 )
416
417 # Cache successful results
418 if result.success and use_cache:
419 await self._cache.put(target_id, result)
420
421 # Update peer performance
422 for peer in result.peers:
423 self._peer_selector.record_success(peer.peer_id, 0.0)
424
425 # Adapt concurrency
426 self._query_count += 1
427 if self._query_count % ALPHA_ADAPTATION_INTERVAL == 0:
428 await self._adapt_alpha()
429
430 return result

432 async def _execute_adaptive_query(
433 self,
434 target_id: bytes,
435 initial_peers: List[PeerEntry],
436 query_func: Callable,
437 ) -> QueryResult:
438 """Execute query with adaptive parallelism."""
439 start_time = time.time()
440
441 context = QueryContext(
442 query_type=QueryType.FIND_PEER,
443 target_id=target_id,
444 )
445
446 # Priority queue of peers by distance
447 pending: List[Tuple[int, PeerEntry]] = []
448 for peer in initial_peers:
449 distance = calculate_distance(peer.peer_id, target_id)
450 heapq.heappush(pending, (distance, peer))
451
452 contacted: Set[bytes] = set()
453 closest_distance = float('inf')
454
455 while pending and not context.is_expired(QUERY_TIMEOUT):
456 # Select peers for this round
457 to_query = []
458
459 while len(to_query) < self._current_alpha and pending:
460 distance, peer = heapq.heappop(pending)
461
462 if peer.peer_id not in contacted:
463 to_query.append((distance, peer))
464
465 if not to_query:
466 break
467
468 # Query in parallel
469 tasks = []
470 for distance, peer in to_query:
471 contacted.add(peer.peer_id)
472 tasks.append(self._query_peer(peer, target_id, query_func))
473
474 results = await asyncio.gather(*tasks, return_exceptions=True)
475
476 # Process results
477 for i, result in enumerate(results):
478 if isinstance(result, Exception):
479 _, peer = to_query[i]
480 self._peer_selector.record_failure(peer.peer_id)
481 continue
482
483 for new_peer in result:
484 distance = calculate_distance(new_peer.peer_id, target_id)
485 heapq.heappush(pending, (distance, new_peer))
486 context.results.append(new_peer)
487
488 # Check if found target
489 if new_peer.peer_id == target_id:
490 context.state = QueryState.COMPLETED
491 break
492
493 if context.state == QueryState.COMPLETED:
494 break
495
496 # Update closest distance
497 if pending:
498 current_closest = pending[0][0]
499 if current_closest >= closest_distance:
500 # No improvement
501 break
502 closest_distance = current_closest
503
504 duration_ms = (time.time() - start_time) * 1000
505
506 return QueryResult(
507 success=context.state == QueryState.COMPLETED,
508 peers=context.results[:K],
509 duration_ms=duration_ms,
510 )

513 self,
514 peer: PeerEntry,
515 target_id: bytes,

518 """Query a single peer with timing."""

520
521 try:
522 result = await asyncio.wait_for(
523 query_func(peer.peer_id, target_id),

530 return result
531
532 except Exception as e:

534 self._peer_selector.record_failure(peer.peer_id)
535 return []
