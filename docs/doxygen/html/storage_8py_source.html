19class DeviceStorage:

22 # Keys
23 DEVICE_KEY_PREFIX = "device:"
24 DEVICE_INDEX_KEY = "device:index"
25 ONLINE_DEVICES_KEY = "devices:online"

68 async def register_device(
69 self,
70 device_id: str,
71 device_type: str,
72 public_key: str,
73 capabilities: list[str],
74 metadata: dict | None = None,
75 ) -> DeviceInfo:
76 """
77 Register a new device.
78
79 Args:
80 device_id: Device identifier
81 device_type: Type of device
82 public_key: Device public key
83 capabilities: List of device capabilities
84 metadata: Optional metadata
85
86 Returns:
87 DeviceInfo
88 """
89 pool = await self._get_pool()
90 now = int(datetime.now().timestamp())
91
92 device_info = DeviceInfo(
93 device_id=device_id,
94 device_type=device_type,
95 public_key=public_key,
96 capabilities=capabilities,
97 status=DeviceStatus.ONLINE,
98 created_at=now,
99 last_seen=now,
100 metadata=metadata or {},
101 )
102
103 # Store device data
104 device_key = self._device_key(device_id)
105 await pool.hset(device_key, mapping=device_info.to_dict())
106 await pool.expire(device_key, self.DEVICE_TTL)
107
108 # Add to type index
109 type_key = self._type_index_key(device_type)
110 await pool.sadd(type_key, device_id)
111
112 # Add to online set
113 await pool.sadd(self.ONLINE_DEVICES_KEY, device_id)
114 await pool.expire(self.ONLINE_DEVICES_KEY, self.ONLINE_TTL)
115
116 logger.info(f"Device registered: {device_id} ({device_type})")
117 return device_info

119 async def get_device(self, device_id: str) -> Optional[DeviceInfo]:
120 """
121 Get device information.

123 Args:
124 device_id: Device identifier
125

127 DeviceInfo or None if not found

129 pool = await self._get_pool()

138 async def update_heartbeat(self, device_id: str) -> bool:
139 """
140 Update device heartbeat.
141
142 Args:
143 device_id: Device identifier
144
145 Returns:
146 True if device exists and was updated
147 """
148 pool = await self._get_pool()
149 device_key = self._device_key(device_id)
150
151 # Check if device exists
152 exists = await pool.exists(device_key)
153 if not exists:
154 return False
155
156 now = int(datetime.now().timestamp())
157 await pool.hset(device_key, "last_seen", now)
158 await pool.hset(device_key, "status", DeviceStatus.ONLINE.value)
159
160 # Refresh online status
161 await pool.sadd(self.ONLINE_DEVICES_KEY, device_id)
162 await pool.expire(self.ONLINE_DEVICES_KEY, self.ONLINE_TTL)
163
164 logger.debug(f"Heartbeat updated for {device_id}")
165 return True

167 async def set_device_status(self, device_id: str, status: DeviceStatus) -> bool:
168 """
169 Set device status.
170
171 Args:
172 device_id: Device identifier
173 status: New status
174
175 Returns:
176 True if device exists
177 """
178 pool = await self._get_pool()
179 device_key = self._device_key(device_id)
180
181 exists = await pool.exists(device_key)
182 if not exists:
183 return False
184
185 await pool.hset(device_key, "status", status.value)
186
187 if status == DeviceStatus.OFFLINE:
188 await pool.srem(self.ONLINE_DEVICES_KEY, device_id)
189 else:
190 await pool.sadd(self.ONLINE_DEVICES_KEY, device_id)
191 await pool.expire(self.ONLINE_DEVICES_KEY, self.ONLINE_TTL)
192
193 return True

234 async def delete_device(self, device_id: str) -> bool:
235 """
236 Delete a device.
237
238 Args:
239 device_id: Device identifier
240
241 Returns:
242 True if device was deleted
243 """
244 pool = await self._get_pool()
245 device_key = self._device_key(device_id)
246
247 # Get device info before deletion
248 device = await self.get_device(device_id)
249 if not device:
250 return False
251
252 # Remove from indexes
253 type_key = self._type_index_key(device.device_type)
254 await pool.srem(type_key, device_id)
255 await pool.srem(self.ONLINE_DEVICES_KEY, device_id)
256
257 # Delete device data
258 await pool.delete(device_key)
259
260 logger.info(f"Device deleted: {device_id}")
261 return True

263 async def cleanup_stale_devices(self, timeout_seconds: int) -> int:
264 """
265 Clean up devices that haven't sent heartbeat recently.
266
267 Args:
268 timeout_seconds: Seconds since last heartbeat to consider stale
269
270 Returns:
271 Number of devices marked offline
272 """
273 pool = await self._get_pool()
274 now = int(datetime.now().timestamp())
275 cutoff = now - timeout_seconds
276 count = 0
277
278 # Get all online devices
279 online_devices = await self.get_online_devices()
280
281 for device_id in online_devices:
282 device = await self.get_device(device_id)
283 if device and device.last_seen < cutoff:
284 await self.set_device_status(device_id, DeviceStatus.OFFLINE)
285 count += 1
286 logger.info(f"Device marked offline: {device_id}")
287
288 return count
