18@dataclass
19class ConnectionMetrics:
20 """连接指标"""
21 total_attempts: int = 0
22 p2p_success: int = 0
23 relay_success: int = 0
24 failed: int = 0
25 total_latency_ms: float = 0.0
26
27 @property
28 def success_rate(self) -> float:
29 """成功率"""
30 if self.total_attempts == 0:
31 return 0.0
32 return (self.p2p_success + self.relay_success) / self.total_attempts
33
34 @property
35 def p2p_rate(self) -> float:
36 """P2P 成功率"""
37 if self.total_attempts == 0:
38 return 0.0
39 return self.p2p_success / self.total_attempts
40
41 @property
42 def avg_latency_ms(self) -> float:
43 """平均延迟"""
44 success_count = self.p2p_success + self.relay_success
45 if success_count == 0:
46 return 0.0
47 return self.total_latency_ms / success_count
