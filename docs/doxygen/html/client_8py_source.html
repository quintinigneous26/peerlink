309 async def connect(self, did: str) -> bool:
310 """
311 Connect to a remote device.
312
313 Attempts P2P connection first, falls back to relay if configured.
314
315 Args:
316 did: Device ID to connect to
317
318 Returns:
319 True if connection successful
320
321 Raises:
322 ConnectionError: If connection fails
323 TimeoutError: If connection times out
324 """
325 async with self._lock:
326 if self._state in [ConnectionState.CONNECTED_P2P, ConnectionState.CONNECTED_RELAY]:
327 logger.warning(f"Already connected to {self._current_peer.did if self._current_peer else 'unknown'}")
328 return True
329
330 self._state = ConnectionState.CONNECTING
331 logger.info(f"Connecting to device: {did}")
332
333 try:
334 # Query peer info from signaling server
335 peer_info = await self._query_peer_info(did)
336 self._current_peer = peer_info
337
338 # Attempt P2P connection
339 p2p_success = await self._try_p2p_connect(peer_info)
340
341 if p2p_success:
342 logger.info(f"P2P connection established to {did}")
343 self._state = ConnectionState.CONNECTED_P2P
344 self._active_transport = self._udp_transport
345 await self._start_receive_loop()
346 await self._start_keepalive()
347 await self._notify_connected()
348 return True
349
350 # Fallback to relay
351 if self.config.auto_relay:
352 logger.info(f"P2P failed, attempting relay connection to {did}")
353 relay_success = await self._try_relay_connect(peer_info)
354
355 if relay_success:
356 logger.info(f"Relay connection established to {did}")
357 self._state = ConnectionState.CONNECTED_RELAY
358 self._active_transport = self._relay_transport
359 await self._start_receive_loop()
360 await self._start_keepalive()
361 await self._notify_connected()
362 return True
363
364 # All connection attempts failed
365 self._state = ConnectionState.FAILED
366 raise ConnectionError(f"Failed to connect to {did}: All attempts failed")
367
368 except Exception as e:
369 self._state = ConnectionState.FAILED
370 logger.error(f"Connection to {did} failed: {e}")
371 await self._notify_error(e)
372 raise ConnectionError(f"Connection failed: {e}") from e

391 async def _try_p2p_connect(self, peer: PeerInfo) -> bool:
392 """Attempt P2P connection via UDP hole punching."""
393 if not self._udp_transport or not peer.public_ip or not peer.public_port:

395 return False
396
397 logger.info(f"Attempting P2P connection to {peer.public_ip}:{peer.public_port}")

400 # Perform hole punching handshake
401 success = await self._hole_punch_handshake(peer)
402
403 if success:
404 logger.info("Hole punching successful, P2P connection established")

414 async def _hole_punch_handshake(self, peer: PeerInfo) -> bool:
415 """
416 Perform UDP hole punching handshake.
417
418 This implements a simplified STUN-like exchange:
419 1. Send handshake to peer's public address
420 2. Expect handshake response
421 3. Verify connection with ping/pong
422 """
423 if not self._udp_transport:
424 return False
425
426 # Create handshake message
427 handshake = create_handshake(
428 sender_did=self.did,
429 receiver_did=peer.did,
430 public_addr=self._public_addr,
431 )
432
433 # Target address - try public first, fall back to local
434 targets = []
435 if peer.public_ip and peer.public_port:
436 targets.append((peer.public_ip, peer.public_port))
437 if peer.local_ip and peer.local_port:
438 targets.append((peer.local_ip, peer.local_port))
439
440 # Send handshake to all candidate addresses
441 for target in targets:
442 try:
443 logger.debug(f"Sending handshake to {target[0]}:{target[1]}")
444 await self._udp_transport.sendto(
445 handshake.encode(), target[0], target[1]
446 )
447 except Exception as e:
448 logger.warning(f"Failed to send handshake to {target}: {e}")
449
450 # Wait for response with timeout
451 deadline = time.time() + self.config.punch_timeout
452
453 while time.time() < deadline:
454 try:
455 # Wait for incoming message with short timeout
456 data, addr = await asyncio.wait_for(
457 self._udp_transport.recvfrom(), timeout=1.0
458 )
459
460 # Parse message
461 msg = parse_message(data)
462 if msg and msg.msg_type == MessageTypes.HANDSHAKE:
463 if msg.sender_did == peer.did:
464 logger.info(f"Received handshake from {peer.did} at {addr}")
465 # Send handshake acknowledgment
466 ack = create_handshake(
467 sender_did=self.did,
468 receiver_did=peer.did,
469 public_addr=self._public_addr,
470 is_ack=True,
471 )
472 await self._udp_transport.sendto(ack.encode(), addr[0], addr[1])
473
474 # Store peer's actual address
475 peer.public_ip = addr[0]
476 peer.public_port = addr[1]
477
478 return True
479
480 except asyncio.TimeoutError:
481 # Try sending handshake again
482 for target in targets:
483 try:
484 await self._udp_transport.sendto(
485 handshake.encode(), target[0], target[1]
486 )
487 except (ConnectionError, OSError) as e:
488 logger.debug(f"Failed to send retry handshake: {e}")
489 except Exception as e:
490 logger.debug(f"Error during handshake wait: {e}")
491
492 return False

516 async def _receive_loop(self) -> None:

518 while self._running and self._active_transport:

534 await self._send_keepalive_response()
