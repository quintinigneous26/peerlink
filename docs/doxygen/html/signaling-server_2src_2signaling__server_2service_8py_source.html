24@asynccontextmanager
25async def lifespan(app: FastAPI):
26 """Manage application lifespan."""
27 # Startup
28 logger.info("Starting Signaling Server...")
29
30 # Start heartbeat cleanup task
31 manager = get_connection_manager()
32
33 async def cleanup_task():
34 """Periodically cleanup stale connections."""
35 while True:
36 await asyncio.sleep(config.heartbeat_interval)
37 cleaned = await manager.cleanup_stale(config.connection_timeout)
38 if cleaned > 0:
39 logger.info(f"Cleaned up {cleaned} stale connections")
40
41 cleanup = asyncio.create_task(cleanup_task())
42
43 logger.info(f"Signaling Server started on ws://{config.host}:{config.ws_port}")
44
45 yield
46
47 # Shutdown
48 logger.info("Shutting down Signaling Server...")
49 cleanup.cancel()
50 logger.info("Signaling Server shut down")

140@app.websocket("/v1/signaling")
141async def websocket_endpoint(
142 websocket: WebSocket,
143 device_id: Optional[str] = Query(None),
144 token: Optional[str] = Query(None),
145):
146 """
147 WebSocket endpoint for device signaling.
148
149 Query parameters:
150 - device_id: Device identifier (required)
151 - token: Optional authentication token
152 """
153 manager = get_connection_manager()
154 handler = MessageHandler()
155
156 # Validate device_id
157 if not device_id:
158 await websocket.close(code=status.WS_1008_POLICY_VIOLATION)
159 return
160
161 # TODO: Verify token with DID service if provided
162
163 # Accept connection
164 await websocket.accept()
165
166 logger.info(f"WebSocket connection request from {device_id}")
167
168 try:
169 # Register device (temporary - will be replaced by proper register message)
170 device_info = await manager.connect(
171 device_id=device_id,
172 ws=websocket,
173 public_key="", # Will be filled in register message
174 capabilities=[],
175 )
176
177 # Send registration confirmation
178 await websocket.send_json({
179 "type": MessageType.REGISTERED.value,
180 "data": {
181 "device_id": device_id,
182 "server_time": int(asyncio.get_event_loop().time()),
183 },
184 })
185
186 # Message loop
187 while True:
188 try:
189 # Receive message with timeout
190 data = await asyncio.wait_for(
191 websocket.receive_json(),
192 timeout=config.ping_timeout,
193 )
194
195 # Parse message
196 message = Message.from_dict(data)
197
198 # Set source device ID
199 message.source_device_id = device_id
200
201 # Handle message
202 response = await handler.handle_message(device_id, message)
203
204 # Send response if any
205 if response:
206 await websocket.send_json(response)
207
208 except asyncio.TimeoutError:
209 # Send ping to keep connection alive
210 await websocket.send_json({
211 "type": MessageType.PING.value,
212 "data": {},
213 "timestamp": int(asyncio.get_event_loop().time()),
214 })
215
216 except WebSocketDisconnect:
217 logger.info(f"WebSocket disconnected: {device_id}")
218 break
219
220 except Exception as e:
221 logger.error(f"Error processing message from {device_id}: {e}")
222
223 # Send error response
224 error = ErrorResponse(
225 code=ErrorCode.INTERNAL_ERROR,
226 message=str(e),
227 )
228 try:
229 await websocket.send_json(error.to_dict())
230 except Exception:
231 break
232
233 except WebSocketDisconnect:
234 logger.info(f"WebSocket disconnected during handshake: {device_id}")
235
236 except Exception as e:
237 logger.error(f"WebSocket error for {device_id}: {e}")
238
239 finally:
240 # Cleanup
241 await manager.disconnect(device_id)
242 logger.info(f"Connection closed for {device_id}")
