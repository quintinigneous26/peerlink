93 def check_rate_limit(self, request: Request) -> tuple[bool, dict]:
94 """
95 Check if request is allowed under rate limits.
96
97 Args:
98 request: FastAPI request object
99
100 Returns:
101 Tuple of (allowed: bool, info: dict with limit details)
102 """
103 client_id = self._get_client_id(request)
104 now = time.time()
105 state = self._clients[client_id]
106
107 # Check if client is temporarily blocked
108 if state.blocked_until > now:
109 return False, {
110 "error": "rate_limit_exceeded",
111 "retry_after": int(state.blocked_until - now),
112 "client_id": client_id,
113 }
114
115 # Cleanup old requests
116 state.minute_requests = self._cleanup_old_requests(state.minute_requests, 60)
117 state.hour_requests = self._cleanup_old_requests(state.hour_requests, 3600)
118 state.burst_requests = self._cleanup_old_requests(state.burst_requests, 1)
119
120 # Check burst limit (1 second)
121 if len(state.burst_requests) >= self.config.burst_size:
122 state.blocked_until = now + 1 # Block for 1 second
123 return False, {
124 "error": "burst_limit_exceeded",
125 "retry_after": 1,
126 "limit": self.config.burst_size,
127 "window": "1 second",
128 }
129
130 # Check minute limit
131 if len(state.minute_requests) >= self.config.requests_per_minute:
132 state.blocked_until = now + 60 # Block for 1 minute
133 return False, {
134 "error": "minute_limit_exceeded",
135 "retry_after": 60,
136 "limit": self.config.requests_per_minute,
137 "window": "1 minute",
138 }
139
140 # Check hour limit
141 if len(state.hour_requests) >= self.config.requests_per_hour:
142 state.blocked_until = now + 3600 # Block for 1 hour
143 return False, {
144 "error": "hour_limit_exceeded",
145 "retry_after": 3600,
146 "limit": self.config.requests_per_hour,
147 "window": "1 hour",
148 }
149
150 # Record this request
151 state.minute_requests.append(now)
152 state.hour_requests.append(now)
153 state.burst_requests.append(now)
154
155 # Return success with remaining counts
156 return True, {
157 "minute_remaining": self.config.requests_per_minute - len(state.minute_requests),
158 "hour_remaining": self.config.requests_per_hour - len(state.hour_requests),
159 "burst_remaining": self.config.burst_size - len(state.burst_requests),
160 }

162 def middleware(self, request: Request, call_next):
163 """
164 ASGI middleware for rate limiting.
165
166 Args:
167 request: FastAPI request
168 call_next: Next middleware/handler
169
170 Returns:
171 Response or raises HTTPException
172 """
173 # Skip rate limiting for health checks
174 if request.url.path == "/health":
175 return call_next(request)
176
177 allowed, info = self.check_rate_limit(request)
178
179 if not allowed:
180 raise HTTPException(
181 status_code=status.HTTP_429_TOO_MANY_REQUESTS,
182 detail={
183 "success": False,
184 "error": {
185 "code": info.get("error", "rate_limit_exceeded").upper(),
186 "message": f"Rate limit exceeded. Retry after {info.get('retry_after', 60)} seconds",
187 "retry_after": info.get("retry_after", 60),
188 },
189 },
190 headers={"Retry-After": str(info.get("retry_after", 60))},
191 )
192
193 # Add rate limit headers to response
194 response = call_next(request)
195 response.headers["X-RateLimit-Minute-Remaining"] = str(info.get("minute_remaining", 0))
196 response.headers["X-RateLimit-Hour-Remaining"] = str(info.get("hour_remaining", 0))
197
198 return response

237 def cleanup_stale_clients(self, max_age_seconds: int = 3600) -> int:
238 """
239 Remove clients with no recent activity.
240
241 Args:
242 max_age_seconds: Maximum age of inactive clients
243
244 Returns:
245 Number of clients removed
246 """
247 cutoff = time.time() - max_age_seconds
248 stale = []
249
250 for client_id, state in self._clients.items():
251 # Check if all request lists are empty or old
252 minute_old = all(t < cutoff - 60 for t in state.minute_requests)
253 hour_old = all(t < cutoff for t in state.hour_requests)
254 burst_old = all(t < cutoff - 1 for t in state.burst_requests)
255
256 if minute_old and hour_old and burst_old and state.blocked_until < cutoff:
257 stale.append(client_id)
258
259 for client_id in stale:
260 del self._clients[client_id]
261
262 return len(stale)

277def rate_limit_decorator(func: Callable) -> Callable:
278 """
279 Decorator for rate limiting individual endpoints.
280
281 Usage:
282 @app.get("/api/endpoint")
283 @rate_limit_decorator
284 async def my_endpoint(request: Request):
285 ...
286 """
287 @wraps(func)
288 async def wrapper(*args, **kwargs):
289 # Find request in args or kwargs
290 request = None
291 for arg in args:
292 if isinstance(arg, Request):
293 request = arg
294 break
295 if request is None:
296 request = kwargs.get("request")
297
298 if request:
299 limiter = get_rate_limiter()
300 allowed, info = limiter.check_rate_limit(request)
301 if not allowed:
302 raise HTTPException(
303 status_code=status.HTTP_429_TOO_MANY_REQUESTS,
304 detail={
305 "success": False,
306 "error": {
307 "code": "RATE_LIMIT_EXCEEDED",
308 "message": "Too many requests",
309 "retry_after": info.get("retry_after", 60),
310 },
311 },
312 headers={"Retry-After": str(info.get("retry_after", 60))},
313 )
314
315 return await func(*args, **kwargs)
316
317 return wrapper
