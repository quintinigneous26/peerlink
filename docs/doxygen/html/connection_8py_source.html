35 async def connect(
36 self,
37 device_id: str,
38 ws: WebSocket,
39 public_key: str,
40 capabilities: list[str],
41 metadata: dict | None = None,
42 ) -> DeviceInfo:
43 """
44 Register a new device connection.
45
46 Args:
47 device_id: Device identifier
48 ws: WebSocket connection
49 public_key: Device public key
50 capabilities: List of device capabilities
51 metadata: Optional metadata
52
53 Returns:
54 DeviceInfo
55 """
56 async with self._lock:
57 # Check if device already connected
58 if device_id in self._devices:
59 # Close old connection
60 old_device = self._devices[device_id]
61 try:
62 await old_device.ws.close()
63 except Exception:
64 pass
65
66 device = DeviceInfo(
67 device_id=device_id,
68 ws=ws,
69 public_key=public_key,
70 capabilities=capabilities,
71 metadata=metadata or {},
72 )
73
74 self._devices[device_id] = device
75 logger.info(f"Device connected: {device_id} ({len(self._devices)} total)")
76
77 return device

79 async def disconnect(self, device_id: str) -> None:
80 """
81 Disconnect a device.
82
83 Args:
84 device_id: Device identifier
85 """
86 async with self._lock:
87 if device_id in self._devices:
88 device = self._devices[device_id]
89
90 # Close any pending connection requests
91 to_remove = [
92 session_id
93 for session_id, requester in self._pending_requests.items()
94 if requester == device_id
95 ]
96 for session_id in to_remove:
97 del self._pending_requests[session_id]
98
99 # Update related sessions
100 for session in self._sessions.values():
101 if session.device_a == device_id or session.device_b == device_id:
102 session.status = ConnectionStatus.DISCONNECTED
103
104 del self._devices[device_id]
105 logger.info(f"Device disconnected: {device_id} ({len(self._devices)} total)")

135 async def send_message(self, device_id: str, message: dict) -> bool:
136 """
137 Send a message to a device.
138
139 Args:
140 device_id: Target device ID
141 message: Message dictionary
142
143 Returns:
144 True if sent successfully
145 """
146 device = self.get_device(device_id)
147 if not device:
148 return False
149
150 try:
151 await device.ws.send_json(message)
152 return True
153 except Exception as e:
154 logger.error(f"Failed to send message to {device_id}: {e}")
155 await self.disconnect(device_id)
156 return False

158 async def broadcast(self, message: dict, exclude: set[str] | None = None) -> int:
159 """
160 Broadcast a message to all connected devices.
161
162 Args:
163 message: Message dictionary
164 exclude: Device IDs to exclude
165
166 Returns:
167 Number of devices message was sent to
168 """
169 exclude = exclude or set()
170 count = 0
171
172 for device_id in self._devices:
173 if device_id not in exclude:
174 if await self.send_message(device_id, message):
175 count += 1
176
177 return count

290 Update device heartbeat timestamp.

304 async def cleanup_stale(self, timeout: int) -> int:
305 """
306 Clean up stale connections.
307
308 Args:
309 timeout: Seconds since last heartbeat
310
311 Returns:
312 Number of devices disconnected
313 """
314 import time
315 now = int(time.time())
316 count = 0
317
318 async with self._lock:
319 to_remove = []
320 for device_id, device in self._devices.items():
321 if now - device.last_heartbeat > timeout:
322 to_remove.append(device_id)
323
324 for device_id in to_remove:
325 await self.disconnect(device_id)
326 count += 1
327
328 return count
