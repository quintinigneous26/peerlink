83class ProtocolRegistry:
84 """
85 Registry for supported protocols and their handlers.
86
87 Maps protocol IDs to their handler implementations.
88 """
89
90 def __init__(self):
91 self._handlers: Dict[str, Callable[[], ProtocolHandler]] = {}
92
93 def register(self, protocol_id: str, handler_factory: Callable[[], ProtocolHandler]) -> None:
94 """
95 Register a protocol handler.
96
97 Args:
98 protocol_id: Protocol ID (e.g., "/ipfs/id/1.0.0")
99 handler_factory: Factory function to create handler instances
100 """
101 self._handlers[protocol_id] = handler_factory
102 logger.debug(f"Registered protocol: {protocol_id}")
103
104 def unregister(self, protocol_id: str) -> None:
105 """Unregister a protocol."""
106 if protocol_id in self._handlers:
107 del self._handlers[protocol_id]
108 logger.debug(f"Unregistered protocol: {protocol_id}")
109
110 def get_supported_protocols(self) -> list[str]:
111 """Get list of supported protocol IDs."""
112 return list(self._handlers.keys())
113
114 def create_handler(self, protocol_id: str) -> Optional[ProtocolHandler]:
115 """
116 Create a handler instance for the given protocol.
117
118 Args:
119 protocol_id: Protocol ID
120
121 Returns:
122 Handler instance or None if not supported
123 """
124 factory = self._handlers.get(protocol_id)
125 if factory:
126 return factory()
127 return None
128
129 def is_supported(self, protocol_id: str) -> bool:
130 """Check if a protocol is supported."""
131 return protocol_id in self._handlers

134class ProtocolNegotiationClient:
135 """
136 High-level client for protocol negotiation and execution.
137
138 Combines multistream-select negotiation with protocol handler execution.
139 """
140
141 def __init__(
142 self,
143 registry: ProtocolRegistry,
144 timeout: float = 30.0,
145 ):
146 """
147 Initialize the negotiation client.
148
149 Args:
150 registry: Protocol registry
151 timeout: Negotiation timeout in seconds
152 """
153 self.registry = registry
154 self.negotiator = ProtocolNegotiator(timeout=timeout)
155
156 async def dial(
157 self,
158 conn: StreamReaderWriter,
159 protocols: list[str],
160 ) -> tuple[str, Any]:
161 """
162 Dial a peer and negotiate a protocol.
163
164 Acts as the initiator in the negotiation.
165
166 Args:
167 conn: Connection to the peer
168 protocols: Preferred protocols (in priority order)
169
170 Returns:
171 Tuple of (negotiated_protocol, handler_result)
172
173 Raises:
174 NegotiationError: If negotiation fails
175 """
176 # Perform protocol negotiation
177 protocol = await self.negotiator.negotiate(conn, protocols)
178 logger.info(f"Negotiated protocol: {protocol}")
179
180 # Execute the protocol
181 handler = self.registry.create_handler(protocol)
182 if handler is None:
183 raise NegotiationError(f"No handler for protocol: {protocol}")
184
185 result = await handler.handle(conn)
186 return protocol, result
187
188 async def accept(
189 self,
190 conn: StreamReaderWriter,
191 ) -> tuple[str, Any]:
192 """
193 Accept an incoming connection and handle protocol negotiation.
194
195 Acts as the responder in the negotiation.
196
197 Args:
198 conn: Incoming connection
199
200 Returns:
201 Tuple of (negotiated_protocol, handler_result)
202
203 Raises:
204 NegotiationError: If negotiation fails
205 """
206 supported = self.registry.get_supported_protocols()
207
208 async def protocol_handler(protocol_id: str, conn: StreamReaderWriter):
209 """Handler called after successful negotiation."""
210 handler = self.registry.create_handler(protocol_id)
211 if handler is None:
212 raise NegotiationError(f"No handler for protocol: {protocol_id}")
213 return await handler.handle(conn)
214
215 # Perform protocol negotiation as responder
216 protocol = await self.negotiator.handle_negotiate(
217 conn,
218 supported,
219 protocol_handler,
220 )
221 logger.info(f"Accepted protocol: {protocol}")
222
223 # Result is already returned by protocol_handler
224 return protocol, None
