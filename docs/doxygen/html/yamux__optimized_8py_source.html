55class OptimizedYamuxStream(YamuxStream):
56 """
57 优化的 Yamux 流
58
59 改进：
60 - 读写锁分离
61 - 零拷贝数据路径
62 - 批量写入
63 """
64
65 def __init__(self, *args, **kwargs):
66 super().__init__(*args, **kwargs)
67
68 # 替换为读写锁分离
69 # 原: _read_lock + _write_lock (继承自父类)
70 # 优化: 使用不同的锁以减少竞争
71
72 # 批量写入缓冲
73 self._write_batch: List[bytes] = []
74 self._batch_size = 0
75 self._max_batch_size = 64 * 1024 # 64KB
76
77 async def write_batch(self, chunks: List[bytes]) -> int:
78 """
79 批量写入数据
80
81 比多次 write() 更高效，减少锁竞争。
82 """
83 if self._reset:
84 raise YamuxClosedError("Stream was reset")
85
86 if self._local_closed:
87 raise YamuxClosedError("Local side already closed")
88
89 if not chunks:
90 return 0
91
92 # 等待流建立
93 if not self._established.is_set():
94 await self._established.wait()
95
96 total_size = sum(len(c) for c in chunks)
97
98 # 发送单帧（更高效）
99 combined = b"".join(chunks)
100 frame = YamuxFrame(
101 type=FrameType.DATA,
102 stream_id=self._id,
103 length=len(combined),
104 data=combined
105 )
106 await self._session._send_frame(frame)
107
108 return total_size

113class StreamIDPool:
114 """
115 流 ID 预分配池
116
117 减少流创建时的锁竞争。
118 """
119
120 def __init__(
121 self,
122 start_id: int,
123 increment: int = STREAM_ID_INCREMENT,
124 pool_size: int = STREAM_ID_POOL_SIZE,
125 ):
126 self._next_id = start_id
127 self._increment = increment
128 self._pool: asyncio.Queue[int] = asyncio.Queue(maxsize=pool_size)
129 self._lock = asyncio.Lock()
130
131 # 预填充池
132 self._refill(pool_size)
133
134 async def get(self) -> int:
135 """获取流 ID（无锁快速路径）"""
136 try:
137 return self._pool.get_nowait()
138 except asyncio.QueueEmpty:
139 # 池空，补充
140 await self._refill_async()
141 return await self._pool.get()
142
143 def _refill(self, count: int) -> None:
144 """补充池（同步）"""
145 for _ in range(count):
146 if self._pool.full():
147 break
148 stream_id = self._next_id
149 self._next_id += self._increment
150 try:
151 self._pool.put_nowait(stream_id)
152 except asyncio.QueueFull:
153 break
154
155 async def _refill_async(self) -> None:
156 """异步补充池"""
157 async with self._lock:
158 self._refill(min(self._pool.maxsize, self._pool.maxsize // 2))
159
160 @property
161 def available(self) -> int:
162 """可用 ID 数量"""
163 return self._pool.qsize()

168class OptimizedYamuxSession(YamuxSession):
169 """
170 优化的 Yamux 会话
171
172 改进：
173 1. 流 ID 预分配池
174 2. 批量流创建
175 3. 快速路径帧发送
176 4. 连接统计
177 """
178
179 def __init__(self, *args, **kwargs):
180 super().__init__(*args, **kwargs)
181
182 # 流 ID 池（替代原来的单一锁分配）
183 self._stream_id_pool = StreamIDPool(
184 start_id=self._next_stream_id,
185 increment=STREAM_ID_INCREMENT,
186 )
187
188 # 统计信息
189 self._stats = {
190 "streams_opened": 0,
191 "streams_closed": 0,
192 "frames_sent": 0,
193 "frames_received": 0,
194 "bytes_sent": 0,
195 "bytes_received": 0,
196 }
197
198 async def open_stream(self) -> OptimizedYamuxStream:
199 """
200 打开新流（优化版）
201
202 使用预分配池，无锁快速路径。
203 """
204 if self._closed:
205 raise YamuxClosedError("Session closed")
206
207 # 从池获取流 ID（无锁）
208 stream_id = await self._stream_id_pool.get()
209
210 async with self._streams_lock:
211 if stream_id in self._streams:
212 raise YamuxProtocolError(f"Stream ID conflict: {stream_id}")
213
214 stream = OptimizedYamuxStream(
215 stream_id=stream_id,
216 session=self,
217 is_initiator=True,
218 config=self._config,
219 )
220 self._streams[stream_id] = stream
221
222 # 发送 SYN
223 frame = YamuxFrame(
224 type=FrameType.DATA,
225 flags=FrameFlag.SYN,
226 stream_id=stream_id,
227 length=0
228 )
229 await self._send_frame(frame)
230
231 stream._set_state(StreamState.SYN_SENT)
232
233 self._stats["streams_opened"] += 1
234
235 return stream
236
237 async def open_streams_batch(
238 self,
239 count: int,
240 ) -> List[OptimizedYamuxStream]:
241 """
242 批量打开流
243
244 比单独调用 open_stream() 更高效。
245 """
246 if self._closed:
247 raise YamuxClosedError("Session closed")
248
249 if count > BATCH_STREAM_CREATE_MAX:
250 raise YamuxProtocolError(f"Batch size too large: {count}")
251
252 streams = []
253 stream_ids = []
254
255 # 批量获取流 ID
256 for _ in range(count):
257 stream_id = await self._stream_id_pool.get()
258 stream_ids.append(stream_id)
259
260 async with self._streams_lock:
261 for stream_id in stream_ids:
262 if stream_id in self._streams:
263 continue
264
265 stream = OptimizedYamuxStream(
266 stream_id=stream_id,
267 session=self,
268 is_initiator=True,
269 config=self._config,
270 )
271 self._streams[stream_id] = stream
272 streams.append(stream)
273
274 # 批量发送 SYN 帧
275 syn_frames = [
276 YamuxFrame(
277 type=FrameType.DATA,
278 flags=FrameFlag.SYN,
279 stream_id=s._id,
280 length=0
281 )
282 for s in streams
283 ]
284
285 await self._send_frames_batch(syn_frames)
286
287 for stream in streams:
288 stream._set_state(StreamState.SYN_SENT)
289
290 self._stats["streams_opened"] += len(streams)
291
292 return streams
293
294 async def _send_frames_batch(self, frames: List[YamuxFrame]) -> None:
295 """
296 批量发送帧
297
298 减少系统调用次数。
299 """
300 if self._closed:
301 raise YamuxClosedError("Session closed")
302
303 # 组合所有帧
304 data = b"".join(f.pack() for f in frames)
305
306 self._writer.write(data)
307 await self._writer.drain()
308
309 self._stats["frames_sent"] += len(frames)
310 self._stats["bytes_sent"] += len(data)
311
312 async def _send_frame(self, frame: YamuxFrame) -> None:
313 """发送单帧（覆盖父类方法，添加统计）"""
314 if self._closed:
315 raise YamuxClosedError("Session closed")
316
317 data = frame.pack()
318 self._writer.write(data)
319 await self._writer.drain()
320
321 self._stats["frames_sent"] += 1
322 self._stats["bytes_sent"] += len(data)
323
324 @property
325 def statistics(self) -> Dict[str, int]:
326 """获取会话统计"""
327 return self._stats.copy()
328
329 def get_performance_metrics(self) -> Dict[str, float]:
330 """获取性能指标"""
331 uptime = time.time() - self._stats.get("created_at", time.time())
332
333 return {
334 "uptime_seconds": uptime,
335 "streams_per_second": self._stats["streams_opened"] / uptime if uptime > 0 else 0,
336 "avg_frame_size": (
337 self._stats["bytes_sent"] / self._stats["frames_sent"]
338 if self._stats["frames_sent"] > 0
339 else 0
340 ),
341 "active_streams": len(self._streams),
342 "pool_available": self._stream_id_pool.available,
343 }
