37async def send_stun_request(
38 host: str, port: int, timeout: float = 2.0
39) -> Optional[Tuple[str, int]]:
40 """
41 Send STUN binding request and get mapped address.
42
43 Args:
44 host: STUN server hostname
45 port: STUN server port
46 timeout: Request timeout in seconds
47
48 Returns:
49 Tuple of (mapped_ip, mapped_port) or None
50 """
51 from .messages import (
52 MAGIC_COOKIE,
53 MessageType,
54 StunMessage,
55 parse_xor_mapped_address,
56 )
57
58 # Create STUN binding request
59 transaction_id = asyncio.get_event_loop().time_ns().to_bytes(12, "big")
60
61 message = StunMessage(
62 message_type=MessageType.BINDING_REQUEST,
63 message_length=0,
64 magic_cookie=MAGIC_COOKIE,
65 transaction_id=transaction_id,
66 attributes=[],
67 )
68
69 request_data = message.serialize()
70
71 try:
72 # Create UDP socket
73 sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
74 sock.settimeout(timeout)
75
76 # Send request
77 sock.sendto(request_data, (host, port))
78
79 # Receive response
80 data, addr = sock.recvfrom(512)
81 sock.close()
82
83 # Parse response
84 response = StunMessage.parse(data)
85 if response and response.message_type == MessageType.BINDING_RESPONSE:
86 # Get XOR-MAPPED-ADDRESS attribute
87 attr = response.get_attribute(0x0020) # XOR-MAPPED-ADDRESS
88 if attr:
89 result = parse_xor_mapped_address(attr.value)
90 if result:
91 return result
92
93 return None
94
95 except (socket.timeout, socket.error, OSError):
96 return None

99async def detect_nat_type(
100 stun_server: str,
101 stun_port: int = 3478,
102 stun_alt_server: Optional[str] = None,
103 stun_alt_port: Optional[int] = None,
104) -> NATTestResult:
105 """
106 Detect NAT type using RFC 3489 algorithm.
107
108 This performs the STUN test sequence:
109 1. Test I: Request to server A from same address/port
110 2. Test II: Request to server A with changed IP and port
111 3. Test III: Request to server A with changed port only
112
113 Args:
114 stun_server: Primary STUN server address
115 stun_port: Primary STUN server port
116 stun_alt_server: Alternate STUN server (for test II)
117 stun_alt_port: Alternate STUN server port (for test III)
118
119 Returns:
120 NATTestResult with detected NAT type
121 """
122 # Default to using same server for tests
123 if not stun_alt_server:
124 stun_alt_server = stun_server
125 if not stun_alt_port:
126 stun_alt_port = stun_port
127
128 # Test I: Basic binding request
129 result = await send_stun_request(stun_server, stun_port)
130
131 if result is None:
132 # No response at all - likely blocked by firewall
133 return NATTestResult(
134 nat_type=NATType.BLOCKED,
135 public_ip=None,
136 public_port=None,
137 changed_ip_response=False,
138 changed_port_response=False,
139 )
140
141 public_ip, public_port = result
142
143 # Test II: Request from different address (simulated by using alt server)
144 # In practice, this requires the server to respond from a different IP
145 alt_result = await send_stun_request(stun_alt_server, stun_alt_port)
146 changed_ip_response = alt_result is not None
147
148 # Test III: Request from different port
149 # In practice, this requires server to respond from different port
150 changed_port_response = changed_ip_response # Simplified
151
152 # Determine NAT type based on RFC 3489 algorithm
153 if not changed_ip_response:
154 # No response from changed address/port
155 if public_ip == get_local_ip():
156 # Public IP matches local - no NAT
157 nat_type = NATType.OPEN_INTERNET
158 else:
159 # NAT detected
160 nat_type = NATType.SYMMETRIC
161 else:
162 # Response from changed address
163 if public_ip == get_local_ip():
164 nat_type = NATType.OPEN_INTERNET
165 else:
166 # Further tests needed to distinguish between cone types
167 # For simplicity, default to full cone
168 nat_type = NATType.FULL_CONE
169
170 return NATTestResult(
171 nat_type=nat_type,
172 public_ip=public_ip,
173 public_port=public_port,
174 changed_ip_response=changed_ip_response,
175 changed_port_response=changed_port_response,
176 )
