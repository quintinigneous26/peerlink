26@dataclass
27class ProviderRecord:
28 """
29 Record of a peer providing a specific content key.
30
31 Attributes:
32 peer_id: ID of the provider peer
33 key: Content key (CID hash)
34 timestamp: When this record was created
35 expires: When this record expires
36 """
37 peer_id: bytes
38 key: bytes
39 timestamp: float = field(default_factory=time.time)
40 expires: float = field(default_factory=lambda: time.time() + PROVIDER_EXPIRATION)
41
42 def is_expired(self) -> bool:
43 """Check if provider record has expired."""
44 return time.time() > self.expires
45
46 def refresh(self, ttl: float = PROVIDER_EXPIRATION) -> None:
47 """Refresh the provider record."""
48 self.timestamp = time.time()
49 self.expires = time.time() + ttl

52def compute_key(key: bytes) -> bytes:
53 """
54 Compute the DHT key for content.
55
56 Uses SHA-256 hash of the input key.

58 Args:
59 key: Input key (can be CID bytes or any content identifier)

61 Returns:
62 32-byte DHT key

75 def __init__(self, max_providers: int = MAX_PROVIDERS_PER_KEY):
76 """
77 Initialize provider manager.
78
79 Args:
80 max_providers: Maximum providers to track per key
81 """
82 self.max_providers = max_providers
83 self._providers: Dict[bytes, Set[bytes]] = defaultdict(set) # key -> peer_ids
84 self._records: Dict[tuple[bytes, bytes], ProviderRecord] = {} # (key, peer_id) -> record
85 self._lock = asyncio.Lock()
86 self._local_providers: Set[bytes] = set() # Keys we're providing

88 async def add_provider(
89 self,
90 key: bytes,
91 peer_id: bytes,
92 ttl: float = PROVIDER_EXPIRATION
93 ) -> bool:
94 """
95 Add a provider record.
96
97 Args:
98 key: Content key (CID)
99 peer_id: ID of the provider peer
100 ttl: Time to live for this record
101
102 Returns:
103 True if provider was added
104 """
105 dht_key = compute_key(key)
106 record_key = (dht_key, peer_id)
107
108 async with self._lock:
109 # Check if key has too many providers
110 if len(self._providers[dht_key]) >= self.max_providers:
111 if peer_id not in self._providers[dht_key]:
112 # Replace oldest provider
113 oldest = None
114 oldest_time = float('inf')
115 for pid in self._providers[dht_key]:
116 rec = self._records.get((dht_key, pid))
117 if rec and rec.timestamp < oldest_time:
118 oldest_time = rec.timestamp
119 oldest = pid
120
121 if oldest:
122 self._providers[dht_key].remove(oldest)
123 del self._records[(dht_key, oldest)]
124
125 # Add or update record
126 if peer_id in self._providers[dht_key]:
127 # Update existing record
128 self._records[record_key].refresh(ttl)
129 else:
130 # Create new record
131 self._providers[dht_key].add(peer_id)
132 self._records[record_key] = ProviderRecord(
133 peer_id=peer_id,
134 key=dht_key,
135 expires=time.time() + ttl
136 )
137
138 return True

140 async def remove_provider(self, key: bytes, peer_id: bytes) -> bool:
141 """
142 Remove a provider record.
143
144 Args:
145 key: Content key (CID)
146 peer_id: ID of the provider peer
147
148 Returns:
149 True if provider was found and removed

151 dht_key = compute_key(key)

162
163 return True

165 return False

167 def get_providers(self, key: bytes, max_count: int = 20) -> List[ProviderRecord]:
168 """
169 Get providers for a content key.
170
171 Args:
172 key: Content key (CID)
173 max_count: Maximum number of providers to return
174
175 Returns:
176 List of provider records
177 """
178 dht_key = compute_key(key)
179
180 # Get provider IDs for this key
181 peer_ids = list(self._providers.get(dht_key, set()))
182
183 # Build list of records
184 records: List[ProviderRecord] = []
185 for peer_id in peer_ids[:max_count]:
186 record = self._records.get((dht_key, peer_id))
187 if record and not record.is_expired():
188 records.append(record)
189
190 return records

245 async def cleanup_expired(self) -> int:
246 """
247 Remove expired provider records.
248
249 Returns:
250 Number of records removed
251 """
252 async with self._lock:
253 now = time.time()
254 to_remove: List[tuple[bytes, bytes]] = []
255
256 for record_key, record in self._records.items():
257 if record.expires < now:
258 to_remove.append(record_key)
259
260 for dht_key, peer_id in to_remove:
261 del self._records[(dht_key, peer_id)]
262 self._providers[dht_key].discard(peer_id)
263
264 # Clean up empty sets
265 if not self._providers[dht_key]:
266 del self._providers[dht_key]
267
268 return len(to_remove)

303def create_provider_message(
304 key: bytes,
305 peer_id: bytes
306) -> bytes:
307 """
308 Create a provider announcement message.
309
310 Args:
311 key: Content key being provided
312 peer_id: ID of the provider peer
313
314 Returns:
315 Encoded message bytes
316 """
317 # Simple encoding: key_len (1 byte) + key + peer_id (32 bytes)
318 key_len = len(key)
319 if key_len > 255:
320 raise ValueError("Key too long")
321
322 return struct.pack('>B', key_len) + key + peer_id

325def parse_provider_message(data: bytes) -> tuple[bytes, bytes]:
326 """
327 Parse a provider announcement message.
328
329 Args:
330 data: Encoded message bytes
331
332 Returns:
333 Tuple of (key, peer_id)
334
335 Raises:
336 ValueError: If message is invalid
337 """
338 if len(data) < 33:
339 raise ValueError("Invalid provider message")
340
341 key_len = data[0]
342 key = data[1:1 + key_len]
343
344 if len(data) < 1 + key_len + 32:
345 raise ValueError("Invalid provider message")
346
347 peer_id = data[1 + key_len:1 + key_len + 32]
348
349 return key, peer_id
