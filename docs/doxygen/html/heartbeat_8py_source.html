101 async def _heartbeat_loop(self) -> None:
102 """心跳循环"""
103 while self._running:
104 try:
105 # 发送心跳
106 success = await self._send_heartbeat()
107
108 if success:
109 self._on_heartbeat_success()
110 else:
111 await self._on_heartbeat_failure()
112
113 # 等待下一次心跳
114 await asyncio.sleep(self._interval_ms / 1000)
115
116 except asyncio.CancelledError:
117 break
118 except Exception as e:

123 """发送心跳包"""
124 try:
125 start_time = time.time()

127 success = await self._send_func(self._heartbeat_data)

129 latency = (time.time() - start_time) * 1000

138 def _on_heartbeat_success(self) -> None:
139 """心跳成功处理"""
140 self._stats.total_received += 1
141 self._stats.consecutive_success += 1
142 self._stats.consecutive_failure = 0
143 self._stats.last_success_time = time.time()
144
145 # 更新平均延迟
146 if self._stats.total_received == 1:
147 self._stats.avg_latency_ms = self._stats.last_latency_ms
148 else:
149 # 指数移动平均
150 alpha = 0.2
151 self._stats.avg_latency_ms = (
152 alpha * self._stats.last_latency_ms +
153 (1 - alpha) * self._stats.avg_latency_ms
154 )
155
156 # 自适应延长心跳间隔
157 if self._profile and self._profile.heartbeat_max_ms > self._interval_ms:
158 if self._stats.consecutive_success >= self._profile.success_to_extend:
159 new_interval = min(
160 self._interval_ms + 5000,
161 self._profile.heartbeat_max_ms
162 )
163 if new_interval != self._interval_ms:
164 self._interval_ms = new_interval
165 logger.debug(f"心跳间隔延长至 {self._interval_ms}ms")

167 async def _on_heartbeat_failure(self) -> None:
168 """心跳失败处理"""
169 self._stats.consecutive_failure += 1
170 self._stats.consecutive_success = 0
171
172 logger.warning(f"心跳失败 ({self._stats.consecutive_failure}次)")
173
174 # 自适应缩短心跳间隔
175 if self._profile:
176 self._interval_ms = max(
177 self._interval_ms - 5000,
178 self._profile.heartbeat_min_ms
179 )
180
181 # 连续失败超过阈值，触发超时回调
182 if self._stats.consecutive_failure >= 3:
183 logger.error("心跳连续失败 3 次，触发超时")
184 if self._on_timeout:
185 await self._on_timeout()
