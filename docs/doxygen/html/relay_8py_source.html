46def parse_stun_message(data: bytes) -> Optional[StunMessage]:
47 """Parse STUN message from bytes."""
48 if len(data) < 20:
49 return None
50
51 message_type, message_length, magic_cookie = struct.unpack("!HHI", data[:8])
52
53 if magic_cookie != MAGIC_COOKIE:
54 return None
55
56 transaction_id = data[8:20]
57
58 # Parse attributes
59 attributes: List[StunAttribute] = []
60 offset = 20
61 remaining = message_length
62
63 while remaining > 0 and offset + 4 <= len(data):
64 attr_type, attr_length = struct.unpack("!HH", data[offset:offset + 4])
65 offset += 4
66
67 padded_length = (attr_length + 3) & ~3
68
69 if offset + padded_length > len(data):
70 break
71
72 attr_value = data[offset:offset + attr_length]
73 attributes.append(StunAttribute(type=attr_type, value=attr_value))
74
75 offset += padded_length
76 remaining -= (4 + padded_length)
77
78 return StunMessage(
79 message_type=message_type,
80 message_length=message_length,
81 magic_cookie=magic_cookie,
82 transaction_id=transaction_id,
83 attributes=attributes,
84 )

87def create_error_response(transaction_id: bytes, error_code: ErrorCode, message: str) -> bytes:
88 """Create STUN error response."""
89 class_digit = error_code // 100
90 number = error_code % 100
91
92 reason = message.encode("utf-8")
93 error_attr = struct.pack(
94 "!HHBB",
95 AttributeType.ERROR_CODE,
96 len(reason) + 4,
97 0,
98 class_digit
99 )
100 error_attr += struct.pack("!B", number) + reason
101
102 padding = (4 - ((len(reason) + 4) % 4)) % 4
103 error_attr += b"\x00" * padding
104
105 msg = StunMessage(
106 message_type=MessageType.BINDING_ERROR_RESPONSE,
107 message_length=len(error_attr),
108 magic_cookie=MAGIC_COOKIE,
109 transaction_id=transaction_id,
110 attributes=[
111 StunAttribute(type=AttributeType.ERROR_CODE, value=error_attr[4:])
112 ],
113 )
114
115 return _serialize_stun_message(msg)

118def _serialize_stun_message(msg: StunMessage) -> bytes:
119 """Serialize STUN message to bytes."""
120 attr_data = b""
121 for attr in msg.attributes:
122 attr_header = struct.pack("!HH", attr.type, len(attr.value))
123 attr_data += attr_header + attr.value
124 padding = (4 - (len(attr.value) % 4)) % 4
125 attr_data += b"\x00" * padding
126
127 message_length = len(attr_data)
128 header = struct.pack(
129 "!HHI",
130 msg.message_type,
131 message_length,
132 msg.magic_cookie,
133 )
134
135 return header + msg.transaction_id + attr_data

229 async def _handle_control_message(
230 self, data: bytes, client_addr: Tuple[str, int]
231 ) -> Optional[bytes]:
232 """
233 Handle TURN control message.
234
235 Args:
236 data: Message data
237 client_addr: Client's (ip, port)
238
239 Returns:
240 Response bytes or None
241 """
242 # Parse STUN message
243 message = StunMessage.parse(data)
244
245 if message is None:
246 logger.warning(f"Invalid STUN message from {client_addr}")
247 return None
248
249 # Handle based on message class (method)
250 message_class = message.message_type >> 6
251
252 if message_class == 0x01: # Binding request (STUN)
253 # Return binding response
254 return self._create_binding_response(message, client_addr)
255
256 elif message_class == 0x03: # ALLOCATE
257 return await self._handle_allocate(message, client_addr)
258
259 elif message_class == 0x04: # REFRESH
260 return await self._handle_refresh(message, client_addr)
261
262 elif message_class == 0x06: # SEND
263 return await self._handle_send(message, client_addr)
264
265 elif message_class == 0x08: # CREATE_PERMISSION
266 return await self._handle_create_permission(message, client_addr)
267
268 else:
269 logger.warning(f"Unknown message class {message_class} from {client_addr}")
270 return create_error_response(
271 message.transaction_id,
272 ErrorCode.BAD_REQUEST,
273 f"Unknown message class: {message_class}",
274 )

308 TurnErrorCode.ALLOCATION_MISMATCH,
309 "Client already has allocation",
310 )
311
312 # Get requested lifetime

314 lifetime = None

519 async def _relay_loop(self, allocation: TurnAllocation, sock: socket.socket):
520 """Relay data loop for an allocation."""
521 loop = asyncio.get_event_loop()
522
523 while not allocation.is_expired():
524 try:
525 data, peer_addr = await asyncio.wait_for(
526 loop.sock_recvfrom(sock, 1500),
527 timeout=1.0,
528 )
529
530 # Check bandwidth
531 if not await self.bandwidth_limiter.throttle_read(
532 allocation.allocation_id, len(data)
533 ):
534 continue
535
536 # Update stats
537 allocation.record_received(len(data))
538 allocation.peer_addr = peer_addr
539 await self.throughput_monitor.record_read(len(data))
540
541 # In real implementation, relay data to client
542 # This would send a DATA indication back to the client
543
544 except asyncio.TimeoutError:
545 continue
546 except asyncio.CancelledError:
547 break
548 except Exception as e:
549 logger.error(f"Error in relay loop: {e}")
550
551 sock.close()
552 self._relay_sockets.pop(allocation.relay_addr[1], None)
