204def create_xor_address_attr(ip: str, port: int, transaction_id: bytes) -> bytes:
205 """
206 Create XOR address attribute (for XOR-MAPPED-ADDRESS, XOR-PEER-ADDRESS, XOR-RELAYED-ADDRESS).
207
208 Args:
209 ip: IP address string
210 port: Port number
211 transaction_id: 96-bit transaction ID
212
213 Returns:
214 Attribute value bytes
215 """
216 cookie_bytes = struct.pack("!I", MAGIC_COOKIE)
217
218 # Determine IP family
219 if "." in ip:
220 family = 0x01 # IPv4
221 ip_bytes = bytes(map(int, ip.split(".")))
222
223 # XOR port with (magic_cookie >> 16)
224 xor_port = port ^ ((MAGIC_COOKIE >> 16) & 0xFFFF)
225
226 # XOR IP with magic cookie
227 xor_ip = bytes(ip_bytes[i] ^ cookie_bytes[i] for i in range(4))
228 else:
229 family = 0x02 # IPv6
230 ip_bytes = bytes.fromhex(ip.replace(":", ""))
231
232 # XOR port
233 xor_port = port ^ ((MAGIC_COOKIE >> 16) & 0xFFFF)
234
235 # XOR IP with magic cookie + transaction ID
236 xor_ip = bytes(
237 ip_bytes[i] ^ (cookie_bytes[i % 4] if i < 4 else transaction_id[i - 4])
238 for i in range(16)
239 )
240
241 return struct.pack("!BBH", 0, family, xor_port) + xor_ip
