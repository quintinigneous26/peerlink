96@dataclass
97class YamuxFrame:
98 """Yamux 帧结构"""
99 version: int = VERSION
100 type: FrameType = FrameType.DATA
101 flags: int = 0
102 stream_id: int = 0
103 length: int = 0
104 data: bytes = b""
105
106 def pack(self) -> bytes:
107 """打包帧为字节"""
108 header = struct.pack(
109 ">BBHII",
110 self.version,
111 self.type,
112 self.flags,
113 self.stream_id,
114 self.length
115 )
116 return header + self.data
117
118 @classmethod
119 def unpack(cls, data: bytes) -> 'YamuxFrame':
120 """从字节解包帧"""
121 if len(data) < HEADER_SIZE:
122 raise ValueError(f"帧数据过短: {len(data)} < {HEADER_SIZE}")
123
124 version, type_val, flags, stream_id, length = struct.unpack(">BBHII", data[:HEADER_SIZE])
125
126 frame = cls(
127 version=version,
128 type=FrameType(type_val),
129 flags=flags,
130 stream_id=stream_id,
131 length=length,
132 data=data[HEADER_SIZE:HEADER_SIZE + length] if len(data) > HEADER_SIZE else b""
133 )
134
135 return frame
136
137 def has_flag(self, flag: FrameFlag) -> bool:
138 """检查是否包含指定标志"""
139 return (self.flags & flag) != 0
140
141 def add_flag(self, flag: FrameFlag) -> None:
142 """添加标志"""
143 self.flags |= flag
144
145 def __repr__(self) -> str:
146 flags_str = []
147 if self.has_flag(FrameFlag.SYN):
148 flags_str.append("SYN")
149 if self.has_flag(FrameFlag.ACK):
150 flags_str.append("ACK")
151 if self.has_flag(FrameFlag.FIN):
152 flags_str.append("FIN")
153 if self.has_flag(FrameFlag.RST):
154 flags_str.append("RST")
155
156 return (f"YamuxFrame(type={self.type.name}, "
157 f"flags={'|'.join(flags_str) if flags_str else '0'}, "
158 f"stream_id={self.stream_id}, length={self.length})")

270 async def read(self, n: int = -1) -> bytes:
271 """
272 读取流数据
273
274 Args:
275 n: 读取字节数，-1 表示读取所有可用数据
276
277 Returns:
278 读取的数据
279
280 Raises:
281 YamuxStreamClosed: 流已关闭
282 YamuxStreamReset: 流被重置
283 """
284 async with self._read_lock:
285 if self._reset:
286 raise YamuxStreamReset("流已被重置")
287
288 if self._remote_closed and not self._recv_buffer:
289 return b""
290
291 data = bytearray()
292
293 if n < 0:
294 # 读取所有可用数据
295 while True:
296 if self._recv_buffer:
297 data.extend(self._recv_buffer)
298 self._recv_buffer.clear()
299
300 if self._remote_closed:
301 break
302
303 try:
304 chunk = await asyncio.wait_for(
305 self._recv_queue.get(),
306 timeout=0.1
307 )
308 if chunk is None:
309 break
310 data.extend(chunk)
311 except asyncio.TimeoutError:
312 if data:
313 break
314 continue
315 else:
316 # 读取指定字节数
317 while len(data) < n:
318 if self._recv_buffer:
319 needed = n - len(data)
320 available = len(self._recv_buffer)
321 to_read = min(needed, available)
322 data.extend(self._recv_buffer[:to_read])
323 del self._recv_buffer[:to_read]
324 continue
325
326 if self._remote_closed:
327 break
328
329 try:
330 chunk = await asyncio.wait_for(
331 self._recv_queue.get(),
332 timeout=0.1
333 )
334 if chunk is None:
335 break
336 self._recv_buffer.extend(chunk)
337 except asyncio.TimeoutError:
338 if data:
339 break
340 continue
341
342 # 发送窗口更新
343 if data:
344 await self._send_window_update_internal(len(data))
345
346 return bytes(data)

348 async def write(self, data: bytes) -> int:
349 """
350 写入流数据
351
352 Args:
353 data: 要写入的数据
354
355 Returns:
356 写入的字节数
357
358 Raises:
359 YamuxStreamClosed: 流已关闭
360 YamuxStreamReset: 流被重置
361 YamuxWindowExceeded: 发送窗口已满
362 """
363 async with self._write_lock:
364 if self._reset:
365 raise YamuxStreamReset("流已被重置")
366
367 if self._local_closed:
368 raise YamuxStreamClosed("本地已关闭")
369
370 if not data:
371 return 0
372
373 # 等待流建立
374 if not self._established.is_set():
375 await self._established.wait()
376
377 written = 0
378 total_len = len(data)
379 offset = 0
380
381 while offset < total_len:
382 # 等待窗口
383 while self._send_window <= 0:
384 if self._reset or self._local_closed:
385 raise YamuxStreamClosed("写入时流已关闭")
386 self._send_window_update.clear()
387 await self._send_window_update.wait()
388
389 # 计算可发送大小
390 chunk_size = min(self._send_window, total_len - offset)
391 chunk = data[offset:offset + chunk_size]
392
393 # 发送数据帧
394 frame = YamuxFrame(
395 type=FrameType.DATA,
396 stream_id=self._id,
397 length=len(chunk),
398 data=chunk
399 )
400 await self._session._send_frame(frame)
401
402 # 更新状态
403 self._send_window -= len(chunk)
404 offset += chunk_size
405 written = offset
406
407 return written
408
409 async def close(self) -> None:
410 """
411 半关闭流 (关闭写端)
412
413 发送 FIN 标志表示不再发送数据，但仍可接收。
414 """
415 async with self._write_lock:
416 if self._local_closed:
417 return
418
419 self._local_closed = True
420
421 frame = YamuxFrame(
422 type=FrameType.DATA,
423 flags=FrameFlag.FIN,
424 stream_id=self._id,
425 length=0
426 )
427 await self._session._send_frame(frame)
428
429 # 检查是否完全关闭
430 if self._remote_closed:
431 self._state = StreamState.CLOSED

433 async def reset(self) -> None:
434 """
435 立即重置流
436
437 发送 RST 标志立即关闭流，丢弃所有缓冲数据。
438 """
439 self._reset = True
440 self._state = StreamState.RESET
441
442 frame = YamuxFrame(
443 type=FrameType.DATA,
444 flags=FrameFlag.RST,
445 stream_id=self._id,
446 length=0
447 )
448 await self._session._send_frame(frame)
449
450 # 清空接收队列
451 while not self._recv_queue.empty():
452 try:
453 self._recv_queue.get_nowait()
454 except asyncio.QueueEmpty:
455 break
456
457 # 通知等待的读者
458 await self._recv_queue.put(None)

460 async def _handle_data(self, frame: YamuxFrame) -> None:
461 """处理接收到的数据帧"""
462 if self._reset:
463 return
464
465 if frame.has_flag(FrameFlag.RST):
466 self._reset = True
467 self._state = StreamState.RESET
468 await self._recv_queue.put(None)
469 return
470
471 if frame.has_flag(FrameFlag.FIN):
472 self._remote_closed = True
473 self._state = StreamState.LOCAL_CLOSE if self._local_closed else StreamState.REMOTE_CLOSE
474 await self._recv_queue.put(None)
475 return
476
477 if frame.data:
478 self._recv_window -= len(frame.data)
479 await self._recv_queue.put(frame.data)
480
481 async def _handle_window_update(self, frame: YamuxFrame) -> None:
482 """处理窗口更新"""
483 delta = frame.length
484 self._send_window += delta
485 self._send_window_update.set()
486
487 async def _send_window_update_internal(self, delta: int) -> None:
488 """发送窗口更新"""
489 frame = YamuxFrame(
490 type=FrameType.WINDOW_UPDATE,
491 stream_id=self._id,
492 length=delta
493 )
494 await self._session._send_frame(frame)
495 self._recv_window += delta
496
497 def _set_state(self, state: StreamState) -> None:
498 """设置流状态"""
499 self._state = state
500 if state == StreamState.ESTABLISHED:
501 self._established.set()

600 async def open_stream(self) -> YamuxStream:
601 """
602 打开新流
603
604 发送 SYN 标志初始化新流。
605
606 Returns:
607 新的 YamuxStream 实例
608
609 Raises:
610 YamuxClosedError: 会话已关闭
611 YamuxProtocolError: 协议错误
612 """
613 if self._closed:
614 raise YamuxClosedError("会话已关闭")
615
616 async with self._stream_id_lock:
617 stream_id = self._next_stream_id
618 self._next_stream_id += STREAM_ID_INCREMENT
619
620 async with self._streams_lock:
621 if stream_id in self._streams:
622 raise YamuxProtocolError(f"流ID冲突: {stream_id}")
623
624 stream = YamuxStream(
625 stream_id=stream_id,
626 session=self,
627 is_initiator=True,
628 config=self._config
629 )
630 self._streams[stream_id] = stream
631
632 # 发送 SYN
633 frame = YamuxFrame(
634 type=FrameType.DATA,
635 flags=FrameFlag.SYN,
636 stream_id=stream_id,
637 length=0
638 )
639 await self._send_frame(frame)
640
641 stream._set_state(StreamState.SYN_SENT)
642 self._unacked_streams += 1
643
644 return stream

655 """

657 raise YamuxClosedError("会话已关闭")

661 async def close(self, code: GoAwayCode = GoAwayCode.NORMAL) -> None:
662 """
663 关闭会话
664
665 Args:
666 code: 关闭原因码
667 """
668 async with self._close_lock:
669 if self._closed:
670 return
671
672 self._closed = True
673 self._goaway_code = code
674
675 # 发送 GoAway
676 frame = YamuxFrame(
677 type=FrameType.GO_AWAY,
678 stream_id=0,
679 length=code
680 )
681 await self._send_frame(frame)
682
683 # 关闭所有流
684 async with self._streams_lock:
685 for stream in self._streams.values():
686 if not stream.is_closed:
687 await stream.reset()
688
689 # 停止任务
690 if self._read_task:
691 self._read_task.cancel()
692 if self._keepalive_task:
693 self._keepalive_task.cancel()
694
695 # 关闭连接
696 try:
697 self._writer.close()
698 await self._writer.wait_closed()
699 except Exception:
700 pass

702 async def ping(self, timeout: float = 5.0) -> float:
703 """
704 发送 Ping 测量 RTT
705
706 Args:
707 timeout: 超时时间(秒)
708
709 Returns:
710 往返时间(秒)
711
712 Raises:
713 YamuxClosedError: 会话已关闭
714 asyncio.TimeoutError: Ping 超时
715 """
716 if self._closed:
717 raise YamuxClosedError("会话已关闭")
718
719 async with self._ping_lock:
720 ping_id = self._ping_counter
721 self._ping_counter = (self._ping_counter + 1) & 0xFFFFFFFF
722
723 future: asyncio.Future[float] = asyncio.Future()
724 self._pending_pings[ping_id] = future
725
726 frame = YamuxFrame(
727 type=FrameType.PING,
728 flags=FrameFlag.SYN,
729 stream_id=0,
730 length=ping_id
731 )
732 start_time = asyncio.get_event_loop().time()
733 await self._send_frame(frame)
734
735 try:
736 await asyncio.wait_for(future, timeout=timeout)
737 rtt = asyncio.get_event_loop().time() - start_time
738 return rtt
739 except asyncio.TimeoutError:
740 self._pending_pings.pop(ping_id, None)
741 raise

744 """发送帧"""

746 raise YamuxClosedError("会话已关闭")
747
748 data = frame.pack()
749 self._writer.write(data)
750 await self._writer.drain()

754 def _start_read_loop(self) -> None:
755 """启动读取循环"""
756 async def read_loop():
757 try:
758 await self._read_loop()
759 except asyncio.CancelledError:

761 except Exception as e:

768 """启动保活"""
769 async def keepalive():
770 while not self._closed:
771 try:
772 await asyncio.sleep(self._config.keepalive_interval)
773 if not self._closed:
774 await self.ping()
775 except asyncio.CancelledError:
776 break
777 except Exception:
778 pass
779
780 self._keepalive_task = asyncio.create_task(keepalive())

782 async def _read_loop(self) -> None:
783 """读取循环"""
784 buffer = bytearray()
785
786 while not self._closed:
787 # 读取头部
788 header = await self._reader.readexactly(HEADER_SIZE)
789 buffer.extend(header)
790
791 # 解析头部
792 version, type_val, flags, stream_id, length = struct.unpack(">BBHII", buffer[:HEADER_SIZE])
793
794 frame_type = FrameType(type_val)
795
796 # 读取数据
797 data = b""
798 if length > 0:
799 data = await self._reader.readexactly(length)
800
801 frame = YamuxFrame(
802 version=version,
803 type=frame_type,
804 flags=flags,
805 stream_id=stream_id,
806 length=length,
807 data=data
808 )
809
810 logger.debug(f"接收帧: {frame}")
811
812 # 处理帧
813 await self._handle_frame(frame)
814
815 buffer.clear()

817 async def _handle_frame(self, frame: YamuxFrame) -> None:
818 """处理接收到的帧"""
819 # 检查版本
820 if frame.version != VERSION:
821 logger.warning(f"不支持的版本: {frame.version}")
822 await self.close(GoAwayCode.PROTOCOL_ERROR)
823 return
824
