110class ProtocolNegotiator:
111 """
112 multistream-select 协议协商器
113
114 支持两种角色:
115 - Initiator: 发起协商，提议协议
116 - Responder: 响应协商，接受或拒绝协议
117
118 同时打开场景:
119 - 双方同时发送 multistream 协议 ID
120 - 后续通过 tie-breaker 决定主导权
121 """
122

124 self,
125 timeout: float = 30.0,
126 simultaneous_open_delay: float = 0.1,

128 """
129 初始化协商器

132 timeout: 协商超时时间（秒）
133 simultaneous_open_delay: 同时打开场景下的延迟（秒）
134 """
135 self.timeout = timeout
136 self.simultaneous_open_delay = simultaneous_open_delay

138 async def negotiate(
139 self,
140 conn: StreamReaderWriter,
141 protocols: list[str],
142 ) -> str:
143 """
144 作为发起方进行协议协商
145
146 Args:
147 conn: 连接对象
148 protocols: 希望协商的协议列表（按优先级排序）
149
150 Returns:
151 协商成功的协议 ID
152
153 Raises:
154 NegotiationError: 协商失败
155 ValueError: 参数无效
156 """
157 if not protocols:
158 raise ValueError("协议列表不能为空")
159
160 # 验证所有协议 ID
161 for proto in protocols:
162 if not is_valid_protocol_id(proto):
163 raise ValueError(f"无效的协议 ID: {proto}")
164
165 tried_protocols = []
166
167 try:
168 # 步骤1: 发送 multistream 协议 ID
169 logger.debug(f"[Initiator] 发送 multistream 协议 ID")
170 await self._send_with_timeout(conn, encode_message(MULTISTREAM_PROTOCOL_ID))
171
172 # 步骤2: 接收对方的 multistream 协议 ID
173 logger.debug("[Initiator] 等待对方 multistream 协议 ID")
174 response = await self._recv_with_timeout(conn)
175 handshake_msg = decode_message(response)
176
177 if not is_multistream_protocol(handshake_msg):
178 raise HandshakeError(
179 f"期望收到 multistream 协议 ID",
180 expected=MULTISTREAM_PROTOCOL_ID,
181 actual=handshake_msg
182 )
183
184 logger.debug(f"[Initiator] 握手成功: {handshake_msg}")
185
186 # 步骤3: 尝试每个协议
187 for proto in protocols:
188 tried_protocols.append(proto)
189 logger.debug(f"[Initiator] 提议协议: {proto}")
190
191 await self._send_with_timeout(conn, encode_message(proto))
192
193 response = await self._recv_with_timeout(conn)
194 response_msg = decode_message(response)
195
196 if is_multistream_protocol(response_msg):
197 # 对方也发送了 multistream 协议 ID，说明是同时打开场景
198 # 需要重新发送协议提议
199 logger.debug("[Initiator] 检测到同时打开，重新提议协议")
200 await self._send_with_timeout(conn, encode_message(proto))
201
202 response = await self._recv_with_timeout(conn)
203 response_msg = decode_message(response)
204
205 if response_msg == proto:
206 # 对方接受该协议
207 logger.info(f"[Initiator] 协议协商成功: {proto}")
208 return proto
209 elif is_na_response(response_msg):
210 # 对方拒绝该协议，尝试下一个
211 logger.debug(f"[Initiator] 协议 {proto} 不被支持，尝试下一个")
212 continue
213 else:
214 # 未知响应
215 logger.warning(f"[Initiator] 未知响应: {response_msg}")
216 raise NegotiationError(f"未知的协商响应: {response_msg}")
217
218 # 所有协议都被拒绝
219 raise NegotiationError(
220 "没有找到共同支持的协议",
221 tried_protocols=tried_protocols
222 )
223
224 except asyncio.TimeoutError:
225 raise NegotiationError(
226 f"协议协商超时 ({self.timeout}s)",
227 tried_protocols=tried_protocols
228 )

230 async def handle_negotiate(
231 self,
232 conn: StreamReaderWriter,
233 supported: list[str],
234 protocol_handler: Optional[Callable[[str, StreamReaderWriter], Awaitable[None]]] = None,
235 ) -> str:
236 """
237 作为响应方处理协议协商
238
239 Args:
240 conn: 连接对象
241 supported: 支持的协议列表
242 protocol_handler: 协议处理函数 (协议协商成功后调用)
243
244 Returns:
245 协商成功的协议 ID
246
247 Raises:
248 NegotiationError: 协商失败
249 """
250 if not supported:
251 raise ValueError("支持的协议列表不能为空")
252
253 # 创建支持协议的集合用于快速查找
254 supported_set = set(supported)
255
256 try:
257 # 步骤1: 接收对方的 multistream 协议 ID（可选同时发送）
258 logger.debug("[Responder] 等待对方 multistream 协议 ID")
259
260 # 同时发送 multistream 协议 ID
261 send_task = asyncio.create_task(
262 self._send_with_timeout(conn, encode_message(MULTISTREAM_PROTOCOL_ID))
263 )
264 recv_task = asyncio.create_task(self._recv_with_timeout(conn))
265
266 # 等待接收和发送都完成
267 await send_task
268 response = await recv_task
269 handshake_msg = decode_message(response)
270
271 if not is_multistream_protocol(handshake_msg):
272 raise HandshakeError(
273 f"期望收到 multistream 协议 ID",
274 expected=MULTISTREAM_PROTOCOL_ID,
275 actual=handshake_msg
276 )
277
278 logger.debug(f"[Responder] 握手成功: {handshake_msg}")
279
280 # 步骤2: 持续接收协议提议并响应
281 while True:
282 # 接收协议提议
283 response = await self._recv_with_timeout(conn)
284 proposed_proto = decode_message(response)
285
286 logger.debug(f"[Responder] 收到协议提议: {proposed_proto}")
287
288 # 检查是否是 multistream 协议 ID (同时打开场景)
289 if is_multistream_protocol(proposed_proto):
290 logger.debug("[Responder] 检测到同时打开，等待实际协议提议")
291 continue
292
293 # 检查协议是否支持
294 if proposed_proto in supported_set:
295 # 支持该协议，echo 响应
296 logger.info(f"[Responder] 接受协议: {proposed_proto}")
297 await self._send_with_timeout(conn, encode_message(proposed_proto))
298
299 # 调用协议处理器（如果提供）
300 if protocol_handler:
301 await protocol_handler(proposed_proto, conn)
302
303 return proposed_proto
304 else:
305 # 不支持该协议，响应 "na"
306 logger.debug(f"[Responder] 协议 {proposed_proto} 不支持")
307 await self._send_with_timeout(conn, encode_message(NA_RESPONSE))
308
309 except asyncio.TimeoutError:
310 raise NegotiationError(f"协议协商超时 ({self.timeout}s)")

312 async def full_negotiate(
313 self,
314 conn: StreamReaderWriter,
315 protocols: list[str],
316 ) -> str:
317 """
318 完整协商流程，同时处理发起和响应
319
320 用于同时打开场景，双方同时发送协议提议。
321
322 Args:
323 conn: 连接对象
324 protocols: 我方支持的协议列表
325
326 Returns:
327 协商成功的协议 ID
328
329 Raises:
330 NegotiationError: 协商失败
331 """
332 # 同时发送 multistream 协议 ID
333 send_task = asyncio.create_task(
334 self._send_with_timeout(conn, encode_message(MULTISTREAM_PROTOCOL_ID))
335 )
336 recv_task = asyncio.create_task(self._recv_with_timeout(conn))
337
338 await send_task
339 response = await recv_task
340 handshake_msg = decode_message(response)
341
342 if not is_multistream_protocol(handshake_msg):
343 raise HandshakeError(
344 f"期望收到 multistream 协议 ID",
345 expected=MULTISTREAM_PROTOCOL_ID,
346 actual=handshake_msg
347 )
348
349 # 同时发送第一个协议提议
350 first_proto = protocols[0]
351 send_task = asyncio.create_task(
352 self._send_with_timeout(conn, encode_message(first_proto))
353 )
354 recv_task = asyncio.create_task(self._recv_with_timeout(conn))
355
356 await send_task
357 response = await recv_task
358 response_msg = decode_message(response)
359
360 if is_multistream_protocol(response_msg):
361 # 同时打开，重新发送协议提议
362 await self._send_with_timeout(conn, encode_message(first_proto))
363 response = await self._recv_with_timeout(conn)
364 response_msg = decode_message(response)
365
366 if response_msg == first_proto:
367 return first_proto
368 elif is_na_response(response_msg):
369 # 尝试下一个协议
370 for proto in protocols[1:]:
371 await self._send_with_timeout(conn, encode_message(proto))
372 response = await self._recv_with_timeout(conn)
373 response_msg = decode_message(response)
374 if response_msg == proto:
375 return proto
376
377 raise NegotiationError("没有找到共同支持的协议")

379 async def _send_with_timeout(self, conn: StreamReaderWriter, data: bytes) -> None:
380 """带超时的发送"""
381 try:
382 await asyncio.wait_for(conn.write(data), timeout=self.timeout)
383 except asyncio.TimeoutError:
384 logger.error(f"发送数据超时 ({self.timeout}s)")
385 raise

387 async def _recv_with_timeout(self, conn: StreamReaderWriter) -> bytes:
388 """带超时的接收"""
389 try:
390 data = await asyncio.wait_for(conn.read(), timeout=self.timeout)
391 if not data:
392 raise ConnectionError("连接已关闭")
393 return data
394 except asyncio.TimeoutError:
395 logger.error(f"接收数据超时 ({self.timeout}s)")
396 raise
