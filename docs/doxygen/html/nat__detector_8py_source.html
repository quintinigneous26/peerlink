23 async def detect(self) -> NATInfo:
24 """
25 检测 NAT 类型
26
27 检测流程（RFC 3489 简化版）：
28 1. 向 Server A 发送请求 → 获取映射地址 M1
29 2. 向 Server B 发送请求 → 获取映射地址 M2
30 3. 比较 M1 和 M2：
31 - 相同 → 非对称 NAT
32 - 不同 → 对称 NAT
33 """
34 logger.info("开始 NAT 类型检测...")
35
36 servers = self.stun_client.servers
37 if len(servers) < 1:
38 logger.error("没有可用的 STUN 服务器")
39 return NATInfo(type=NATType.UNKNOWN)
40
41 # Test 1: 基础连接测试
42 server_a = servers[0].split(":")
43 host_a = server_a[0]
44 port_a = int(server_a[1]) if len(server_a) > 1 else 3478
45
46 response1 = await self.stun_client.binding_request(host_a, port_a)
47
48 if not response1.success:
49 logger.error(f"STUN Test 1 失败: {response1.error}")
50 return NATInfo(type=NATType.UNKNOWN)
51
52 logger.debug(f"Test 1: {response1.mapped_ip}:{response1.mapped_port}")
53
54 # 获取本地地址
55 local_ip, local_port = self._get_local_address()
56
57 # 检测 NAT 类型
58 if self._is_public_ip(response1.mapped_ip):
59 # 公网 IP，可能是 Full Cone
60 nat_type = NATType.FULL_CONE
61 else:
62 nat_type = await self._detect_nat_type(response1, servers)
63
64 # 构建 NAT 信息
65 nat_info = NATInfo(
66 type=nat_type,
67 public_ip=response1.mapped_ip,
68 public_port=response1.mapped_port,
69 local_ip=local_ip,
70 local_port=local_port,
71 )
72
73 # 检测 CGNAT
74 nat_info.is_cgnat = self._detect_cgnat(response1)
75
76 logger.info(f"NAT 检测完成: {nat_info.type.value}, CGNAT: {nat_info.is_cgnat}")
77
78 return nat_info

80 async def _detect_nat_type(
81 self,
82 first_response: STUNResponse,
83 servers: List[str],
84 ) -> NATType:
85 """检测 NAT 类型"""
86 if len(servers) < 2:
87 return NATType.PORT_RESTRICTED
88
89 # 向第二个服务器发送请求
90 server_b = servers[1].split(":")
91 host_b = server_b[0]
92 port_b = int(server_b[1]) if len(server_b) > 1 else 3478
93
94 response2 = await self.stun_client.binding_request(host_b, port_b)
95
96 if not response2.success:
97 return NATType.UNKNOWN
98
99 # 比较映射地址
100 addr1 = (first_response.mapped_ip, first_response.mapped_port)
101 addr2 = (response2.mapped_ip, response2.mapped_port)
102
103 if addr1 != addr2:
104 # 映射地址不同 → 对称 NAT
105 return NATType.SYMMETRIC
106
107 # 映射地址相同 → 端口限制锥（最常见）
108 return NATType.PORT_RESTRICTED

119 def _is_public_ip(self, ip: str) -> bool:
120 """检查是否为公网 IP"""
121 private_prefixes = ["10.", "172.16.", "172.17.", "172.18.", "172.19.",
122 "172.20.", "172.21.", "172.22.", "172.23.", "172.24.",
123 "172.25.", "172.26.", "172.27.", "172.28.", "172.29.",
124 "172.30.", "172.31.", "192.168.", "127.", "169.254."]
125 for prefix in private_prefixes:
126 if ip.startswith(prefix):
127 return False
128 return True
