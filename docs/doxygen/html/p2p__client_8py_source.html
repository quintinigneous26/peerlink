90 @classmethod
91 def parse(cls, data: bytes) -> Optional["StunMessage"]:
92 """
93 Parse STUN message from bytes.
94
95 Args:
96 data: Raw message bytes
97
98 Returns:
99 StunMessage or None if invalid
100 """
101 if len(data) < STUN_HEADER_SIZE:
102 return None
103
104 # Parse header
105 message_type, message_length, magic_cookie = struct.unpack(
106 "!HHI", data[:8]
107 )
108
109 # Validate magic cookie
110 if magic_cookie != MAGIC_COOKIE:
111 return None
112
113 # Extract transaction ID (96 bits = 12 bytes)
114 transaction_id = data[8:20]
115
116 # Parse attributes
117 attributes = []
118 offset = STUN_HEADER_SIZE
119 remaining = message_length
120
121 while remaining > 0 and offset + 4 <= len(data):
122 attr_type, attr_length = struct.unpack("!HH", data[offset:offset + 4])
123 offset += 4
124
125 # Attributes are padded to 4-byte boundary
126 padded_length = (attr_length + 3) & ~3
127
128 if offset + padded_length > len(data):
129 break
130
131 attr_value = data[offset:offset + attr_length]
132 attributes.append(StunAttribute(type=attr_type, value=attr_value))
133
134 offset += padded_length
135 remaining -= (4 + padded_length)
136
137 return cls(
138 message_type=message_type,
139 message_length=message_length,
140 magic_cookie=magic_cookie,
141 transaction_id=transaction_id,
142 attributes=attributes,
143 )

145 def get_attribute(self, attr_type: int) -> Optional[StunAttribute]:
146 """Get attribute by type."""
147 for attr in self.attributes:
148 if attr.type == attr_type:
149 return attr
150 return None

164def parse_xor_mapped_address(data: bytes, transaction_id: bytes) -> Optional[Tuple[str, int]]:
165 """
166 Parse XOR-MAPPED-ADDRESS attribute (RFC 5389 Section 15.2).
167
168 The address is XORed with the magic cookie (for IPv4) or
169 magic cookie + transaction ID (for IPv6).
170
171 Args:
172 data: Attribute value bytes (after type+length header)
173 transaction_id: 12-byte transaction ID from STUN header
174
175 Returns:
176 Tuple of (ip_address, port) or None if parsing fails
177 """
178 if len(data) < 4:
179 return None
180
181 # First byte is reserved (should be 0)
182 # Second byte is address family
183 reserved = data[0]
184 family = data[1]
185
186 # Port is XORed with magic cookie >> 16
187 xor_port = struct.unpack("!H", data[2:4])[0]
188 port = xor_port ^ ((MAGIC_COOKIE >> 16) & 0xFFFF)
189
190 if family == 0x01: # IPv4
191 if len(data) < 8:
192 return None
193
194 # IP address is XORed with magic cookie
195 xor_ip_bytes = data[4:8]
196 cookie_bytes = struct.pack("!I", MAGIC_COOKIE)
197 ip_bytes = bytes(xor_ip_bytes[i] ^ cookie_bytes[i] for i in range(4))
198 ip = ".".join(str(b) for b in ip_bytes)
199
200 return (ip, port)
201
202 elif family == 0x02: # IPv6
203 if len(data) < 20:
204 return None
205
206 # IP address is XORed with magic cookie + transaction ID
207 xor_ip_bytes = data[4:20]
208 cookie_bytes = struct.pack("!I", MAGIC_COOKIE)
209 xor_key = cookie_bytes + transaction_id
210
211 if len(xor_key) != 16:
212 return None
213
214 ip_bytes = bytes(xor_ip_bytes[i] ^ xor_key[i] for i in range(16))
215
216 # Convert to IPv6 string format
217 hex_parts = []
218 for i in range(0, 16, 2):
219 hex_parts.append(f"{ip_bytes[i]:02x}{ip_bytes[i+1]:02x}")
220 ip = ":".join(hex_parts)
221
222 return (ip, port)
223
224 return None

228 """
229 Parse MAPPED-ADDRESS attribute (non-XOR, RFC 5389 Section 15.1).
230
231 Args:
232 data: Attribute value bytes
233
234 Returns:
235 Tuple of (ip_address, port) or None

238 return None

246 return None
247 ip_bytes = data[4:8]
248 ip = ".".join(str(b) for b in ip_bytes)
249 return (ip, port)
250

253 return None

257 hex_parts.append(f"{ip_bytes[i]:02x}{ip_bytes[i+1]:02x}")
258 ip = ":".join(hex_parts)
259 return (ip, port)

261 return None

264def verify_fingerprint(message: bytes, fingerprint_value: bytes) -> bool:
265 """
266 Verify STUN Fingerprint attribute (RFC 5389 Section 15.5).
267
268 The fingerprint is CRC-32 of the message XORed with 0x5354554e.
269
270 Args:
271 message: Complete STUN message bytes (including fingerprint attribute)
272 fingerprint_value: The fingerprint value from the attribute
273
274 Returns:
275 True if fingerprint is valid
276 """
277 import binascii
278
279 # Fingerprint covers entire message except the fingerprint attribute itself
280 # The attribute header (4 bytes) + value (4 bytes) = 8 bytes at the end
281 if len(message) < 8:
282 return False
283
284 # Message without fingerprint attribute
285 msg_for_crc = message[:-8]
286
287 # Calculate CRC-32
288 crc = binascii.crc32(msg_for_crc) & 0xFFFFFFFF
289
290 # XOR with magic value
291 expected_fingerprint = crc ^ 0x5354554e
292
293 # Compare
294 actual_fingerprint = struct.unpack("!I", fingerprint_value)[0]
295
296 return expected_fingerprint == actual_fingerprint

390 def _build_stun_request(self) -> Tuple[bytes, bytes]:
391 """
392 Build STUN Binding Request.
393
394 Returns:
395 Tuple of (request_bytes, transaction_id)
396 """
397 # Generate random transaction ID (96 bits)
398 transaction_id = secrets.token_bytes(12)
399
400 # STUN header:
401 # - Message Type: 2 bytes (0x0001 = Binding Request)
402 # - Message Length: 2 bytes (0 for request without attributes)
403 # - Magic Cookie: 4 bytes (0x2112A442)
404 # - Transaction ID: 12 bytes
405 header = struct.pack(
406 "!HHI12s",
407 MessageType.BINDING_REQUEST,
408 0, # No attributes
409 MAGIC_COOKIE,
410 transaction_id
411 )
412
413 return header, transaction_id

415 def _parse_stun_response(self, data: bytes, transaction_id: bytes) -> Optional[Tuple[str, int]]:
416 """
417 Parse STUN Binding Response, extract public address.
418
419 Args:
420 data: Raw response bytes
421 transaction_id: Expected transaction ID
422
423 Returns:
424 Tuple of (ip, port) or None if parsing fails
425 """
426 # Parse STUN message
427 message = StunMessage.parse(data)
428
429 if message is None:
430 print("Failed to parse STUN message")
431 return None
432
433 # Validate message type
434 if message.message_type != MessageType.BINDING_RESPONSE:
435 print(f"Unexpected message type: 0x{message.message_type:04x}")
436 return None
437
438 # Validate magic cookie
439 if message.magic_cookie != MAGIC_COOKIE:
440 print(f"Invalid magic cookie: 0x{message.magic_cookie:08x}")
441 return None
442
443 # Validate transaction ID
444 if message.transaction_id != transaction_id:
445 print("Transaction ID mismatch")
446 return None
447
448 # Try XOR-MAPPED-ADDRESS first (preferred)
449 xor_attr = message.get_attribute(AttributeType.XOR_MAPPED_ADDRESS)
450 if xor_attr:
451 result = parse_xor_mapped_address(xor_attr.value, transaction_id)
452 if result:
453 return result
454
455 # Fallback to MAPPED-ADDRESS
456 mapped_attr = message.get_attribute(AttributeType.MAPPED_ADDRESS)
457 if mapped_attr:
458 result = parse_mapped_address(mapped_attr.value)
459 if result:
460 return result
461
462 print("No valid address attribute found in STUN response")
463 return None
