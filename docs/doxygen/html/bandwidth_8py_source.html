35class TokenBucket:
36 """
37 Token bucket rate limiter.
38
39 Allows bursts up to bucket capacity while maintaining average rate.
40 """
41
42 def __init__(self, rate: int, capacity: int):
43 """
44 Initialize token bucket.
45
46 Args:
47 rate: Tokens per second (bytes per second)
48 capacity: Maximum bucket size (burst allowance)
49 """
50 self.rate = rate
51 self.capacity = capacity
52 self.tokens = float(capacity)
53 self.last_update = time.time()
54 self._lock = asyncio.Lock()
55
56 async def consume(self, tokens: int) -> bool:
57 """
58 Try to consume tokens.
59
60 Args:
61 tokens: Number of tokens to consume
62
63 Returns:
64 True if tokens were consumed, False if insufficient tokens
65 """
66 async with self._lock:
67 now = time.time()
68 elapsed = now - self.last_update
69
70 # Add new tokens based on elapsed time
71 self.tokens += elapsed * self.rate
72 self.tokens = min(self.tokens, self.capacity)
73 self.last_update = now
74
75 if self.tokens >= tokens:
76 self.tokens -= tokens
77 return True
78 return False
79
80 async def wait_for_tokens(self, tokens: int) -> bool:
81 """
82 Wait until enough tokens are available.
83
84 Args:
85 tokens: Number of tokens needed
86
87 Returns:
88 True if tokens acquired, False if rate is zero
89 """
90 if self.rate <= 0:
91 return False
92
93 while not await self.consume(tokens):
94 # Calculate wait time
95 async with self._lock:
96 deficit = tokens - self.tokens
97 wait_time = deficit / self.rate
98 self.last_update = time.time()
99
100 await asyncio.sleep(min(wait_time, 1.0))
101
102 return True
103
104 def get_available_tokens(self) -> int:
105 """Get currently available tokens."""
106 return int(self.tokens)

139 async def throttle_read(self, allocation_id: str, size: int) -> bool:
140 """
141 Throttle read operation.
142
143 Args:
144 allocation_id: Allocation ID
145 size: Number of bytes to read
146
147 Returns:
148 True if read allowed, False if rate limited
149 """
150 # Check global limit
151 if not await self._global_read_bucket.consume(size):
152 return False
153
154 # Check per-allocation limit
155 limiters = self._limiters.setdefault(allocation_id, {})
156 if "read" not in limiters:
157 limiters["read"] = TokenBucket(
158 rate=self.default_limit.read_bps,
159 capacity=self.default_limit.burst_bps,
160 )
161
162 if await limiters["read"].consume(size):
163 stats = self._stats[allocation_id]
164 stats.bytes_read += size
165 return True
166
167 return False

254class ThroughputMonitor:
255 """
256 Monitor throughput rates.
257
258 Tracks bytes transferred and calculates current rates.
259 """
260
261 def __init__(self, window_seconds: int = 5):
262 """
263 Initialize throughput monitor.
264
265 Args:
266 window_seconds: Time window for rate calculation
267 """
268 self.window = window_seconds
269 self._read_samples: list = []
270 self._write_samples: list = []
271 self._lock = asyncio.Lock()
272
273 async def record_read(self, bytes_count: int):
274 """
275 Record read operation.
276
277 Args:
278 bytes_count: Number of bytes read
279 """
280 async with self._lock:
281 self._read_samples.append((time.time(), bytes_count))
282 await self._cleanup_old_samples()
283
284 async def record_write(self, bytes_count: int):
285 """
286 Record write operation.
287
288 Args:
289 bytes_count: Number of bytes written
290 """
291 async with self._lock:
292 self._write_samples.append((time.time(), bytes_count))
293 await self._cleanup_old_samples()
294
295 async def _cleanup_old_samples(self):
296 """Remove samples older than window."""
297 cutoff = time.time() - self.window
298
299 self._read_samples = [(t, b) for t, b in self._read_samples if t > cutoff]
300 self._write_samples = [(t, b) for t, b in self._write_samples if t > cutoff]
301
302 async def get_read_rate(self) -> float:
303 """
304 Get current read rate (bytes/second).
305
306 Returns:
307 Read rate
308 """
309 async with self._lock:
310 await self._cleanup_old_samples()
311 if not self._read_samples:
312 return 0.0
313
314 total = sum(b for _, b in self._read_samples)
315 return total / self.window
316
317 async def get_write_rate(self) -> float:
318 """
319 Get current write rate (bytes/second).
320
321 Returns:
322 Write rate
323 """
324 async with self._lock:
325 await self._cleanup_old_samples()
326 if not self._write_samples:
327 return 0.0
328
329 total = sum(b for _, b in self._write_samples)
330 return total / self.window
331
332 async def get_stats(self) -> Dict[str, float]:
333 """
334 Get throughput statistics.
335
336 Returns:
337 Dictionary with stats
338 """
339 read_rate = await self.get_read_rate()
340 write_rate = await self.get_write_rate()
341
342 return {
343 "read_rate_bps": read_rate,
344 "write_rate_bps": write_rate,
345 "read_rate_mbps": read_rate / 1_000_000,
346 "write_rate_mbps": write_rate / 1_000_000,
347 }
