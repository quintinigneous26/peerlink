69 def add_transport(self, transport: Transport, priority: int = 0) -> None:
70 """
71 添加传输协议
72
73 Args:
74 transport: 传输实例
75 priority: 优先级（数字越小优先级越高）
76 """
77 for protocol in transport.protocols():
78 key = f"{priority}:{protocol}"
79 self._transports[key] = transport
80
81 # 更新优先级顺序
82 self._prefer_order = sorted(self._transports.keys())
83
84 logger.debug(
85 f"Added transport {transport.protocols()} "
86 f"with priority {priority}"
87 )
88
89 def remove_transport(self, protocol: str) -> None:
90 """移除传输协议"""
91 keys_to_remove = [k for k in self._transports if k.endswith(f":{protocol}")]
92 for key in keys_to_remove:
93 del self._transports[key]
94
95 self._prefer_order = sorted(self._transports.keys())
96 logger.debug(f"Removed transport {protocol}")
97
98 def get_transport(self, protocol: str) -> Optional[Transport]:
99 """获取指定协议的传输"""
100 for key in self._prefer_order:
101 if key.endswith(f":{protocol}"):
102 return self._transports[key]
103 return None
104
105 def list_transports(self) -> List[str]:
106 """列出所有已注册的传输协议"""
107 protocols = set()
108 for key in self._transports:
109 _, protocol = key.split(":", 1)
110 protocols.add(protocol)
111 return sorted(protocols)

113 async def listen(
114 self,
115 addr: str,
116 protocol: Optional[str] = None
117 ) -> Listener:
118 """
119 开始监听
120
121 Args:
122 addr: 监听地址
123 protocol: 指定使用的传输协议，None表示自动选择
124
125 Returns:
126 监听器
127
128 Raises:
129 ListenError: 监听失败
130 """
131 if self._closed:
132 raise ListenError("Transport manager is closed")
133
134 transport = self._select_transport(protocol)
135 if not transport:
136 raise ListenError(f"No transport available for protocol: {protocol}")
137
138 try:
139 listener = await transport.listen(addr)
140 self._listeners[addr] = listener
141 logger.info(f"Listening on {addr} via {transport.protocols()}")
142 return listener
143 except Exception as e:
144 raise ListenError(f"Listen failed: {e}") from e

146 async def dial(
147 self,
148 addr: str,
149 peer_id: str,
150 protocol: Optional[str] = None
151 ) -> UpgradedConnection:
152 """
153 建立连接
154
155 Args:
156 addr: 目标地址
157 peer_id: 目标对等节点ID
158 protocol: 指定使用的传输协议，None表示按优先级自动选择
159
160 Returns:
161 升级后的连接
162
163 Raises:
164 DialError: 连接失败
165 """
166 if self._closed:
167 raise DialError("Transport manager is closed")
168
169 if not self._upgrader:
170 raise DialError("No upgrader configured")
171
172 # 尝试按优先级拨号
173 last_error = None
174 tried_protocols = []
175
176 for key in self._prefer_order:
177 if protocol and not key.endswith(f":{protocol}"):
178 continue
179
180 transport = self._transports[key]
181 _, proto = key.split(":", 1)
182 tried_protocols.append(proto)
183
184 try:
185 logger.debug(f"Dialing {addr} via {proto}")
186 conn = await asyncio.wait_for(
187 transport.dial(addr),
188 timeout=self._dial_timeout
189 )
190
191 # 升级连接
192 upgraded = await self._upgrader.upgrade_outbound(conn, peer_id)
193 logger.info(f"Connected to {peer_id} via {proto}")
194 return upgraded
195
196 except asyncio.TimeoutError:
197 logger.debug(f"Dial timeout via {proto}")
198 last_error = DialError(f"Dial timeout via {proto}")
199 except Exception as e:
200 logger.debug(f"Dial failed via {proto}: {e}")
201 last_error = DialError(f"Dial failed: {e}")
202
203 raise DialError(
204 f"All dial attempts failed. Tried: {tried_protocols}. "
205 f"Last error: {last_error}"
206 )

208 async def accept(self) -> tuple[str, UpgradedConnection]:
209 """
210 接受传入连接
211
212 Returns:
213 (peer_id, upgraded_connection) 元组
214
215 Raises:
216 TransportError: 接受失败
217 """
218 if not self._listeners:
219 raise TransportError("No active listeners")
220
221 if not self._upgrader:
222 raise TransportError("No upgrader configured")
223
224 # 等待任意监听器有连接
225 tasks = []
226 for addr, listener in self._listeners.items():
227 task = asyncio.create_task(self._accept_from_listener(listener, addr))
228 tasks.append(task)
229
230 done, pending = await asyncio.wait(
231 tasks,
232 return_when=asyncio.FIRST_COMPLETED
233 )
234
235 # 取消其他任务
236 for task in pending:
237 task.cancel()
238
239 for task in done:
240 return await task
241
242 raise TransportError("Accept failed")

244 async def _accept_from_listener(
245 self,
246 listener: Listener,
247 addr: str
248 ) -> tuple[str, UpgradedConnection]:
249 """从指定监听器接受连接"""
250 try:
251 conn = await listener.accept()
252
253 # 升级连接
254 upgraded = await self._upgrader.upgrade_inbound(conn)
255
256 logger.info(
257 f"Accepted connection from {upgraded.peer_id} "
258 f"via {upgraded.security_protocol}"
259 )
260 return (upgraded.peer_id, upgraded)
261
262 except Exception as e:
263 logger.error(f"Accept error on {addr}: {e}")
264 raise

270 async def close(self) -> None:
271 """关闭传输管理器"""
272 if self._closed:
273 return
274
275 self._closed = True
276
277 # 关闭所有监听器
278 close_tasks = []
279 for addr, listener in self._listeners.items():
280 close_tasks.append(listener.close())
281
282 if close_tasks:
283 await asyncio.gather(*close_tasks, return_exceptions=True)
284
285 self._listeners.clear()
286
287 # 关闭所有传输
288 close_tasks = []
289 for transport in set(self._transports.values()):
290 close_tasks.append(transport.close())
291
292 if close_tasks:
293 await asyncio.gather(*close_tasks, return_exceptions=True)
294
295 self._transports.clear()
296
297 logger.info("Transport manager closed")

299 def _select_transport(self, protocol: Optional[str]) -> Optional[Transport]:
300 """选择传输"""
301 if protocol:
302 return self.get_transport(protocol)
303
304 # 返回优先级最高的传输
305 if self._prefer_order:
306 return self._transports[self._prefer_order[0]]
307
308 return None

311class TransportBuilder:

318 def __init__(self, local_peer_id: str):
319 self._local_peer_id = local_peer_id
320 self._transports: List[tuple[Transport, int]] = []
321 self._security_transports: List[Any] = []
322 self._muxers: List[Any] = []
323
324 def add_transport(self, transport: Transport, priority: int = 0) -> "TransportBuilder":
325 """添加传输"""
326 self._transports.append((transport, priority))
327 return self

329 def add_security(self, security_transport: Any) -> "TransportBuilder":
330 """添加安全传输"""
331 self._security_transports.append(security_transport)

334 def add_muxer(self, muxer: Any) -> "TransportBuilder":
335 """添加流复用器"""
336 self._muxers.append(muxer)
337 return self

339 def build(self) -> TransportManager:
340 """构建传输管理器"""
341 # 创建升级器
342 upgrader = TransportUpgrader(
343 security_transports=self._security_transports,
344 muxers=self._muxers
345 )
346
347 # 创建管理器
348 manager = TransportManager(self._local_peer_id, upgrader)
349
350 # 添加传输
351 for transport, priority in self._transports:
352 manager.add_transport(transport, priority)
353
354 return manager

357def create_transport_manager(
358 local_peer_id: str,
359 transports: Optional[List[Transport]] = None,
360 security_transports: Optional[List[Any]] = None,
361 muxers: Optional[List[Any]] = None

363 """
