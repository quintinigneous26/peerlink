64 """信令消息类型"""
65 OFFER = "offer"

71@dataclass
72class SignalingMessage:
73 """信令消息"""
74 type: SignalingMessageType
75 data: Dict[str, Any]
76 peer_id: str = ""
77
78 def to_json(self) -> str:
79 """转换为 JSON"""
80 return json.dumps({
81 "type": self.type.value,
82 "data": self.data,
83 "peer_id": self.peer_id,

86 @classmethod
87 def from_json(cls, data: str) -> 'SignalingMessage':
88 """从 JSON 解析"""
89 obj = json.loads(data)

92 data=obj["data"],

175 async def read(self, n: int = -1) -> bytes:
176 """
177 读取数据
178
179 Args:
180 n: 读取字节数，-1 表示读取所有
181
182 Returns:
183 读取的数据
184 """
185 if self._closed:
186 raise ConnectionError("连接已关闭")

188 try:
189 data = await asyncio.wait_for(
190 self._read_queue.get(),
191 timeout=30.0

196 return data[:n]

202 """
203 写入数据
204

206 data: 要写入的数据

209 写入的字节数

212 raise ConnectionError("连接已关闭")

321 def add_ice_candidate(self, candidate_dict: Dict[str, Any]) -> None:
322 """
323 添加 ICE 候选
324
325 Args:
326 candidate_dict: ICE 候选字典
327 """

329 candidate = RTCIceCandidate(

340 def _setup_callbacks(self) -> None:
341 """设置回调函数"""
342 # ICE 候选回调
343 @self._pc.on("icecandidate")
344 def on_ice_candidate(candidate):
345 if candidate:
346 candidate_dict = {
347 "sdpMid": candidate.sdpMid,
348 "sdpMLineIndex": candidate.sdpMLineIndex,
349 "candidate": candidate.candidate,
350 }
351 self._ice_candidates.append(candidate_dict)
352 logger.debug(f"ICE 候选: {candidate.candidate[:50]}...")
353 else:
354 # ICE 收集完成
355 self._ice_gathering_complete.set()
356 logger.debug("ICE 候选收集完成")
357
358 # ICE 连接状态回调
359 @self._pc.on("iceconnectionstatechange")
360 def on_ice_connection_state_change():
361 self._ice_connection_state = self._pc.iceConnectionState
362 logger.debug(f"ICE 连接状态: {self._ice_connection_state}")
363
364 if self._ice_connection_state in ["failed", "disconnected"]:
365 # 连接失败或断开
366 if not self._closed:
367 asyncio.create_task(self.close())
368
369 # 连接状态回调
370 @self._pc.on("connectionstatechange")
371 def on_connection_state_change():
372 self._connection_state = self._pc.connectionState
373 logger.debug(f"连接状态: {self._connection_state}")
374
375 # DataChannel 回调（应答方）
376 if not self._is_initiator:
377 @self._pc.on("datachannel")
378 def on_datachannel(channel):
379 self._data_channel = channel
380 self._setup_data_channel_callbacks(channel)
381
382 def _setup_data_channel_callbacks(self, channel: 'DataChannel') -> None:
383 """设置 DataChannel 回调"""
384 @channel.on("open")
385 def on_datachannel_open():
386 logger.debug("DataChannel 已打开")
387
388 @channel.on("message")
389 def on_datachannel_message(message):
390 if isinstance(message, str):
391 message = message.encode("utf-8")
392 if not self._closed:
393 asyncio.create_task(self._read_queue.put(message))
394
395 @channel.on("close")
396 def on_datachannel_close():
397 logger.debug("DataChannel 已关闭")
398 if not self._closed:
399 asyncio.create_task(self.close())
400
401 @channel.on("error")
402 def on_datachannel_error(error):
403 logger.error(f"DataChannel 错误: {error}")
404 if not self._closed:
405 asyncio.create_task(self.close())
406
407 async def _wait_for_ice_gathering(self, timeout: float = 10.0) -> None:
408 """等待 ICE 收集完成"""
409 try:
410 await asyncio.wait_for(
411 self._ice_gathering_complete.wait(),
412 timeout=timeout
413 )
414 except asyncio.TimeoutError:
415 logger.warning("ICE 收集超时")

420class WebRTCListener(Listener):
421 """
422 WebRTC 监听器
423
424 监听传入的 WebRTC 连接请求。
425 通过信令服务器接收 SDP Offer 并创建 Answer。
426 """
427
428 def __init__(
429 self,
430 transport: 'WebRTCTransport',
431 local_address: str,
432 ):
433 self._transport = transport
434 self._local_address = local_address
435 self._closed = False
436
437 # 接受队列
438 self._accept_queue: asyncio.Queue[WebRTCConnection] = asyncio.Queue()
439
440 # 待处理的连接
441 self._pending_connections: Dict[str, WebRTCConnection] = {}
442
443 @property
444 def transport(self) -> 'WebRTCTransport':
445 """获取关联的传输"""
446 return self._transport
447
448 async def accept(self) -> WebRTCConnection:
449 """
450 接受传入连接
451
452 Returns:
453 新接受的连接
454 """
455 if self._closed:
456 raise ListenerError("监听器已关闭")
457
458 return await self._accept_queue.get()
459
460 async def close(self) -> None:
461 """关闭监听器"""
462 if self._closed:
463 return

481 except Exception:
482 pass

485

488 return self._closed
489
490 @property
491 def addresses(self) -> List[tuple]:
492 """获取监听地址列表"""
493 # WebRTC 使用 ICE 候选作为地址

495 return [("webrtc", self._local_address)]

497 async def handle_offer(self, peer_id: str, offer_sdp: str) -> str:
498 """
499 处理传入的 Offer
500
501 Args:
502 peer_id: 对端 ID
503 offer_sdp: Offer SDP
504
505 Returns:
506 Answer SDP
507 """
508 if self._closed:
509 raise ListenerError("监听器已关闭")
510
511 # 创建连接
512 conn = await self._transport._create_connection(is_initiator=False)
513
514 # 设置远程描述
515 from aiortc import RTCSessionDescription
516 offer = RTCSessionDescription(sdp=offer_sdp, type="offer")
517 await conn.set_remote_description(offer)
518
519 # 创建 Answer
520 answer = await conn.create_answer()
521
522 # 添加到待处理连接
523 self._pending_connections[peer_id] = conn
524
525 return answer.sdp
526
527 async def finalize_connection(self, peer_id: str) -> WebRTCConnection:
528 """
529 完成连接建立
530
531 Args:
532 peer_id: 对端 ID
533
534 Returns:
535 建立的连接
536 """
537 conn = self._pending_connections.pop(peer_id, None)
538 if not conn:
539 raise ListenerError(f"未找到待处理的连接: {peer_id}")
540
541 await self._accept_queue.put(conn)
542 return conn

689 async def handle_signaling_message(self, message: SignalingMessage) -> None:
690 """
691 处理信令消息
692
693 Args:
694 message: 信令消息
695 """
696 if message.type == SignalingMessageType.OFFER:
697 await self._handle_offer(message)
698 elif message.type == SignalingMessageType.ANSWER:
699 await self._handle_answer(message)
700 elif message.type == SignalingMessageType.ICE_CANDIDATE:
701 await self._handle_ice_candidate(message)
702 else:
703 logger.warning(f"未知的信令消息类型: {message.type}")

705 def _create_rtc_config(self) -> 'RTCConfiguration':
706 """创建 RTC 配置"""
707 ice_servers = []
708
709 # 添加 STUN 服务器
710 for stun_url in self._stun_servers:
711 ice_servers.append({"urls": [stun_url]})
712
713 # 添加自定义 ICE 服务器
714 for server in self._ice_servers:
715 ice_servers.append({
716 "urls": server.urls,
717 "username": server.username,
718 "credential": server.credential,
719 })
720
721 return RTCConfiguration(iceServers=ice_servers)

723 async def _create_connection(self, is_initiator: bool) -> WebRTCConnection:
724 """创建 WebRTC 连接"""
725 # 创建 PeerConnection
726 pc = RTCPeerConnection(configuration=self._rtc_config)
727
728 # 创建 DataChannel（发起方）
729 data_channel = None
730 if is_initiator:
731 data_channel = pc.createDataChannel(
732 DEFAULT_DATA_CHANNEL_LABEL,
733 ordered=True,
734 )
735 # 设置 DataChannel 回调
736 # 注意：回调需要在连接对象中设置
737
738 # 创建连接对象
739 conn = WebRTCConnection(
740 peer_connection=pc,
741 data_channel=data_channel,
742 is_initiator=is_initiator,
743 )
744
745 # 设置 DataChannel 回调（发起方）
746 if data_channel:
747 conn._setup_data_channel_callbacks(data_channel)
748
749 self._connections.append(conn)
750 return conn

752 async def _send_signaling_message(self, message: SignalingMessage) -> None:
753 """发送信令消息"""
754 if self._on_signaling_message:
755 await self._on_signaling_message(message)
756 else:
757 logger.warning("未设置信令回调，消息未发送")

759 async def _handle_offer(self, message: SignalingMessage) -> None:
760 """处理 Offer 消息"""
761 offer_sdp = message.data.get("sdp")
762 if not offer_sdp:
763 logger.error("Offer 消息缺少 SDP")
764 return
765
766 # 创建 Answer
767 from aiortc import RTCSessionDescription
768
769 # 创建连接
770 conn = await self._create_connection(is_initiator=False)
771
772 # 设置远程描述
773 offer = RTCSessionDescription(sdp=offer_sdp, type="offer")
774 await conn.set_remote_description(offer)
775
776 # 创建 Answer
777 answer = await conn.create_answer()
778
779 # 发送 Answer
780 await self._send_signaling_message(SignalingMessage(
781 type=SignalingMessageType.ANSWER,
782 data={
783 "sdp": answer.sdp,
784 "peer_id": message.peer_id,
785 },
786 ))
787
788 async def _handle_answer(self, message: SignalingMessage) -> None:
789 """处理 Answer 消息"""
790 answer_sdp = message.data.get("sdp")
791 if not answer_sdp:
792 logger.error("Answer 消息缺少 SDP")
793 return
794
795 from aiortc import RTCSessionDescription
796 answer = RTCSessionDescription(sdp=answer_sdp, type="answer")
797
798 # 找到对应的连接并设置远程描述
799 # 这里简化处理，实际应该根据 peer_id 匹配
800 for conn in self._connections:
801 if conn.is_initiator and not conn.get_remote_description():
802 await conn.set_remote_description(answer)
803 break
804
805 async def _handle_ice_candidate(self, message: SignalingMessage) -> None:
806 """处理 ICE 候选消息"""
