18class FallbackReason(Enum):
19 """降级原因"""
20 SYMMETRIC_SYMMETRIC = "symmetric_symmetric" # 双对称 NAT
21 SYMMETRIC_TIMEOUT = "symmetric_timeout" # 对称 NAT 超时
22 MOBILE_CGNAT = "mobile_cgnat" # 移动 CGNAT

56 def decide(
57 self,
58 local_nat: NATInfo,
59 peer_nat: NATInfo,
60 local_isp: ISP,
61 peer_isp: ISP,
62 punch_duration_ms: int = 0,
63 ) -> FallbackDecision:
64 """
65 决定是否需要降级
66
67 决策优先级：
68 1. 双对称 NAT → 立即降级（0ms）
69 2. 移动 CGNAT + 对称 → 快速降级（<1s）
70 3. 跨运营商失败多次 → 降级
71 4. 跨境超时多次 → 降级
72 5. 单次打孔超时 → 重试，3次后降级
73 """
74
75 # 规则 1: 双对称 NAT
76 if local_nat.is_symmetric() and peer_nat.is_symmetric():
77 logger.info("降级决策: 双对称 NAT")
78 return FallbackDecision(
79 should_fallback=True,
80 reason=FallbackReason.SYMMETRIC_SYMMETRIC,
81 delay_ms=0,
82 description="双方都是对称 NAT，无法直接穿透",
83 )
84
85 # 规则 2: 移动 CGNAT + 对称 NAT
86 local_profile = get_isp_profile(local_isp)
87 peer_profile = get_isp_profile(peer_isp)
88
89 if local_nat.is_cgnat and local_nat.is_symmetric():
90 if local_isp == ISP.CHINA_MOBILE or local_isp == ISP.CHINA_RAILCOM:
91 logger.info("降级决策: 移动 CGNAT + 对称 NAT")
92 return FallbackDecision(
93 should_fallback=True,
94 reason=FallbackReason.MOBILE_CGNAT,
95 delay_ms=1000, # 1秒后降级
96 description="移动 CGNAT + 对称 NAT，快速降级",
97 )
98
99 if peer_nat.is_cgnat and peer_nat.is_symmetric():
100 if peer_isp == ISP.CHINA_MOBILE or peer_isp == ISP.CHINA_RAILCOM:
101 logger.info("降级决策: 对端移动 CGNAT + 对称 NAT")
102 return FallbackDecision(
103 should_fallback=True,
104 reason=FallbackReason.MOBILE_CGNAT,
105 delay_ms=1000,
106 description="对端移动 CGNAT + 对称 NAT",
107 )
108
109 # 规则 3: 跨运营商失败
110 if is_cross_isp(local_isp, peer_isp):
111 if self._punch_fail_count >= 3:
112 logger.info(f"降级决策: 跨运营商失败 {self._punch_fail_count} 次")
113 return FallbackDecision(
114 should_fallback=True,
115 reason=FallbackReason.CROSS_ISP_FAILED,
116 delay_ms=0,
117 description=f"跨运营商打孔失败 {self._punch_fail_count} 次",
118 )
119
120 # 规则 4: 跨境超时
121 if is_cross_border(local_isp, peer_isp):
122 if punch_duration_ms > 10000: # 超过 10 秒
123 logger.info(f"降级决策: 跨境超时 {punch_duration_ms}ms")
124 return FallbackDecision(
125 should_fallback=True,
126 reason=FallbackReason.CROSS_BORDER_TIMEOUT,
127 delay_ms=0,
128 description=f"跨境打孔超时 {punch_duration_ms}ms",
129 )
130
131 # 规则 5: 打孔超时（累计）
132 if self._punch_fail_count >= 3:
133 logger.info(f"降级决策: 打孔失败 {self._punch_fail_count} 次")
134 return FallbackDecision(
135 should_fallback=True,
136 reason=FallbackReason.PUNCH_FAIL_COUNT,
137 delay_ms=0,
138 description=f"打孔累计失败 {self._punch_fail_count} 次",
139 )
140
141 # 规则 6: 对称 NAT 超时（缩短等待）
142 if local_nat.is_symmetric() or peer_nat.is_symmetric():
143 if punch_duration_ms > 2000: # 对称 NAT 只等 2 秒
144 logger.info(f"降级决策: 对称 NAT 超时 {punch_duration_ms}ms")
145 return FallbackDecision(
146 should_fallback=True,
147 reason=FallbackReason.SYMMETRIC_TIMEOUT,
148 delay_ms=0,
149 description=f"对称 NAT 打孔超时 {punch_duration_ms}ms",
150 )
151
152 # 不需要降级
153 return FallbackDecision(
154 should_fallback=False,
155 description="继续尝试打孔",
156 )
