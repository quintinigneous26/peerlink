51class CachedSessionTicket:
52 """
53 Cached QUIC session ticket for 0-RTT.
54
55 Attributes:
56 ticket_data: Raw ticket data
57 server_name: Server hostname
58 transport_params: Transport parameters
59 created_at: Creation timestamp
60 expires_at: Expiration timestamp
61 last_used: Last usage timestamp
62 usage_count: Number of times used
63 status: Ticket status
64 """
65 ticket_data: bytes
66 server_name: str
67 transport_params: Dict[str, Any]
68 created_at: float = field(default_factory=time.time)

71 usage_count: int = 0

106class SessionTicketStore:
107 """
108 Storage for QUIC session tickets.
109
110 Implements LRU eviction and automatic cleanup.
111 """
112
113 def __init__(self, max_tickets: int = MAX_SESSION_TICKETS):
114 self._max_tickets = max_tickets
115 self._tickets: Dict[str, CachedSessionTicket] = {}
116 self._lock = asyncio.Lock()
117 self._cleanup_task: Optional[asyncio.Task] = None
118
119 def start(self) -> None:
120 """Start background cleanup task."""
121 if not self._cleanup_task:
122 self._cleanup_task = asyncio.create_task(self._cleanup_loop())
123
124 async def stop(self) -> None:
125 """Stop cleanup task."""
126 if self._cleanup_task:
127 self._cleanup_task.cancel()
128 try:
129 await self._cleanup_task
130 except asyncio.CancelledError:
131 pass

133 async def get_ticket(
134 self,
135 server_name: str,
136 ) -> Optional[CachedSessionTicket]:
137 """
138 Get a valid ticket for the server.
139
140 Args:
141 server_name: Server hostname
142
143 Returns:
144 CachedSessionTicket if available, None otherwise
145 """
146 async with self._lock:
147 ticket = self._tickets.get(server_name)
148
149 if ticket and ticket.is_valid:
150 ticket.mark_used()
151 return ticket
152
153 # Remove invalid ticket
154 if ticket:
155 del self._tickets[server_name]
156
157 return None
158
159 async def store_ticket(
160 self,
161 server_name: str,
162 ticket_data: bytes,
163 transport_params: Dict[str, Any],
164 ) -> CachedSessionTicket:
165 """
166 Store a new session ticket.
167
168 Args:
169 server_name: Server hostname
170 ticket_data: Raw ticket data
171 transport_params: QUIC transport parameters
172
173 Returns:
174 CachedSessionTicket instance
175 """
176 async with self._lock:
177 # Evict oldest if at capacity
178 if len(self._tickets) >= self._max_tickets:
179 oldest_key = min(
180 self._tickets.keys(),
181 key=lambda k: self._tickets[k].last_used
182 )
183 del self._tickets[oldest_key]
184
185 ticket = CachedSessionTicket(
186 ticket_data=ticket_data,
187 server_name=server_name,
188 transport_params=transport_params,
189 )
190
191 self._tickets[server_name] = ticket
192 logger.debug(f"Stored session ticket for {server_name}")
193
194 return ticket
195
196 async def revoke_ticket(self, server_name: str) -> None:
197 """Revoke ticket for server."""
198 async with self._lock:
199 ticket = self._tickets.get(server_name)
200 if ticket:
201 ticket.revoke()
202
203 async def clear_tickets(self) -> None:
204 """Clear all tickets."""
205 async with self._lock:
206 self._tickets.clear()
207
208 async def _cleanup_loop(self) -> None:
209 """Periodically cleanup expired tickets."""
210 while True:
211 try:
212 await asyncio.sleep(3600) # Check every hour
213
214 async with self._lock:
215 expired = [
216 server_name
217 for server_name, ticket in self._tickets.items()
218 if ticket.is_expired or ticket.status != SessionTicketStatus.VALID
219 ]
220
221 for server_name in expired:
222 del self._tickets[server_name]
223
224 if expired:
225 logger.debug(f"Cleaned up {len(expired)} expired tickets")
226
227 except asyncio.CancelledError:
228 break
229 except Exception as e:
230 logger.error(f"Session ticket cleanup error: {e}")
231
232 @property
233 def ticket_count(self) -> int:
234 """Get number of stored tickets."""
235 return len(self._tickets)

283 async def connect_with_0rtt(
284 self,
285 host: str,
286 port: int,
287 configuration: Any,
288 early_data: Optional[bytes] = None,
289 protocol_factory: Optional[Callable] = None,
290 ) -> Tuple[Any, bool]:
291 """
292 Establish QUIC connection with 0-RTT if possible.
293
294 Args:
295 host: Target host
296 port: Target port
297 configuration: QUIC configuration
298 early_data: Data to send in 0-RTT
299 protocol_factory: Protocol factory function
300
301 Returns:
302 Tuple of (connection, used_0rtt)
303 """
304 start_time = time.time()
305 used_0rtt = False
306
307 if not self._enable_0rtt:
308 # Regular connection
309 from aioquic.asyncio import connect
310 connection = await connect(
311 host,
312 port,
313 configuration=configuration,
314 create_protocol=protocol_factory,
315 )
316 self._record_handshake(time.time() - start_time, used_0rtt)
317 return connection, False
318
319 # Try to get session ticket
320 ticket = await self._ticket_store.get_ticket(host)
321
322 if ticket and early_data and len(early_data) <= self._max_early_data_size:
323 # 0-RTT connection with early data
324 try:
325 from aioquic.asyncio import connect
326
327 # Add session ticket to configuration
328 configuration.session_ticket = ticket.ticket_data
329
330 connection = await connect(
331 host,
332 port,
333 configuration=configuration,
334 create_protocol=protocol_factory,
335 )
336
337 # Send early data
338 if early_data and hasattr(connection, 'send_early_data'):
339 connection.send_early_data(early_data)
340 self._metrics.early_data_bytes_sent += len(early_data)
341
342 used_0rtt = True
343 logger.debug(f"Established 0-RTT connection to {host}:{port}")
344
345 except Exception as e:
346 logger.warning(f"0-RTT connection failed: {e}, falling back to regular")
347 # Fallback to regular connection
348 from aioquic.asyncio import connect
349 connection = await connect(
350 host,
351 port,
352 configuration=configuration,
353 create_protocol=protocol_factory,
354 )
355 else:
356 # Regular connection (no ticket or no early data)
357 from aioquic.asyncio import connect
358 connection = await connect(
359 host,
360 port,
361 configuration=configuration,
362 create_protocol=protocol_factory,
363 )
364
365 self._record_handshake(time.time() - start_time, used_0rtt)
366 return connection, used_0rtt

437class ConnectionMigration:
438 """
439 QUIC connection migration support.
440
441 Allows connections to survive network changes (e.g., WiFi to cellular).
442 """
443
444 def __init__(self, max_attempts: int = MAX_MIGRATION_ATTEMPTS):
445 self._max_attempts = max_attempts
446 self._active_migrations: Dict[str, asyncio.Task] = {}
447
448 async def migrate_connection(
449 self,
450 connection: Any,
451 new_address: Tuple[str, int],
452 timeout: float = MIGRATION_TIMEOUT,
453 ) -> bool:
454 """
455 Migrate connection to new address.
456
457 Args:
458 connection: QUIC connection to migrate
459 new_address: New (host, port) address
460 timeout: Migration timeout
461
462 Returns:
463 True if migration succeeded
464 """
465 connection_id = id(connection)
466
467 # Cancel any existing migration
468 if connection_id in self._active_migrations:
469 self._active_migrations[connection_id].cancel()
470
471 # Start migration
472 task = asyncio.create_task(
473 self._do_migration(connection, new_address, timeout)
474 )
475 self._active_migrations[connection_id] = task
476
477 try:
478 result = await asyncio.wait_for(task, timeout=timeout)
479 return result
480 except asyncio.TimeoutError:
481 logger.error(f"Connection migration to {new_address} timed out")
482 return False
483 finally:
484 self._active_migrations.pop(connection_id, None)
485
486 async def _do_migration(
487 self,
488 connection: Any,
489 new_address: Tuple[str, int],
490 timeout: float,
491 ) -> bool:
492 """Perform connection migration."""
493 host, port = new_address
494
495 for attempt in range(self._max_attempts):
496 try:
497 # Initiate path validation
498 if hasattr(connection, 'initiate_path_validation'):
499 await connection.initiate_path_validation(host, port)
500
501 # Wait for path to be validated
502 if hasattr(connection, 'wait_path_validated'):
503 await asyncio.wait_for(
504 connection.wait_path_validated(),
505 timeout=timeout / self._max_attempts
506 )
507
508 logger.debug(f"Connection migrated to {host}:{port}")
509 return True
510
511 except Exception as e:
512 logger.warning(
513 f"Migration attempt {attempt + 1} failed: {e}"
514 )
515 await asyncio.sleep(0.5)
516
517 return False
518
519 def cancel_migration(self, connection: Any) -> None:
520 """Cancel ongoing migration for connection."""
521 connection_id = id(connection)
522 task = self._active_migrations.pop(connection_id, None)
523 if task:
524 task.cancel()
