121async def example_message_encoding():
122 """Example: Encoding and decoding DCUtR messages"""
123 logger.info("=== DCUtR Message Encoding Example ===")
124
125 # Create a CONNECT message with observed addresses
126 connect_msg = DCUtRMessage(
127 message_type=DCUtRMessageType.CONNECT,
128 obs_addrs=[
129 b"/ip4/203.0.113.1/tcp/12345",
130 b"/ip4/203.0.113.1/udp/12346/quic",
131 ],
132 )
133
134 # Encode the message
135 encoded = connect_msg.encode()
136 logger.info(f"Encoded CONNECT message: {len(encoded)} bytes")
137 logger.info(f" Hex: {encoded.hex()}")
138
139 # Decode the message
140 length, offset = DCUtRMessage.decode_varint(encoded)
141 msg_data = encoded[offset:offset + length]
142 decoded = DCUtRMessage.decode(msg_data)
143
144 logger.info(f"Decoded message type: {decoded.message_type.name}")
145 logger.info(f"Decoded addresses: {[addr.decode() for addr in decoded.obs_addrs]}")
146
147 # Create a SYNC message
148 sync_msg = DCUtRMessage(message_type=DCUtRMessageType.SYNC)
149 sync_encoded = sync_msg.encode()
150 logger.info(f"Encoded SYNC message: {len(sync_encoded)} bytes")
