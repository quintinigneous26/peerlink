437class MessageCache:
438 """Message cache for gossip."""
439
440 def __init__(self, mcache_len: int = 5):
441 self.mcache_len = mcache_len
442 self.windows: List[Dict[bytes, PubSubMessage]] = [defaultdict(lambda: None)]
443 self.current = 0
444
445 def put(self, msg: PubSubMessage) -> None:
446 """Add message to current window."""
447 msg_id = msg.message_id()
448 self.windows[self.current][msg_id] = msg
449
450 def get(self, msg_id: bytes) -> Optional[PubSubMessage]:
451 """Get message from cache."""
452 for window in self.windows:
453 if msg_id in window:
454 return window[msg_id]
455 return None
456
457 def get_gossip_ids(self, topic: str, gossip_count: int) -> List[bytes]:
458 """Get message IDs for gossip from recent windows."""
459 ids = []
460 for i, window in enumerate(self.windows):
461 if i >= gossip_count:
462 break
463 for msg_id, msg in window.items():
464 if msg.topic == topic:
465 ids.append(msg_id)
466 return ids
467
468 def shift(self) -> None:
469 """Shift to next window."""
470 self.current = (self.current + 1) % self.mcache_len
471
472 # Clear the new current window and reinitialize
473 if len(self.windows) <= self.current:
474 self.windows.append(defaultdict(lambda: None))
475 else:
476 self.windows[self.current].clear()

479class SeenCache:
480 """LRU cache for seen message IDs."""
481
482 def __init__(self, ttl: float = 120.0):
483 self.ttl = ttl
484 self.cache: Dict[bytes, float] = {}
485
486 def seen(self, msg_id: bytes) -> bool:
487 """Check if message ID was seen."""
488 now = time.time()
489 # Clean expired entries
490 expired = [k for k, v in self.cache.items() if now - v > self.ttl]
491 for k in expired:
492 del self.cache[k]
493
494 if msg_id in self.cache:
495 return True
496
497 self.cache[msg_id] = now
498 return False
499
500 def has(self, msg_id: bytes) -> bool:
501 """Check if message ID is in cache without updating."""
502 return msg_id in self.cache

505class GossipSubRouter:
506 """GossipSub v1.1 router implementation."""
507
508 def __init__(self, peer_id: str, config: Optional[GossipSubConfig] = None):
509 self.peer_id = peer_id
510 self.config = config or GossipSubConfig()
511
512 # Topic subscriptions
513 self.my_topics: Set[str] = set()
514
515 # Peer tracking
516 self.peers: Set[str] = set() # All known pubsub peers
517 self.peer_topics: Dict[str, Set[str]] = defaultdict(set) # Topics each peer is subscribed to
518
519 # Mesh and fanout
520 self.mesh: Dict[str, Set[str]] = defaultdict(set) # Subscribed topics mesh
521 self.fanout: Dict[str, Set[str]] = defaultdict(set) # Unsubscribed topics
522 self.fanout_last_pub: Dict[str, float] = {} # Last publish time for fanout
523
524 # Message cache
525 self.mcache = MessageCache(self.config.mcache_len)
526 self.seen_cache = SeenCache(self.config.seen_ttl)
527
528 # Backoff state
529 self.backoff: Dict[Tuple[str, str], float] = {} # (topic, peer) -> expiry
530
531 # Peer scores (basic implementation)
532 self.peer_scores: Dict[str, float] = defaultdict(float)
533
534 # Sequence number for publishing
535 self.seqno: int = 0
536 self._seqno_lock = asyncio.Lock()
537
538 def add_peer(self, peer_id: str) -> None:
539 """Add a known pubsub peer."""
540 self.peers.add(peer_id)
541
542 def remove_peer(self, peer_id: str) -> None:
543 """Remove a peer."""
544 self.peers.discard(peer_id)
545 self.peer_topics.pop(peer_id, None)
546
547 # Remove from all meshes and fanouts
548 for topic in list(self.mesh.keys()):
549 self.mesh[topic].discard(peer_id)
550 for topic in list(self.fanout.keys()):
551 self.fanout[topic].discard(peer_id)
552
553 def handle_subscription(self, peer_id: str, sub: SubOpts) -> None:
554 """Handle subscription announcement from peer."""
555 if sub.subscribe:
556 self.peer_topics[peer_id].add(sub.topicid)
557 else:
558 self.peer_topics[peer_id].discard(sub.topicid)
559
560 def get_peers(self, topic: str, exclude: Optional[Set[str]] = None) -> Set[str]:
561 """Get peers subscribed to a topic."""
562 exclude = exclude or set()
563 return {
564 p for p in self.peers
565 if topic in self.peer_topics[p] and p not in exclude
566 }
567
568 async def join_topic(self, topic: str) -> None:
569 """Join a topic (subscribe)."""
570 if topic in self.my_topics:
571 return
572
573 self.my_topics.add(topic)
574
575 # Move from fanout if exists
576 if topic in self.fanout:
577 self.mesh[topic] = self.fanout.pop(topic)
578 self.fanout_last_pub.pop(topic, None)
579 else:
580 # Select D peers to graft
581 topic_peers = self.get_peers(topic)
582 to_graft = random.sample(list(topic_peers), min(self.config.D, len(topic_peers)))
583 self.mesh[topic] = set(to_graft)
584
585 async def leave_topic(self, topic: str) -> None:
586 """Leave a topic (unsubscribe)."""
587 if topic not in self.my_topics:
588 return
589
590 self.my_topics.discard(topic)
591 self.mesh.pop(topic, None)
592
593 def get_mesh_peers(self, topic: str) -> Set[str]:
594 """Get mesh peers for a topic."""
595 return self.mesh.get(topic, set())
596
597 def get_fanout_peers(self, topic: str) -> Set[str]:
598 """Get fanout peers for a topic."""
599 return self.fanout.get(topic, set())
600
601 async def publish(self, topic: str, data: bytes, from_peer: str) -> Tuple[List[str], List[str]]:
602 """Get target peers for publishing. Returns (mesh_targets, fanout_targets)."""
603 async with self._seqno_lock:
604 self.seqno += 1
605 seqno_bytes = struct.pack(">Q", self.seqno)
606
607 # Create message
608 msg = PubSubMessage(
609 from_peer=from_peer,
610 data=data,
611 seqno=seqno_bytes,
612 topic=topic,
613 )
614
615 # Add to cache
616 self.mcache.put(msg)
617 self.seen_cache.seen(msg.message_id())
618
619 # Get targets
620 mesh_targets: List[str] = []
621 fanout_targets: List[str] = []
622
623 if topic in self.my_topics:
624 # Subscribed: send to mesh
625 mesh_targets = list(self.mesh.get(topic, set()))
626 elif topic in self.fanout:
627 # Has fanout
628 fanout_targets = list(self.fanout[topic])
629 self.fanout_last_pub[topic] = time.time()
630 else:
631 # Create new fanout
632 topic_peers = self.get_peers(topic)
633 to_add = random.sample(list(topic_peers), min(self.config.D, len(topic_peers)))
634 self.fanout[topic] = set(to_add)
635 self.fanout_last_pub[topic] = time.time()
636 fanout_targets = to_add
637
638 return mesh_targets, fanout_targets
639
640 def handle_message(self, msg: PubSubMessage, from_peer: str) -> bool:
641 """Handle incoming message. Returns True if message should be processed."""
642 msg_id = msg.message_id()
643
644 # Check if seen
645 if self.seen_cache.seen(msg_id):
646 return False
647
648 # Add to cache
649 self.mcache.put(msg)
650
651 # Don't forward our own messages
652 if msg.from_peer == self.peer_id:
653 return False
654
655 return True
656
657 def get_forward_targets(self, topic: str, from_peer: str) -> List[str]:
658 """Get targets for forwarding a message."""
659 targets = set()
660
661 # Add mesh peers
662 if topic in self.mesh:
663 targets.update(self.mesh[topic])
664
665 # Remove source peer
666 targets.discard(from_peer)
667
668 return list(targets)

670 def heartbeat(self) -> ControlMessage:
671 """Perform heartbeat and return control messages to send."""
672 control = ControlMessage()
673
674 # Mesh maintenance
675 for topic, mesh_peers in list(self.mesh.items()):
676 # Check undersubscription
677 if len(mesh_peers) < self.config.D_low:
678 topic_peers = self.get_peers(topic, exclude=mesh_peers)
679 to_graft = random.sample(
680 list(topic_peers),
681 min(self.config.D - len(mesh_peers), len(topic_peers))
682 )
683 for peer in to_graft:
684 mesh_peers.add(peer)
685 control.graft.append(ControlGraft(topic_id=topic))
686
687 # Check oversubscription
688 elif len(mesh_peers) > self.config.D_high:
689 to_remove_count = len(mesh_peers) - self.config.D
690 to_remove = random.sample(list(mesh_peers), to_remove_count)
691 for peer in to_remove:
692 mesh_peers.remove(peer)
693 self._add_backoff(topic, peer)
694 prune = ControlPrune(topic_id=topic)
695 # Could add PX peers here
696 control.prune.append(prune)
697
698 # Fanout maintenance
699 now = time.time()
700 for topic, fanout_peers in list(self.fanout.items()):
701 # Check TTL
702 last_pub = self.fanout_last_pub.get(topic, 0)
703 if now - last_pub > self.config.fanout_ttl:
704 self.fanout.pop(topic, None)
705 self.fanout_last_pub.pop(topic, None)
706 continue
707
708 # Ensure minimum size
709 if len(fanout_peers) < self.config.D:
710 topic_peers = self.get_peers(topic, exclude=fanout_peers)
711 to_add = random.sample(
712 list(topic_peers),
713 min(self.config.D - len(fanout_peers), len(topic_peers))
714 )
715 fanout_peers.update(to_add)
716
717 # Gossip emission
718 all_topics = set(self.mesh.keys()) | set(self.fanout.keys())
719 for topic in all_topics:
720 mids = self.mcache.get_gossip_ids(topic, self.config.mcache_gossip)
721 if mids:
722 # Select D random peers for gossip
723 topic_peers = self.get_peers(topic)
724 mesh_peers = self.mesh.get(topic, set())
725 fanout_peers = self.fanout.get(topic, set())
726
727 gossip_targets = list(topic_peers - mesh_peers - fanout_peers)
728 gossip_targets = random.sample(
729 gossip_targets,
730 min(self.config.D, len(gossip_targets))
731 )
732
733 if gossip_targets:
