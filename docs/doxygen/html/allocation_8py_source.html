26class PortPool:
27 """
28 Manages a pool of available relay ports.
29
30 Uses a bitmap to track port availability for efficient allocation.
31 """
32
33 def __init__(self, min_port: int = DEFAULT_MIN_PORT, max_port: int = DEFAULT_MAX_PORT):
34 """
35 Initialize port pool.
36
37 Args:
38 min_port: Minimum port number
39 max_port: Maximum port number (inclusive)
40 """
41 self.min_port = min_port
42 self.max_port = max_port
43 self.total_ports = max_port - min_port + 1
44 self._available_ports: Set[int] = set(range(min_port, max_port + 1))
45 self._lock = asyncio.Lock()
46
47 async def acquire(self) -> Optional[int]:
48 """
49 Acquire a port from the pool.
50
51 Returns:
52 Port number or None if pool exhausted
53 """
54 async with self._lock:
55 if not self._available_ports:
56 return None
57 # Random selection for load distribution
58 port = random.choice(tuple(self._available_ports))
59 self._available_ports.remove(port)
60 logger.debug(f"Acquired port {port}, {len(self._available_ports)} remaining")
61 return port
62
63 async def release(self, port: int) -> bool:
64 """
65 Release a port back to the pool.
66
67 Args:
68 port: Port to release
69
70 Returns:
71 True if port was released, False if not in pool
72 """
73 async with self._lock:
74 if self.min_port <= port <= self.max_port:
75 if port not in self._available_ports:
76 self._available_ports.add(port)
77 logger.debug(f"Released port {port}, {len(self._available_ports)} available")
78 return True
79 return False
80
81 def available_count(self) -> int:
82 """Get number of available ports."""
83 return len(self._available_ports)
84
85 def usage_percentage(self) -> float:
86 """Get port pool usage as percentage."""
87 used = self.total_ports - len(self._available_ports)
88 return (used / self.total_ports) * 100

91class AllocationManager:
92 """
93 Manages TURN allocations.
94
95 Handles allocation creation, refresh, permission management, and cleanup.

98 def __init__(
99 self,
100 min_port: int = DEFAULT_MIN_PORT,
101 max_port: int = DEFAULT_MAX_PORT,
102 default_lifetime: int = DEFAULT_ALLOCATION_LIFETIME,
103 max_allocations: int = DEFAULT_MAX_ALLOCATIONS,
104 ):
105 """
106 Initialize allocation manager.
107
108 Args:
109 min_port: Minimum relay port
110 max_port: Maximum relay port
111 default_lifetime: Default allocation lifetime in seconds
112 max_allocations: Maximum concurrent allocations
113 """
114 self.port_pool = PortPool(min_port, max_port)
115 self.default_lifetime = default_lifetime
116 self.max_allocations = max_allocations
117 self._allocations: Dict[str, TurnAllocation] = {}
118 self._client_to_allocation: Dict[Tuple[str, int], str] = {}
119 self._relay_to_allocation: Dict[Tuple[str, int], str] = {}
120 self._lock = asyncio.Lock()
121 self._cleanup_task: Optional[asyncio.Task] = None
122
123 async def start(self):
124 """Start allocation manager and cleanup task."""
125 self._cleanup_task = asyncio.create_task(self._cleanup_loop())
126 logger.info("Allocation manager started")
127
128 async def stop(self):
129 """Stop allocation manager."""
130 if self._cleanup_task:
131 self._cleanup_task.cancel()
132 try:
133 await self._cleanup_task
134 except asyncio.CancelledError:
135 pass
136 logger.info("Allocation manager stopped")

138 async def _cleanup_loop(self):
139 """Periodically cleanup expired allocations."""
140 while True:
141 try:
142 await asyncio.sleep(60) # Check every minute
143 await self.cleanup_expired()
144 except asyncio.CancelledError:
145 break
146 except Exception as e:
147 logger.error(f"Error in cleanup loop: {e}")

149 async def create_allocation(
150 self,
151 client_addr: Tuple[str, int],
152 relay_addr: Tuple[str, int],
153 transport: str = "udp",
154 lifetime: Optional[int] = None,
155 ) -> Optional[TurnAllocation]:
156 """
157 Create a new allocation.
158
159 Args:
160 client_addr: Client's (ip, port)
161 relay_addr: Relay server's (ip, port)
162 transport: Transport protocol
163 lifetime: Requested lifetime (uses default if None)
164
165 Returns:
166 TurnAllocation or None if max allocations reached
167 """
168 async with self._lock:
169 # Check allocation limit
170 if len(self._allocations) >= self.max_allocations:
171 logger.warning(f"Max allocations ({self.max_allocations}) reached")
172 return None
173
174 # Check if client already has allocation
175 if client_addr in self._client_to_allocation:
176 logger.warning(f"Client {client_addr} already has allocation")
177 return None
178
179 # Allocate port
180 port = await self.port_pool.acquire()
181 if port is None:
182 logger.warning("No available ports for allocation")
183 return None
184
185 # Create allocation
186 allocation_id = str(uuid.uuid4())
187 lifetime = lifetime or self.default_lifetime
188
189 allocation = TurnAllocation(
190 allocation_id=allocation_id,
191 client_addr=client_addr,
192 relay_addr=(relay_addr[0], port),
193 lifetime=lifetime,
194 transport=transport,
195 )
196
197 # Store allocation
198 self._allocations[allocation_id] = allocation
199 self._client_to_allocation[client_addr] = allocation_id
200 self._relay_to_allocation[allocation.relay_addr] = allocation_id
201
202 logger.info(
203 f"Created allocation {allocation_id} for {client_addr} -> {allocation.relay_addr}"
204 )
205
206 return allocation

275 async def delete_allocation(self, allocation_id: str) -> bool:
276 """
277 Delete an allocation.
278
279 Args:
280 allocation_id: Allocation ID
281
282 Returns:
283 True if allocation was deleted
284 """
285 async with self._lock:
286 allocation = self._allocations.get(allocation_id)
287 if not allocation:
288 return False
289
290 # Release port
291 await self.port_pool.release(allocation.relay_addr[1])
292
293 # Remove from indexes
294 self._client_to_allocation.pop(allocation.client_addr, None)
295 self._relay_to_allocation.pop(allocation.relay_addr, None)
296 self._allocations.pop(allocation_id, None)
297
298 logger.info(f"Deleted allocation {allocation_id}")
299 return True

320 async def cleanup_expired(self) -> int:
321 """
322 Remove all expired allocations.
323
324 Returns:
325 Number of allocations removed
326 """
327 expired_ids = []
328
329 async with self._lock:
330 for allocation_id, allocation in self._allocations.items():
331 if allocation.is_expired():
332 expired_ids.append(allocation_id)
333
334 for allocation_id in expired_ids:
335 await self.delete_allocation(allocation_id)
336
337 if expired_ids:
338 logger.info(f"Cleaned up {len(expired_ids)} expired allocations")
339
340 return len(expired_ids)

342 def get_stats(self) -> dict:
343 """
344 Get allocation manager statistics.
345
346 Returns:
347 Statistics dictionary
348 """
349 active_allocations = [a for a in self._allocations.values() if not a.is_expired()]
350
351 total_bytes_sent = sum(a.bytes_sent for a in active_allocations)
352 total_bytes_received = sum(a.bytes_received for a in active_allocations)
353
354 return {
355 "total_allocations": len(self._allocations),
356 "active_allocations": len(active_allocations),
357 "max_allocations": self.max_allocations,
358 "port_pool_available": self.port_pool.available_count(),
359 "port_pool_usage": f"{self.port_pool.usage_percentage():.1f}%",
360 "total_bytes_sent": total_bytes_sent,
361 "total_bytes_received": total_bytes_received,
