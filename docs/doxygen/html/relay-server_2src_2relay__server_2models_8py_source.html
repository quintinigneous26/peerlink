91 """Bandwidth statistics."""
92
93 current_bps: float = 0.0 # Bytes per second
94 peak_bps: float = 0.0

99 def add_sample(self, bytes_count: int) -> None:
100 """Add a sample."""
101 import time
102 self._samples.append((time.time(), bytes_count))
103 self.total_bytes += bytes_count
104
105 # Clean old samples
106 cutoff = time.time() - self.window_size
107 self._samples = [s for s in self._samples if s[0] > cutoff]
108
109 def get_current_rate(self) -> float:
110 """Get current rate in bytes per second."""
111 import time
112 if not self._samples:
113 return 0.0
114
115 # Calculate rate from recent samples
116 total = sum(s[1] for s in self._samples)
117 if len(self._samples) < 2:
118 return 0.0
119
120 duration = self._samples[-1][0] - self._samples[0][0]
121 if duration > 0:
122 rate = total / duration
123 if rate > self.peak_bps:
124 self.peak_bps = rate
125 self.current_bps = rate
126 return rate
127
128 return 0.0
